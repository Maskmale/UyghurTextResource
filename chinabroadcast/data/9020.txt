# قىزىل مايكىلىقلار قوشۇنى داۋاملىق نامايىش ئۆتكۈزۈپ يىڭلاكنى
قوللايدىغانلىقىنى بىلدۈردى

2014-05-12 12:41:59     مەنبەسى: شىنجاڭ خەلق رادىيو ئىستانسىسى  

سىنگاپور «بىرلەشمە سەھەر گېزىتى»نىڭ تۈنۆگۈنكى خەۋىرىگە قارىغاندا، تايلاندىكى
ھۆكۈمەتنى قوللايدىغان قىزىل مايكىلىقلار قوشۇنىنىڭ نەچچە ئون مىڭ ئەزاسى مۇشۇ
ئاينىڭ 10-كۈنى باڭكوكنىڭ غەربىي شەھەر ئەتراپىدا كەڭ كۆلەملىك يىغىلىش ئۆتكۈزۈپ،
ھۆكۈمەتكە قارشى كۈچلەر قوزغىغان «سىياسىي ئۆزگىرىش» ۋە ئۇلارنىڭ خەلق سايلىمىغان
ھۆكۈمەتنى ھوقۇققا ئېرىشتۈرۈش غەرىزىگە قارشى تۇرىدىغانلىقىنى بىلدۈرگەن ھەم
بۇنىڭ ئىچكى ئۇرۇش پەيدا قىلىدىغانلىقى ھەققىدە ئاگاھلاندۇرغان. تايلاند يۇقىرى
پالاتاسى بۈگۈن يىغىن ئاچىدۇ، يىغىندا ۋاقىتلىق باش ۋەزىرنىڭ تەيىنلىنىش-
تەيىنلەنمەسلىكى دىققەت-ئېتىبارنى قوزغىماقتا.  
  قارانچۇق ھۆكۈمەتنىڭ ئەسلى زۇڭلىسى يىڭلاكنىڭ قوللىغۇچىلىرى يىڭلاكنىڭ ئاساسىي
قانۇن سوتى تەرىپىدىن «ئاساسىي قانۇنغا خىلاپلىق قىلدى» دەپ قارىلىپ، ۋەزىپىسىدىن
قالدۇرۇلغانلىقىنى «ئەدلىيە سىياسىي ئۆزگىرىشى» دەپ سۈپەتلىگەن. قىزىل
مايكىلىقلار قوشۇنىنىڭ ئاساسلىق رەھبىرى چادۇروڭ: «بىز دېموكراتىيەگە، ئاساسىي
قانۇنغا ئۇيغۇن بولمىغان باش ۋەزىرنى قوبۇل قىلالمايمىز. بىز تاكى دۆلىتىمىز
دېموكراتىيە يولىغا قايتىپ كەلگۈچە ئۈزلۈكسىز نامايىش قىلىمىز» دېگەن. ھۆكۈمەتكە
قارشى كۈچلەرنىڭ سودىيە، يۇقىرى پالاتانىڭ يېڭى باشلىقى ۋە باشقا ئالاقىدار
زاتلارنى بىر ۋاقىتلىق ئىچكى كابېنت تەسىس قىلىش ياكى ۋاقىتلىق باش ۋەزىر
تەيىنلەشكە كۈشكۈرتكەنلىكىگە قارىتا، چادۇروڭ: «بۇ مۇمكىن ئەمەس ھەم قانۇنسىز
قىلمىش. ئەگەر راست شۇنداق بولسا، ئۇ ئىچكى ئۇرۇشنىڭ پىلتىسىگە ئايلىنىپ،
تايلاندنى ئاپەتكە دۇچار قىلىدۇ» دېگەن. ئۇ يەنە، ئەگەر ۋاقىتلىق ئىچكى كابېنت
تەسىس قىلىنسا ياكى ۋاقىتلىق باش ۋەزىر تەيىنلەنسە، قىزىل مايكىلىقلار قوشۇنىنىڭ
نامايىشنى تېخىمۇ كۈچەيتىدىغانلىقىنى بىلدۈرگەن.  
  «مۇستەبىتلىككە قارشى دېموكراتىك ئىتتىپاق»، يەنى قىزىل مايكىلىقلار قوشۇنىنىڭ
باياناتچىسى تاناۋۇمۇ: «بۇ قېتىملىق كەڭ كۆلەملىك يىغىلىشنىڭ مەقسىتى ــــ خەلق
سايلىمىغان بىر كىشىنى باش ۋەزىرلىككە تەيىنلەش ئۇسۇلىنى ئېتىراپ
قىلمايدىغانلىقىمىزنى بىلدۈرۈشتىن ئىبارەت. بىز پەقەت دېموكراتىك ئۇسۇلدا
سايلىنىپ چىققان باش ۋەزىرنى ئېتىراپ قىلىمىز. ھۆكۈمەتكە قارشى نامايىشچىلارنىڭ
ئاخىرقى مەقسىتى مەيلى ئاۋازسىز سىياسىي ئۆزگىرىش ياكى ھەربىي سىياسىي ئۆزگىرىش
ئۇسۇلىنى قوللانسۇن، ئىشقىلىپ بىر مەيدان سىياسىي ئۆزگىرىش قوزغاش. ئەلۋەتتە بىز
بۇنداق قىلمىشقا ھەرگىز يول قويالمايمىز» دېگەن. قىزىل مايكىلىقلار قوشۇنىنىڭ
داھىيلىرىنىڭ بىرى بولغان كۋانساي بولسا، ئۆزلىرىنىڭ جەڭ قىلىشقا
تەييارلانغانلىقى، ئەمما زوراۋانلىققا ئەمەس، بەلكى ئاممىنى قوزغاشقا تايىنىپ،
دېموكراتىيەنى قولغا كەلتۈرۈشكە تىرىشىدىغانلىقىنى تەكىتلىگەن ھەم داۋاملىق
نامايىش قىلىپ، يىڭلاكنى قوللاش توغرىسىدا قەسەم بەرگەن.  
  ھۆكۈمەت ئامانلىق ساقلاش ئەترىتىدىكى ئەمەلدارنىڭ كۆرسىتىشىچە، مۇشۇ ئاينىڭ
10-كۈنىدىكى يىغىلىشقا قاتناشقان قىزىل مايكىلىقلار قوشۇنى ئەزالىرىنىڭ سانى
تەخمىنەن 50 مىڭ ئەتراپىدا بولۇپ، ساقچى تەرەپ يىغىلىش مەيدانى ئەتراپىغا 3
مىڭدىن ئارتۇق ساقچى كۈچىنى ئامانلىق ساقلاشقا ئەۋەتكەن. قىزىل مايكىلىقلار
قوشۇنى نامايىشنى ئۇدا بىر قانچە كۈن داۋاملاشتۇرۇشنى پىلانلىغان ھەم سايلام
ئىدارىسىدىن ئەسلى پىلان بويىچە بۇ يىل 7-ئايدا قايتىدىن چوڭ سايلام ئۆتكۈزۈشنى
تەلەپ قىلغان.  
  تايلاند ئاساسىي قانۇن سوت مەھكىمىسى مۇشۇ ئاينىڭ 7-كۈنى يىڭلاكنىڭ چوقۇم ئورۇن
بوشىتىشى كېرەكلىكى توغرىسىدا ھۆكۈم چىقارغان، شۇنىڭ بىلەن بىرگە يەنە ئىچكى
كابېنتنىڭ 9 نەپەر ئەزاسىنى ۋەزىپىسىدىن قالدۇرغان. ئىچكى كابېنت كەينىدىنلا
مۇئاۋىن باش ۋەزىر، قوشۇمچە سودا مىنىستىرى نىۋاتتۇمروڭنى قارانچۇق ھۆكۈمەت
زۇڭلىلىقىغا كۆرسەتكەن. تايلاند چىرىكلىككە قارشى تۇرۇش كومىتېتىمۇ 2-كۈنى
يىڭلاكنى گۈرۈچ سېتىۋېلىش سىياسىتىدە مەسئۇلىيەتسىزلىك قىلدى دەپ بېكىتكەن ھەم
يۇقىرى پالاتاغا يىڭلاكنى ئەيىبلەش تەكلىپى بەرگەن. چىرىكلىككە قارشى تۇرۇش
كومىتېتى يەنە يىڭلاك ئۈستىدىن سوتقا ئەرز سۇنۇشى مۇمكىن ئىكەن.  
  ھۆكۈمەتكە قارشى نامايىشچىلار پۇرسەتتىن پايدىلىنىپ، مۇشۇ ئاينىڭ 9-كۈنىدىن
باشلاپ قوللىغۇچىلىرىنى يىغىپ، كوچىلاردا نامايىش ئۆتكۈزۈش ھەم تېلىۋىزىيە
ئىستانسىسى ۋە ھۆكۈمەت ئورگانلىرىنى مۇھاسىرىگە ئېلىش ئارقىلىق، تاكسىن
ھاكىمىيىتىنىڭ قالدۇقلىرىنى تەلتۆكۈس تازىلاش ۋە قارانچۇق ھۆكۈمەتنىڭ ئورنىغا
ۋاقىتلىق ھۆكۈمەت تەسىس قىلىشنى تەلەپ قىلغان. ھۆكۈمەتكە قارشى نامايىشچىلارنىڭ
رەھبىرى سۇتېپ مۇشۇ ئاينىڭ 10-كۈنى، يۇقىرى پالاتا قاتارلىق ئالاقىدار
دائىرىلەرنىڭ دەرھال ۋاقىتلىق باش ۋەزىر تەيىنلىشىنى مۇراجىئەت قىلغان ھەم
نىۋاتتۇمروڭنى «قانۇنسىز» قارانچۇق زۇڭلى دەپ ئەيىبلىگەن. سۇتېپ: «تاكىسننىڭ
قورچىقى يىڭلاك نومۇس قىلماي يەنە يېڭى قارانچۇق زۇڭلىنى تەيىنلىدى، ئۇلارنىڭ
قىلمىشى قانۇنغا خىلاپ. بۇ قارانچۇق ھۆكۈمەت قانۇنغا ئۇيغۇن ئەمەس. مۇنداقچە
ئېيتقاندا، ھازىر تايلاندتا دۆلەتنى باشقۇرغۇدەك بىر ھەقىقىي ھۆكۈمەتمۇ يوق»
دېگەن. ئۇ يەنە قوللىغۇچىلىرىغا: «بىزگە غەلىبىگە ئېرىشەيلا دەپ قالدۇق، شۇڭا
ھەرگىز چېكىنمەيمىز» دېگەن.  
  تايلاند يۇقىرى پالاتاسى مۇشۇ ئاينىڭ 9-كۈنى، ھۆكۈمەتكە قارشى سەپتىكى
سۇراساينى يېڭى پارلامېنت باشلىقلىقىغا كۆرسەتكەن، ئۇ بۈگۈن يۇقىرى پالاتا يىغىنى
ئاچماقچى. بۇ قېتىملىق يىغىندا ۋاقىتلىق باش ۋەزىرنىڭ تەيىنلىنىش-تەيىنلەنمەسلىكى
كۈچلۈك دىققەت-ئېتىبارنى قوزغىماقتا. تۆۋەن پالاتا سايلىمى ئىزچىل قايتا
ئۆتكۈزۈلمىگەنلىكتىن، يېڭى ھۆكۈمەت ۋە پارلامېنتنى قۇرۇپ چىقىشمۇ مۇمكىن
بولمىغان. شۇڭا يۇقىرى پالاتا نۆۋەتتە تايلاندىكى بىردىنبىر قانۇن چىقىرىش
ھوقۇقىنى يۈرگۈزەلەيدىغان ئاپپارات بولۇپ، تەسىر كۈچى ئىنتايىن زور.  
  سىياسىي ۋەزىيەتتىكى قاتماللىق ياخشىلانمىغاننىڭ ئۈستىگە، ھۆكۈمەتنى
قوللايدىغان ۋە قارشى تۇرىدىغان كۈچلەر ئاممىنى نامايىش ئۆتكۈزۈشكە سەپەرۋەر
قىلغانلىقتىن، كىشىلەر بىر مەيدان زوراۋانلىق توقۇنۇشىنىڭ قاش بىلەن كىرپىك
ئارىلىقىدا قالغانلىقىدىن ئەندىشە قىلىشماقتا. چياڭماي ئۇنۋېرسىتېتى
سىياسەتشۇناسلىق تەتقىقاتخانىسىنىڭ مۇدىرى پاۋلنىڭ  بىلدۈرۈشىچە، تايلاند
ۋەزىيىتىنىڭ نۆۋەتتىكى تەرەققىياتى ناھايىتى تېزلا ھەربىي سىياسىي ئۆزگىرىشكە
ئايلىنىشى مۇمكىن ئىكەن. بىراق تايلاند قۇرۇقلۇق ئارمىيەسىنىڭ قوماندانى بايو
مۇشۇ ئاينىڭ 10-كۈنى، ھەربىي سىياسىي ئۆزگىرىشنىڭ ھۆكۈمەتكە مايىل ۋە ھۆكۈمەتكە
قارشى ئىككى تەرەپ ئارىسىدىكى توقۇنۇشنى ھەل قىلالمايلا قالماستىن، ئەكسىچە
تېخىمۇ كۆپ ئەيىبلەشكە ئۇچرايدىغانلىقىنى قايتا تەكىتلىگەن. ئۇ: «سىياسىي
ۋەزىيەتتىكى قاتماللىقنى چوقۇم قانۇنىي يول بىلەن ھەل قىلىش كېرەك» دېگەن. يېڭى
قارانچۇق زۇڭلى نىۋاتتۇمروڭمۇ: «تايلاند پەقەت دېموكراتىيە يولىدا ماڭغاندىلا،
نۆۋەتتىكى مەسىلىنى تىنچ ھەل قىلىشتا ئۈمىد تۇغۇلىدۇ. ھازىرقى ئىچكى كابېنت تاكى
چوڭ سايلامدا يېڭى ھۆكۈمەت سايلىنىپ چىققۇچە، مەجبۇرىيىتىنى داۋاملىق ئادا
قىلىدۇ» دېگەن.

ئىستېلى قىزىق لېنىيىسى پروگراممىسىنىڭ «قاتناش 10 مىڭ چاقىرىملىق سەپىرى چوڭ

ۋىلايىتىدە  ئۆتكۈزىلىدۇ

ئىستانسىمىز ئاپتونوم رايونلۇق خەلق ھۆكۈمىتى كەسپتىكى ناتوغرا ئىستېللارنى
