# تەرەققىياتقا يېڭىلىق يارىتىش ئارقىلىق يېتەكچىلىك قىلىپ، كەلگۈسىنى كۆزلەيلى

2015-03-09 13:08:24     مەنبەسى: شىنجاڭ خەلق رادىيو ئىستانسىسى  

　دەۋر جۇڭگونىڭ تەرەققىي قىلىپ روناق تېپىشىغا تارىخىي پۇرسەت ئاتا قىلدى،
شۇنداقلا دادىللىق، ماھىرلىق بىلەن يېڭىلىق ياراتقۇچىلار ئۈچۈن كەڭ تارىخىي سەھنە
ھازىرلاپ بەردى. جۇڭگوچە سوتسىيالىزم ئىشلىرى مىسلىسىز يېڭى، ئۇلۇغۋار ئىش، ‹‹تۆت
ئومۇميۈزلۈك›› ئىستراتېگىيەلىك ئورۇنلاشتۇرمىنى ماس ئىلگىرى سۈرۈش مەۋقەنى
ھازىرغا قويۇپ، كەلگۈسىنى كۆزلەيدىغان پۈتۈنلەي يېڭى تەرەققىيات باسقۇچىدۇر.　　باش
شۇجى شى جىنپىڭ مەملىكەتلىك خەلق قۇرۇلتىيىنىڭ يىغىنىغا قاتنىشىۋاتقان شاڭخەي
ۋەكىللەر ئۆمىكىنىڭ قاراپ چىقىشىغا قاتناشقاندا مۇنۇلارنى تەكىتلىدى: ‹‹يېڭىلىق
يارىتىش تەرەققىياتقا يېتەكچىلىك قىلىدىغان بىرىنچى ھەرىكەتلەندۈرگۈچ كۈچ.
يېڭىلىق يارىتىشنى تۇتقانلىق تەرەققىياتنى تۇتقانلىق، يېڭىلىق يارىتىشنى
كۆزلىگەنلىك كەلگۈسىنى كۆزلىگەنلىك››. يېڭىلىق ياراتقاندىلا، تەرەققىياتنىڭ
تەشەببۇسكارلىق ھوقۇقىنى پۇختا ئىگىلەپ، قىممەتلىك ئىستراتېگىيەلىك پۇرسەت
مەزگىلىنى چىڭ تۇتۇپ ۋە ياخشى پايدىلىنىپ، ئىسلاھاتنى ئومۇميۈزلۈك
چوڭقۇرلاشتۇرۇشنىڭ ئاچقۇچلۇق يىلى، دۆلەتنى قانۇن بويىچە ئىدارە قىلىشنىڭ
باشلىنىش يىلى، 12 – بەش يىللىق يىرىك پىلاننى ئومۇميۈزلۈك تاماملاشنىڭ ئاچقۇچلۇق
يىلىدا، يېڭىلىق يارىتىش ۋە تەرەققىياتتا يېڭى نەتىجىلەرنى قولغا كەلتۈرۈپ،
ئىسلاھات، تەرەققىياتنىڭ يېڭىچە سىنىقىدىن ئۆتكىلى بولىدۇ.　　ئۆزگىرىپ تۇرۇۋاتقان
مۇرەككەپ خەلقئارا ۋەزىيەت ۋە دۆلەت ئىچىدىكى ئىسلاھات، تەرەققىيات، مۇقىملىقتىكى
مۈشكۈل ۋەزىپە توغرىسىدا، يولداش شى جىنپىڭ باش شۇجىلىقىدىكى پارتىيە مەركىزىي
كومىتېتى پۈتۈن مەملىكەتتىكى ھەر مىللەت خەلقىنى بىر نىيەت، بىر مەقسەتتە بولۇپ،
تىرىشىپ كۈرەش قىلىشقا ئىتتىپاقلاشتۇرۇپ ۋە باشلاپ، تىرىشىپ كۈرەش قىلىپ،
مۇقىملىق ئاساسىدا ئىلگىرىلەشتەك خىزمەت ئاساسىي رىتىمىدا چىڭ تۇرۇپ، يېڭىلىق
يارىتىش ئارقىلىق تەرەققىياتقا تۈرتكە بولۇش ئىستراتېگىيەسىنى يولغا قويۇپ،
ئۇسۇلنى ئۆزگەرتىپ، قۇرۇلمىنى تەڭشەپ، تەرەققىياتنى ئىلگىرى سۈرۈپ، ئەلگە نەپ
يەتكۈزۈپ، خەلقئارا ۋە دۆلەت ئىچىدىكى كۆپ خىل خىرىسلارغا ئۈنۈملۈك تاقابىل
تۇرۇپ، پارتىيە ۋە دۆلەت ئىشلىرىدا تېز ئىلگىرىلەشلەرنى قولغا كەلتۈرۈپ، ھاياتىي
كۈچنى نامايان قىلدى.　　يېڭىلىق يارىتىش مىللەتنىڭ تەرەققىي قىلىشىنىڭ روھى،
دۆلەتنىڭ گۈللىنىشى، راۋاجلىنىشىدىكى قۇدرەتلىك ھەرىكەتلەندۈرگۈچ كۈچ، شۇنداقلا
جۇڭخۇا مىللەتلىرىنىڭ ئەڭ كۈچلۈك تۇغما تالانتى. يېڭىلىق يارىتىش بولمىسا،
تەرەققىياتمۇ بولمايدۇ، تۈنۈگۈنكى پارلاق نەتىجىلەرمۇ بولمايدۇ، ئەتىكى گۈزەل
كەلگۈسىمۇ بولمايدۇ. تەرەققىياتتا چىڭ تۇرۇش چىڭ قائىدە، تەپەككۇردا يېڭىلىق
يارىتىش ئارقىلىق تەرەققىياتنى پىلانلاپ، يېڭىلىق يارىتىش ئەمەلىيىتى ئارقىلىق
تەرەققىياتنى ئىلگىرى سۈرۈپ، تەرەققىياتقا تۇسقۇنلۇق قىلىدىغان بارلىق ئىدىيەۋى
قاراش ئاسارەتلىرى ۋە تۈزۈلمە – مېخانىزمدىكى تۇسالغۇلارنى تۈگىتىپ، مىليونلىغان
خەلقنىڭ يېڭىلىق يارىتىشتىكى ئەقىل – پاراسىتىنى تولۇق قېزىپ، يېڭىلىق يارىتىش
كۈچىنى تولۇق ئۇرغۇتۇپ، يېڭىلىق يارىتىش ئەمەلىيىتىنى تولۇق ئەۋج ئالدۇرۇپ،
كۆپچىلىكنىڭ ئىگىلىك تىكلەش، يېڭىلىق يارىتىش نەتىجىلىرىنى تېخىمۇ تېز، تېخىمۇ
ئۈنۈملۈك ھالدا ئەمەلىي ئىشلەپچىقىرىش كۈچلىرىگە ئايلاندۇرۇپ، جۇڭگونىڭ قەد
كۆتۈرۈشى ئۈچۈن ئۈزلۈكسىز كۈچلۈك ھەرىكەتلەندۈرگۈچ كۈچ قوشۇپ، ئۇنىۋېرسال دۆلەت
كۈچى رىقابىتى دولقۇنىدا قەيسەرلىك بىلەن، پۇختا ۋە سالماق ئىلگىرىلەش كېرەك.
ئىسلاھاتنىڭ ئاخىرقى مەقسىتى خەلقنى بەخت – سائادەتكە ئېرىشتۈرۈشتۇر، ئاممىنى
تېخىمۇ كۆپ ئېرىشىش تۇيغۇسىغا ئىگە قىلىشتۇر. يېڭىلىق يارىتىپ، ئىسلاھاتنى
چوڭقۇرلاشتۇرۇشنى چۆرىدەپ، ئىسلاھات ئارقىلىق يېڭىلىق يارىتىشنى ئىلگىرى سۈرۈپ،
خەلقنىڭ تۈپ مەنپەئەتىنى تۈپ كوردىنات نۇقتىسى قىلىش ئارقىلىق، ئىشنى ئامما ئەڭ
كۆڭۈل بۆلىدىغان، ئەڭ بىۋاسىتە، ئەڭ رېئال مەنپەئەت مەسىلىلىرىدىن باشلاپ، خەلقنى
ئىسلاھات، يېڭىلىق يارىتىشتا ئۈزلۈكسىز تېخىمۇ كۆپ نەپكە ئېرىشتۈرۈش كېرەك.
(شىنخۇا ئاگېنتلىقى، بېيجىڭ، 5 – مارت تېلېگراممىسى)

تور بېتىمىزدىكى يېڭىلىقلار

 ئىستانسىمىز ئاپتونوم رايونلۇق خەلق ھۆكۈمىتى كەسپتىكى ناتوغرا ئىستېللارنى
تۈزىتىش ئىشخانىسى، ئاپتونوم رايونلۇق قاتناش ـ ترانسپورت نازارىتى، ئاپتونوم
رايونلۇق جامائەت خەۋپسىزلىكى نازارىتى بىلەن بىرلىشىپ ئۆتكۈزىدىغان، كەسپ
