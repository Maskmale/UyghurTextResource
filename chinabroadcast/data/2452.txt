# دىن ئەسەبىيلىكىنى يوقىتىشنى دىن خىزمىتىنىڭ ئەڭ مۇھىم ۋەزىپىسى قىلايلى

2014-07-04 12:22:25     مەنبەسى: شىنجاڭ خەلق رادىيو ئىستانسىسى  

 ج ك پ ئاپتونوم رايونلۇق 8 - نۆۋەتلىك كومىتېتى 7 - ئومۇمىي (كېڭەيتىلگەن)
يىغىنىدا يولداش جاڭ چۈنشيەن مۇنداق دەپ كۆرسەتتى: شىنجاڭنىڭ ھازىرقى دىن
خىزمىتىنى ياخشى ئىشلەشتە، دىن ئەسەبىيلىكىنى يوقىتىشنى چوڭقۇر ئىلگىرى سۈرۈشنى
گەۋدىلىك ئورۇنغا قويۇش كېرەك. بۇ، دىن ئەسەبىيلىكى ئىدىيەسىنىڭ شىنجاڭغا سىڭىپ
كىرىشى ۋە يامرىشى بىلەن بارلىققا كەلگەن مۇرەككەپ ۋەزىيەت، ئىجتىمائىي مۇقىملىق
ۋە ئەبەدىي ئەمىنلىك خىزمىتىنى تېخىمۇ يۇقىرى قاتلامدا تۇرۇپ پىلانلاپ،
مەسىلىلەرنى بايقاش ئېڭىنى تۇرغۇزۇپ، مەسىلىلەرنى ھەل قىلىش يۆنىلىشىدە چىڭ
تۇرۇپ، مەسىلىلەرنى ھەل قىلىش ئارقىلىق خىزمەتلەرنى ئىلگىرى سۈرۈش، تەشەببۇسكار
بولۇش ئۈچۈن قوللىنىلغان ئۈنۈملۈك ۋاستە، شۇنداقلا زوراۋان - تېررورچىلارغا
قاتتىق زەربە بېرىپ، ماھىيەتتىن تۈزەش چارىسى.  
    دىن ئەسەبىيلىكى «ئۈچ خىل كۈچ» نىڭ ئىدىيە ئاساسى ۋە تۈپ مەنبەسى، بىز دىن ئەسەبىيلىكى ئىدىيەسىنىڭ سىڭىپ كىرىشتىكى تۈپ مەقسىتى ۋە جىددىي ۋەزىيىتىنى پۇختا ئىگىلەپ، دىن ئەسەبىيلىكىنى يوقىتىشنى دىن خىزمىتىنى ياخشى ئىشلەشنىڭ مۇھىم ۋەزىپىسى قىلىشىمىز كېرەك. دىن ئەسەبىيلىرى ئۆزلىرىنىڭ سىياسىي مەقسىتىگە يېتىش ئۈچۈن، ئىسلام دىنىغا ھۆرمەت قىلماي، دىن تونىغا ئورىنىۋېلىپ، دىن ئەسەبىيلىكىگە قۇترىتىپ، ئۈزلۈكسىز ۋەقە تۇغدۇرۇپ، ۋەتەننىڭ بىرلىكى ۋە ئىجتىمائىي مۇقىملىققا ئىغۋاگەرچىلىك سالدى. دىن ئەسەبىيلىرى ئىدېئولوگىيە ساھەسى، ئىقتىسادىي تۈزۈم، قانۇن - نىزاملار، مەدەنىيەت - سەنئەت، ئەدەپ - ئەخلاق، ئۆرپ - ئادەت قاتارلىق جەھەتلەردە دىنغا ئېتىقاد قىلىدىغان ئاممىنىڭ ئىدىيەسىنى ھەر تەرەپتىن چىرىتىپ، ئۇلارنى قۇترىتىپ ۋە ئازدۇرۇپ، دىن ئىناقلىقىغا ئەڭ زور تەھدىت سالدى. جەمئىيەتنىڭ ئىناقلىقىغا ئېغىر تەسىر يەتكۈزىدىغان بۇ «يامان سۈپەتلىك ئۆسمە» نى يىلتىزىدىن يوقىتىشتا، دىن ئەسەبىيلىكى ئىدىيەسىنى نورمال دىندىن پەرقلەندۈرۈپ، دىن ئەسەبىيلىكى ئىدىيەسىنى قەتئىي توسۇش ۋە زەربە بېرىش كېرەك.  
    دىنىي ئىشلارنى قانۇن بويىچە باشقۇرۇپ، «ئۈچ قانۇنسىزلىق» نى قاتتىق تۈزەش دىن ئەسەبىيلىكى ئىدىيەسىنىڭ پەيدا بولۇشىغا بوشلۇق قالدۇرماسلىق ۋە تارقىلىشىنى مەنبەسىدىن ئۈزۈشتىكى مۇقەررەر ۋاسىتە. دىن ئەسەبىيلىكىنى يوقىتىش خىزمىتىنى ياخشى ئىشلەشتە، چوقۇم مەنبەدىن باشلاپ چىڭ تۇتۇش، دىننى چەكسىز، تەرتىپسىز ۋە ئەسەبىيلىشىشتەك ناتوغرا يولغا باشلىماسلىق كېرەك. زوراۋانلىق، تېررورلۇق ۋە دىن ئەسەبىيلىكى ئىدىيەسىنى تارقاتقان ۋە ئۇنىڭغا قۇتراتقۇلۇق قىلغان قانۇنسىز جىنايەتچىلەرنى قاتتىق جازالاپ، پەردە ئارقىسىدىكى تەشكىللىگۈچىلەرنى بىتچىت قىلىش؛ دىنىي پائالىيەت بىلەن مىللىي ئۆرپ - ئادەت، دىنغا نورمال ئېتىقاد قىلىش بىلەن دىن ئەسەبىيلىكىنىڭ چېگراسىنى ئېنىق ئايرىپ، دىنىي ئىشلارنى قانۇن بويىچە باشقۇرۇپ، دىننىڭ مەمۇرىي ئىشلار، ئەدلىيە، مائارىپ، توي - تۆكۈن ۋە پىلانلىق تۇغۇتقا ئارىلىشىشى، نورمال ئىجتىمائىي تەرتىپ، خىزمەت تەرتىپى ۋە تۇرمۇش تەرتىپىگە توسقۇنلۇق قىلىشى، دىن ئەسەبىيلىرىنىڭ دىندىن خالىي تۇرمۇشقا ئارىلىشىشىنى قاتتىق چەكلەش كېرەك.  
    زامانىۋى مەدەنىيەتنى يېتەكچى قىلىپ، دىننى سوتسىيالىستىك جەمئىيەتكە ماسلىشىشقا پائال يېتەكلەش ئىدېئولوگىيە ساھەسىدە دىن ئەسەبىيلىكىنى يوقىتىشتىكى ئىستراتېگىيەلىك ئورۇنلاشتۇرۇش. يەكلەش دىن ئەسەبىيلىكى ئىدىيەسىنىڭ ئەڭ كۆرۈنەرلىك ئالاھىدىلىكى، دىن ئەسەبىيلىرى باشقا مىللەت بىلەن دۈشمەنلىشىپلا قالماي، يەنە ئۆز مىللىتى بىلەنمۇ دۈشمەنلىشىدۇ. دەۋر بىلەن تەڭ ئىلگىرىلەۋاتقان زامانىۋى مەدەنىيەت روشەن ئىلمىيلىكى، ئېچىۋېتىلگەنلىكى، خەلقچىللىقى كۈچلۈك بولغان قۇدرەتلىك ھاياتىي كۈچكە ئىگە بولغاچقا، شۈبھىسىز ھالدا نادانلىق، يەكلەش ۋە ئازدۇرۇش بىلەن دىن ئەسەبىيلىكى ئىدىيەسىنى چەكلەيدىغان قۇدرەتلىك قورال ھېسابلىنىدۇ. بىز بۇ قۇدرەتلىك ئىدىيەۋى قورال ئارقىلىق سوتسىيالىستىك يادرولۇق قىممەت قارىشىنى كۈچەپ ئەۋج ئالدۇرۇپ، «تۆتنى تونۇش» تەربىيەسىنى كۈچەيتىپ، زاۋاللىققا يۈزتۇتقان بۇ دىن ئەسەبىيلىكى ئىدىيەسىنى تارىخنىڭ ئەخلەت دۆۋىسىگە سۈپۈرۈپ تاشلىشىمىز كېرەك.  
    دىن ئەسەبىيلىرىنىڭ چېكىدىن ئاشقان يەكلەشچانلىقى ئۇنىڭ ھاياتىي كۈچىنىڭ يوقلۇقىنى بەلگىلىگەن، بىراق ئۇلار ئۆزلۈكىدىن يوقالمايدۇ. بىز ئۇنىۋېرسال تەدبىرلەرنى قوللىنىشتا چىڭ تۇرۇپ، دىنغا ئېتىقاد قىلىدىغان ئاممىنىڭ قانۇنىي دىن ئېھتىياجىغا كاپالەتلىك قىلىپ، دىن ئەسەبىيلىكىنىڭ ھەرىكەت بوشلۇقىنى كىچىكلىتىشىمىز كېرەك. دىنىي تەشكىلاتلارنىڭ رەھبەرلىك ھوقۇقىنىڭ ۋەتەنپەرۋەر دىنىي زاتلارنىڭ قولىدا بولۇشىغا كاپالەتلىك قىلىپ، ھەر دەرىجىلىك ئىسلام دىنى جەمئىيەتلىرىنىڭ دىنغا ئېتىقاد قىلىدىغان ئامما بىلەن ئالاقە باغلاش ۋە ئۇلارنى يېتەكلەش، ئىسلام دىنىنى سوتسىيالىستىك جەمئىيەتكە ماسلاشتۇرۇشتىكى رولىنى پائال جارى قىلدۇرۇشىمىز؛ تەرتىپلىك، مۇۋاپىق ۋە تىزگىنلىگىلى بولۇش پىرىنسىپى بويىچە، دىنغا ئېتىقاد قىلىدىغان ئاممىنىڭ دىنىي بىلىملەر ۋە دىنىي مۇراسىملارنى توغرا ئىگىلەيدىغان قانۇنىي يوللىرىنى ماس قەدەمدە كېڭەيتىپ، ئۇلارنى دىنغا توغرا ئېتىقاد قىلىش ۋە دىن ئەسەبىيلىكىنى يوقىتىشقا يېتەكلىشىمىز؛ ئاپتونوم رايونىمىزدىكى ئالىي مەكتەپ ۋە تېخنىكوملاردا پارتىيەنىڭ دىن خىزمىتى توغرىسىدىكى ئاساسىي فاڭجېنى، دىن ساۋاتلىرى دەرسىنى سىناق تەرىقىسىدە ئۆتۈپ، ئوقۇغۇچىلارنى دىننى توغرا تونۇشقا يېتەكلىشىمىز كېرەك.  
    دىن ئەسەبىيلىكىنى يوقىتىشنى چوڭقۇر ئىلگىرى سۈرۈش، جۇڭگوچە سوتسىيالىزمنى راۋاجلاندۇرۇش ۋە ئۇنىڭدا چىڭ تۇرۇشقا، مىللەتلەر ئىتتىپاقلىقى ۋە دىنلارنىڭ ئىناقلىقىنى ئىلگىرى سۈرۈشكە، ھەر مىللەت ئاممىنى پارتىيە ۋە ھۆكۈمەتنىڭ ئەتراپىغا ئۇيۇشتۇرۇشقا پايدىلىق، شۇنداقلا «ئۈچ خىل كۈچ» نىڭ ئىدىيە مەنبەسىنى يوقىتىپ، بۆلگۈنچىلىككە قارشى كۈرەشنىڭ مۇقەررەر غەلىبىسىنى قولغا كەلتۈرۈشنىڭ مۇھىم تەدبىرى. بىز چوقۇم بۇنى تولۇق ئىزچىللاشتۇرۇپ، ئاپتونوم رايونىمىزنىڭ ئەبەدىي ئەمىنلىكىدىكى چوڭقۇر قاتلاملىق مەسىلىلەرنى تۈپتىن ھەل قىلىشقا پۇختا ئاساس سېلىشىمىز كېرەك.
