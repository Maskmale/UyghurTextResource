# يۈ جىڭشىڭ مەركەز ۋەكىللەر ئۆمىكىنىڭ بىر قىسىم ئەزالىرىنى باشلاپ، شىنجاڭ
ھەربىي رايونى، شىنجاڭ ئىشلەپچىقىرىش- قۇرۇلۇش بىڭتۈەنى، قوراللىق ساقچى قىسىم
شىنجاڭ باش ئەترىتىگە بېرىپ ھال سورىدى

2015-10-01 12:01:23     مەنبەسى: شىنجاڭ خەلق رادىيو ئىستانسىسى  

ئىستانسىمىز خەۋىرى:  
پارتىيە مەركىزىي كومىتېتى سىياسىي بيۇرۇسى دائىمىي كومىتېتىنىڭ ئەزاسى،
مەملىكەتلىك سىياسىي كېڭەشنىڭ رەئىسى، مەركەز ۋەكىللەر ئۆمىكىنىڭ باشلىقى يۈ
جىڭشىڭ 30-سىنتەبىر چۈشتىن بۇرۇن مەركەز ۋەكىللەر ئۆمىكىنىڭ بىر قىسىم ئەزالىرىنى
باشلاپ، شىنجاڭ ھەربىي رايونى، شىنجاڭ ئىشلەپچىقىرىش- قۇرۇلۇش بىڭتۈەنى، قوراللىق
ساقچى قىسىم شىنجاڭ باش ئەترىتىگە بېرىپ ھال سورىدى ھەمدە ئاپتونوم رايونىمىزنىڭ
سىياسىي- قانۇن سىستېمىسىدىكى ۋەكىللەر بىلەن كۆرۈشۈپ، پارتىيە مەركىزىي
كومىتېتى، گوۋۇيۈەن، مەركىزىي ھەربىي كومىتېت شۇنىڭدەك باش شۇجى شى جىنپىڭغا
ۋاكالىتەن، كۆپچىلىككە يۈكسەك ئىھتىرام بىلدۈردى ھەمدە سەمىمىي سالام يوللىدى.  
يۈ جىڭشىڭ ۋە مەركەز ۋەكىللەر ئۆمىكىنىڭ مۇئاۋىن باشلىقى بولغان پارتىيە مەركىزىي
كومىتېتى سىياسىي بيۇرۇسىنىڭ ئەزاسى، گوۋۇيۈەننىڭ مۇئاۋىن زۇڭلىسى ليۇ يەندۇڭ،
مەملىكەتلىك خەلق قۇرۇلتىيى دائىمىي كومىتېتىنىڭ مۇئاۋىن باشلىقى ئەركىن
ئىمىرباقى، مەملىكەتلىك سىياسىي كېڭەشنىڭ مۇئاۋىن رەئىسى جاڭ چىڭلى، مەملىكەتلىك
سىياسىي كېڭەشنىڭ مۇئاۋىن رەئىسى، دۆلەت مىللەتلەر ئىشلىرى كومىتېتىنىڭ مۇدىرى
ۋاڭ جىڭۋىي، مەركىزىي ھەربىي كومىتېتنىڭ ئەزاسى، باش ئارقا سەپ تەمىنات بۆلۈمىنىڭ
باشلىقى جاۋ كېشى شۇنىڭدەك مەركەز ۋەكىللەر ئۆمىكىنىڭ بىر قىسىم ئەزالىرى پارتىيە
مەركىزى كومىتېتى سىياسىي بيۇرۇسىنىڭ ئەزاسى، ئاپتونوم رايونلۇق پارتكومنىڭ
شۇجىسى، شىنجاڭ ھەربىي رايونى پارتكومىنىڭ 1- شۇجىسى، شىنجاڭ ئىشلەپچىقىرىش-
قۇرۇلۇش بىڭتۈەنى پارتكومىنىڭ 1- شۇجىسى، 1- سىياسىي كومىسسارى جاڭ چۈنشيەن،
ئاپتونوم رايونلۇق پارتكومنىڭ مۇئاۋىن شۇجىسى، ئاپتونوم رايوننىڭ رەئىسى شۆھرەت
زاكىرنىڭ ھەمراھلىقىدا، شىنجاڭ ھەربىي رايونىغا بېرىپ، شىنجاڭدا تۇرۇشلۇق
قىسىملاردىكى كوماندىر- جەڭچىلەردىن ھال سورىدى.  
شىنجاڭ ھەربىي رايونىدا، ئۈرۈمچىدىكى دىۋىزىيە دەرىجىلىكتىن يۇقىرى كادىرلار
بىلەن خاتىرە سۈرەتكە چۈشكەندىن كېيىن، يۈ جىڭشىڭ سۆز قىلىپ، شىنجاڭ ھەربىي
رايونىنىڭ ۋەتەن زىمىنىنىڭ پۈتۈنلۈكىنى قوغداش، چېگرا مۇداپىئەسىنى مۇستەھكەملەش،
شىنجاڭنىڭ ئىجتىمائىي مۇقىملىقىنى قوغداش، «3 خىل كۈچ» كە قارشى تۇرۇش، يەرلىكنىڭ
قۇرۇلۇشىغا ياردەم بېرىش، شىنجاڭنىڭ ئىقتىسادىي ۋە ئىجتىمائىي تەرەققىياتىنى
ئىلگىرى سۈرۈش ئۈچۈن قوشقان تۆھپىسىنى تولۇق مۇئەييەنلەشتۈردى. ئۇ كوماندىر-
جەڭچىلەرنىڭ يېڭى ۋەزىيەتتە ئارمىيەنى كۈچلەندۈرۈش نىشانىنى پۇختا ئىگىلەپ، بۇرچ-
ۋەزىپىلەرنى ئادا قىلىش ئىقتىدارىنى ئۆزلۈكسىز يۇقىرى كۆتۈرۈپ، شىنجاڭنىڭ
ئىجتىمائىي مۇقىملىقى ۋە ئەبەدىي ئەمىنلىكىنى قوغداشنى كۈچلۈك كاپالەتكە ئىگە
قىلىشىنى ئۈمىد قىلدى.  
مەركەز ۋەكىللەر ئۆمىكىدىكىلەر شىنجاڭ ئىشلەپچىقىرىش- قۇرۇلۇش بىڭتۈەنى ئورگىنىغا
يېتىپ كەلگەندە، بىڭتۈەن ئورگىنىدىكى كادىرلار ۋە ئىشچى- خىزمەتچىلەرنىڭ قىزغىن
قارشى ئېلىشىغا ئېرىشتى. يۈ جىڭشىڭ قاتارلىقلار بىڭتۈەندىكى دىۋىزىيە- ئىدارە
دەرىجىلىكتىن يۇقىرى كادىرلار بىلەن قول ئېلىشىپ كۆرۈشتى ھەمدە خاتىرە سۈرەتكە
چۈشتى. يۈ جىڭشىڭ سۆز قىلىپ مۇنداق دېدى: ئەۋلاتمۇ- ئەۋلات بىڭتۈەنلىكلەر
شىنجاڭنى ئېچىش، مىللەتلەر بۈيۈك ئىتتىپاقلىقىنى ئىلگىرى سۈرۈش، ئىقتىسادىي ۋە
ئىجتىمائىي تەرەققىياتنى ئالغا سىلجىتىش، چېگرا مۇداپىئەسىنى مۇستەھكەملەش
جەھەتتە شانلىق تۆھپىلەرنى ياراتتى. نۆۋەتتە، شىنجاڭ خىزمىتى يېڭى تارىخىي
باشلىنىش نۇقتىسىدا تۇرماقتا. بىڭتۈەن خىزمىتىنى تېخىمۇ مۇھىم ئىستىراتېگىيەلىك
ئورۇنغا قويۇپ، بىڭتۈەننىڭ مۇقىملىقنى ساقلاش، چېگرا مۇداپىئەسىنى
مۇستەھكەملەشتىكى ئىقتىدار قۇرۇلۇشىنى يەنىمۇ كۈچەيتىپ، بىڭتۈەن بىلەن يەرلىك بىر
ئائىلە بولۇش ئىدىيەسىنى پۇختا تۇرۇغۇزۇپ، ئۇنىۋېرسال ئەمەلىي كۈچىنى ئۆزلۈكسىز
ئاشۇرۇپ، ماددىي ئاساسنى مۇستەھكەملەپ، ھەر قايسى مىللەتلەرنىڭ ئالماشتۇرۇشى ۋە
بېرىش- كېلىشىنى يەنىمۇ ئىلگىرى سۈرۈش كېرەك.     قوراللىق ساقچى قىسىم شىنجاڭ
باش ئەترىتىدە، مەركەز ۋەكىللەر ئۆمىكى ئوفىتسىر- ئەسكەرلەرنىڭ قىزغىن قارشى
ئېلىشىغا ئېرىشتى. قوراللىق ساقچى قىسىم شىنجاڭ باش ئەترىتىنىڭ ئۈرۈمچىدىكى تۈەن
دەرىجىلىكتىن يۇقىرى كادىرلىرى بىلەن كۆرۈشكەندە، يۈ جىڭشىڭ مۇنداق دېدى: يېقىنقى
يىللاردىن بۇيان، «3 خىل كۈچ» نىڭ كەسكىن رىقابىتىگە قارىتا، قوراللىق ساقچى
قىسىم شىنجاڭ باش ئەترىتىدىكى ئوفىتسىر- ئەسكەرلەر مەركەزنىڭ سىياسەت-
ئورۇنلاشتۇرمىلىرىنى قەتئىي ئىزچىللاشتۇرۇپ، دادىل بولۇپ، شىنجاڭنىڭ ئىتتىپاقلىقى
ۋە مۇقىملىقىنى قوغداشتىكى رولىنى ياخشى جارى قىلدۇردى. نۆۋەتتە، شىنجاڭ--
ئالاھىدە مۇھىم دەۋردە تۇرماقتا. قوراللىق ساقچى قىسىم شىنجاڭ باش ئەترىتىدىكى
ئوفىتسىر- ئەسكەرلەرنىڭ زىممىسىگە ئالاھىدە مۇھىم مەسئۇلىيەت يۈكلەنگەن بولۇپ،
بارلىق ئوفىتسىر- ئەسكەرلەرنىڭ پارتىيە مەركىزىي كومىتېتىنىڭ شىنجاڭ خىزمىتى
توغرىسىدىكى سىياسەت- فاڭجىنلىرىنى قەتئىي ئىزچىللاشتۇرۇپ، خەلق ئاممىسى بىلەن
بولغان قان بىلەن گۆشتەك مۇناسىۋىتىنى كۈچەيتىپ، پارتىيەنىڭ قوماندانلىقىغا
بويسۇنىدىغان، ئۇرۇشتا غەلبە قىلالايدىغان، ئىستىلدا دۇرۇس بولغان،
زامانىۋىلاشقان قوراللىق ساقچى قىسىم قۇرۇپ چىقىپ، شىنجاڭنىڭ جەمئىيەت مۇقىملىقى
ۋە ئەبەدىي ئەمىنلىكى ئۈچۈن تېخىمۇ زور تۆھپىلەرنى قوشۇنىنى ئۈمىد قىلىمەن.  
شىنجاڭ مەرھابا مىھمانخانىسى يىغىن مەركىزىنىڭ جەنۇبىي تەرىپىدە، ئاپتونوم
رايونىمىزنىڭ جامائەت خەۋپسىزلىكى، سوت مەھكىمىسى، تەپتىش مەھكىمىسى سىستېمىسىدىن
كەلگەن 100 دىن ئارتۇق ۋەكىل ئۇچىسىغا فورما كىيىپ، رەتلىك تىزىلغانىدى. ئۇلارنىڭ
ئىچىدە، ھەم كۆپ قېتىم تۆھپە ياراتقان، چېچىغا ئاق سانجىغان پىشقەدەم نەمۇنىچىلار
ۋەكىللىرى ھەم ياش ۋە تالانتلىق كادىرلار بار ئىدى. سىياسىي- قانۇن سىستېمىسىدىكى
ۋەكىللەر بىلەن خاتىرە سۈرەتكە چۈشكەندىن كېيىن، يۈ جىڭشىڭ سۆز قىلىپ، سىياسىي-
قانۇن سىستېمىسىدىكى كادىر- ساقچىلارنىڭ دۆلەتنىڭ بىخەتەرلىكىنى قوغداش، جەمئىيەت
مۇقىملىقىنى ساقلاش، خەلقنىڭ مەنپەئەتىنى قوغداش جەھەتتە، زور قۇربانلىق
بەرگەنلىكى ۋە تۆھپە ياراتقانلىقىنى تولۇق مۇئەييەنلەشتۈردى. يۈ جىڭشىڭ مۇنداق
دېدى: سىياسىي- قانۇن ئورگانلىرى-- شىنجاڭنىڭ جەمئىيەت مۇقىملىقىنى قوغداشتىكى
ھالقىلىق كۈچ بولۇپ، شىنجاڭنىڭ ئەبەدىي ئەمىنلىكىنى ئەمەلگە ئاشۇرۇشتا، ئالاھىدە
