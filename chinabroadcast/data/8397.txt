# ئامېرىكىنىڭ <قايتا تەڭپۇڭلاشتۇرۇش> ئىستىراتېگىيەسى ئاسىيا-تىنچ ئوكيان
رايونىنى تەڭپۇڭلاشتۇرالمىدى

2014-06-05 19:02:56     مەنبەسى: شىنجاڭ خەلق رادىيو ئىستانسىسى  

 يېقىندا، ئامېرىكا زۇڭتۇڭى ئوباما ۋېستپوئېنت ھەربىي مەكتىپىنىڭ ئوقۇش پۈتتۈرۈش
مۇراسىمىدا نۇتۇق سۆزلەپ، ئامېرىكىنىڭ دىپلوماتىيە سىياسىتىنى ئاقلىدى. ئوباما
نۇتۇقىدا ئامېرىكىنىڭ جەنۇبىي دېڭىز مەسىلىسى قاتارلىق ئاسىيانىڭ خەۋپسىزلىك
ئىشلىرىغا ئارىلىشىشىنىڭ ئېنىق مەقسىتى ئۈستىدە توختىلىپ، ئامېرىكا ئارمىيەسى
چوقۇم كىرزىسقا ياخشى تەييارلىق قىلىشى كېرەك، جەنۇبىي دېڭىز قاتارلىق جايلاردىكى
«رايون خاراكتېرلىك تاجاۋۇزچىلىق» ئامېرىكىنىڭ ئىتتىپاقداشلىرىغا تەسىر كۆرسىتىپ،
ئامېرىكا ئارمىيەسىنىڭ بۇ ئىشقا چېتىلىپ قېلىشىنى كەلتۈرۈپ چىقىرىشى مۇمكىن،
دېدى.       ئوبامانىڭ نۇتۇقى شۇ زامان فىلىپپىن، ۋېيتنام ۋە ياپونىيە قاتارلىق
دۆلەتلەرنىڭ ئاخبارات ۋاستىلىرىنى قاينىتىۋەتتى. ئۇلار خۇشاللىقىدىن قىن- قىنىغا
پاتماي، ئوبامانىڭ بۇ نۇتۇقى ئامېرىكىنىڭ فىلىپپىن ۋە شەرقىي- جەنۇبىي ئاسىيا
ئەللىرى ئىتتىپاقىدىكى ھەر قايسى دۆلەتلەرنى قوللايدىغانلىقىنى بىلدۈرىدۇ، دەپ
قارىدى. يېقىندىن بۇيان، ئامېرىكا ئەمەلدارلىرىنىڭ جەنۇبىي دېڭىز مەسىلىسى
ئۈستىدە توختىماي جۇڭگونى ئەيىبلەپ، فىلىپپىن بىلەن ۋېيتنامغا يان بېسىشتىن
ئىبارەت خاتا پوزىتسىيەدە بولغانلىقىنى بۇ ئىش بىلەن بىرلەشتۈرۈپ ئويلىساق،
ئوبامانىڭ نۇتۇقى ئامېرىكىنىڭ «ئاسىيا-تىنچ ئوكيان رايونىنى قايتا
تەڭپۇڭلاشتۇرۇش» ئىستىراتىگىيەسىنىڭ ھەقىقىي مەقسىتىدىن تېخىمۇ ھوشيار بولۇش
كېرەكلىكىنى بىلدۈرىدۇ.       ئامېرىكىنىڭ ئاسىيا-تىنچ ئوكيان رايونىدىكى
يېتەكچىلىك ئورنىنى تېخىمۇ كۈچەيتىپ، ئاسىيا-تىنچ ئوكيان رايونىنىڭ ئىقتىسادىي
تەرەققىياتىدىن پايدا ئېلىش ئۈچۈن، ئوباما ھۆكۈمىتى 2009- يىلىنىڭ ئاخىرىدا
ئامېرىكىنىڭ دۇنياۋىي ئىستىراتىگىيەسىنىڭ مەركىزىنى شەرققە يۆتكەيدىغانلىقىنى
رەسمىي ئوتتۇرىغا قويدى. شۇنىڭدىن كېيىن، ئامېرىكىنىڭ «ئاسىيا-تىنچ ئوكيان
رايونىغا قايتىش»، «ئاسىيا-تىنچ ئوكيان رايونىغا يۈزلىنىش» ھەم «ئاسىيا-تىنچ
ئوكيان رايونىنى قايتا تەڭپۇڭلاشتۇرۇش» قاتارلىق ناملار بىلەن ئاتالغان يېڭى
ئاسىيا-تىنچ ئوكيان رايونى ئىستىراتېگىيەسى شەرقىي ئاسىيا رايونىغا زور
ئۆزگىرىشلەرنى ئېلىپ كېلىپ، ئاسىيا-تىنچ ئوكيان رايونىدا ئەنسىزلىكنى كەلتۈرۈپ
چىقاردى. سوغۇق مۇناسىۋەتلەر ئۇرۇشىدىن كېيىن ئىزچىل تىنچ ھالەتتە تۇرۇۋاتقان
تىنچ ئوكياننىڭ تىنچلىقى بۇزۇلدى. شەرقىي ئاسىيا رايونىدا ئۇزۇن مەزگىل مەۋجۇد
بولۇپ تۇرۇۋاتقان ئاراللارنىڭ ئىگىلىك ھوقۇقى ۋە دېڭىز-ئوكيان ھوقۇق- مەنپەئەتى
ئۈستىدىكى تالاش- تارتىش تۇيۇقسىز كەسكىنلىشىپ، يارىشىپ قېلىش يولىدا كۈچەۋاتقان
دۈشمەنلەر ۋە رىقابەتچىلەر قايتىدىن دۈشمەنلىشىپ، قىزىقلىق دەرىجىسى تۆۋەنلەشكە
باشلىغان بىر قىسىم «قىزىق نۇقتىلار» قايتىدىن گەۋدىلىنىپ، رايونلاردىكى ھەربىي
ھازىرلىقلار مۇسابىقىسى ئاستا- ئاستا شەكىللەندى.      ئامېرىكىنىڭ «ئاسىيا-تىنچ
ئوكيان رايونىنى قايتا تەڭپۇڭلاشتۇرۇش» ئىستىراتىگىيەسىنى ئىلگىرى سۈرۈشتىكى
ئاساسلىق ئۇسۇللىرى تۆۋەندىكىچە؛ بىرىنچى، ئامېرىكىنىڭ ياپونىيە، كورىيە،
ئاۋستىرالىيە ۋە فىلىپپىن قاتارلىق ئاسىيا- تىنچ ئوكيان رايونىدىكى
ئىتتىپاقداشلىرى بىلەن بولغان مۇناسىۋىتىنى كۈچەيتىش؛ ئىككىنچىسى، شەرقىي جەنۇبىي
ئاسىيا ۋە جەنۇبىي ئاسىيادىكى دۆلەتلەرنى ئۆزىگە تارتىپ، ئاسىيادىكى يېڭى
شېرىكلىك مۇناسىۋىتىنى كېڭەيتىش؛ ئۈچىنچىسى، ئاسىيا- تىنچ ئوكيان رايونىدىكى
ھەربىي ئىشلار ئورۇنلاشتۇرۇشىنى مۇستەھكەملەپ، ئامېرىكىنىڭ «مەۋجۇدلۇق تۇيغۇسى»نى
كۈچەيتىپ،  «ئاسىيا نۇسخىسى»دىكى باشقۇرۇلىدىغان بومبىدىن مۇداپىئەلىنىش
سېستىمىسىنىڭ قۇرۇلۇشىغا تۈرتكە بولۇپ، «ھاۋا ئارمىيەسى ۋە دېڭىز ئارمىيەسى بىر
گەۋدىلىشىپ ئۇرۇش قىلىش» ئىستىراتىگىيەسىنى ئىلگىرى سۈرۈش؛ تۆتىنچىسى، «تىنچ
ئوكيان ھالقىغان ئىستىراتېگىيەلىك ئىقتىسادىي شېرىكلىك مۇناسىۋىتى كېلىشىمى»نى
زور كۈچ بىلەن ئىلگىرى سۈرۈپ، ئاسىيا- تىنچ ئوكيان رايونىنىڭ ئىقتىسادىي
تەرەققىياتىدىكى يېتەكچىلىك ھوقۇقىنى قايتىدىن تارتىۋېلىشتىن ئىبارەت.      مەيلى
«ئاسىيا-تىنچ ئوكيان رايونىغا قايتىش»، «ئاسىيا-تىنچ ئوكيان رايونىغا يۈزلىنىش»
بولسۇن، ياكى ئاتالمىش «ئاسىيا-تىنچ ئوكيان رايونىنى قايتا تەڭپۇڭلاشتۇرۇش»
ئىستىراتېگىيەسى بولسۇن، بىر قانچە يىلدىن بۇيانقى ئەمەلىيەتتىن ئۆتكۈزۈش ۋە
ئۇنىڭ نەتىجىسى ئاسىيا-تىنچ ئوكيان رايونىدىكى نۇرغۇن دۆلەتلەردە ئامېرىكىنىڭ
ئاسىيا- تىنچ ئوكيان رايونىغا قاراتقان يېڭى ئىستىراتىگىيەسىگە نىسبەتەن
ئومۇميۈزلۈك ئەندىشە پەيدا قىلدى. ئەمەلىيەتتە، «ئاسىيا-تىنچ ئوكيان رايونىنى
قايتا تەڭپۇڭلاشتۇرۇش» ئىستىراتىگىيەسى ھەربىي ئىتتىپاقداش دۆلەتلەر ۋە شېرىكلىك
مۇناسىۋىتى ئورناتقان دۆلەتلەرنى ئامېرىكىنىڭ ئىستىراتىگىيەلىك مەنپەئەتى بىلەن
باغلاپ، ئۇلارنى ئاتالمىش ئامېرىكىغا «جەڭ ئېلان قىلغان» جۇڭگو بىلەن قارشىلىش
ئۇرۇشىنىڭ ئالدىنقى سېپىگە ئىتتىرىپ قويۇپ، ئامېرىكا يېتەكچىلىك قىلغان
بىخەتەرلىك تەرتىپىنى كۈچەيتتى؛ يەنە، دېڭىز-ئوكيان ھوقۇق مەنپەئەتى ئۈستىدىكى
تالاش- تارتىش ۋە ئەتراپتىكى دۆلەتلەرنىڭ جۇڭگونىڭ تەرەققىياتىغا بولغان
ئەندىشىسىدىن پايدىلىنىپ، جۇڭگو بىلەن ئۇنىڭ ئەتراپىدىكى دۆلەتلەرنىڭ
ئوتتۇرىسىدىكى مۇناسىۋەتنى بۇزۇپ، شەرقىي ئاسىيا رايونىنىڭ بىر گەۋدىلىشىش
تەرەققىياتىنى ئاستىلىتىپ قويدى؛ شۇنداقلا جۇڭگونىڭ دېڭىز ئۈستىدىكى ئەمەلىي
كۈچىنى كېڭەيتىش بوشلۇقىنى بىۋاستە چەكلەپ قويدىيۇ، جۇڭگو بىلەن بىۋاستە
قارشىلىشىش ۋە زىددىيەتلىشىشتىن ساقلىنىپ قالدى.      ئامېرىكىنىڭ «ئاسىيا-تىنچ
ئوكيان رايونىنى قايتا تەڭپۇڭلاشتۇرۇش» ئىستىراتېگىيەسى، ھەربىي ئىشلار ۋە
خەۋپسىزلىك مەسىلىسىنى گەۋدىلەندۈرۈپ، تىنچلىققا ئىنتىلىش، تەرەققىيات ئۈستىدە
ئىزدىنىش ۋە ھەمكارلىقنى ئىلگىرى سۈرۈشتىن ئىبارەت دەۋرنىڭ ئېقىمى بىلەن
قارىمۇقارشى بولۇپ قېلىپ، نۇرغۇن دۆلەتلەرنىڭ بىزار بولۇشىنى كەلتۈرۈپ چىقاردى،
شەرقىي- جەنۇبىي ئاسىيا ئەللىرى ئىتتىپاقىدىكى بىر قىسىم دۆلەتلەر ھەتتا
ئەندىشىسىنى ئىپادىلىدى. چۈنكى زېمىن تالاش-تارتىشىشىنىڭ كەسكىنلىشىشى ئۇلارنى
«بىر تەرەپنى تاللاش»قا مەجبۇر قىلدى، لېكىن بۇ ئۇلارنىڭ مەنپەئەتىگە ئۇيغۇن
كەلمەيتتى. ناھايىتى ئېنىقكى، جۇڭگو-فىلىپپىن، جۇڭگو-ۋېيتنام ئوتتۇرىسىدىكى
ئاراللار تالاش-تارتىشى مەۋجۇد بولۇپ تۇرۇۋاتقىنىغا ئىنتايىن ئۇزۇن ۋاقىت بولغان
بولسىمۇ،رايون تىنچلىقى ۋە مۇقىملىقىغا زور زەربە ئېلىپ كەلمىگەنىدى. ئەمما،
2010- يىلى ئامېرىكا جەنۇبىي دېڭىز مەسىلىسىگە بىۋاستە ئارىلىشىشقا باشلىغاندىن

تور بېتىمىزدىكى يېڭىلىقلار

 ئىستانسىمىز ئاپتونوم رايونلۇق خەلق ھۆكۈمىتى كەسپتىكى ناتوغرا ئىستېللارنى
تۈزىتىش ئىشخانىسى، ئاپتونوم رايونلۇق قاتناش ـ ترانسپورت نازارىتى، ئاپتونوم
رايونلۇق جامائەت خەۋپسىزلىكى نازارىتى بىلەن بىرلىشىپ ئۆتكۈزىدىغان، كەسپ
ئىستېلى قىزىق لېنىيىسى پروگراممىسىنىڭ «قاتناش 10 مىڭ چاقىرىملىق سەپىرى چوڭ
تىپتىكى بىۋاستە ئاڭلىتىش پائالىيىتى» 10- ئاينىڭ 16- كۈنى يەنى ئەتە خوتەن
ۋىلايىتىدە  ئۆتكۈزىلىدۇ

ئىستانسىمىز ئاپتونوم رايونلۇق خەلق ھۆكۈمىتى كەسپتىكى ناتوغرا ئىستېللارنى
تۈزىتىش ئىشخانىسى، ئاپتونوم رايونلۇق قاتناش ـ ترانسپورت نازارىتى، ئاپتونوم
رايونلۇق جامائەت خەۋپسىزلىكى نازارىتى بىلەن بىرلىشىپ ئۆتكۈزىدىغان، « كەسپ
ئىستېلى قىزىق لېنىيىسى پروگراممىسىنىڭ قاتناش 10 مىڭ چاقىرىملىق سەپىرى چوڭ
تىپتىكى بىۋاستە ئاڭلىتىش پائالىيىتى مۇشۇ ئاينىڭ 18- كۈنى، يەنى ئەتە بورتالا
شەھىرىدە ئۆتكۈزىلىدۇ. رادىيو ئاڭلىغۇچىلىرىمىزنىڭ بورتالا مۇڭغۇل ئاپتونوم
ئوبلاستىنىڭ يول تىرانسىپورت، يۈك ـ ترانسپورت ۋە قاتناش ئىشلىرىغا ئالاقىدار
