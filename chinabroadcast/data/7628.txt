# فابىئۇس ئۇكرائىنا چوڭ سايلىمىنىڭ ئوڭۇشلۇق بولۇشىنى ئۈمىد قىلدى

2014-05-24 10:31:51     مەنبەسى: شىنجاڭ خەلق رادىيو ئىستانسىسى  

مۇشۇ ئاينىڭ 25-كۈنى ئۇكرائىنا ۋاقىتلىق ھۆكۈمىتى بېكىتكەن چوڭ سايلام كۈنى.
گەرچە بۇ كۈننىڭ يېتىپ كېلىشىگە پەقەت 3 كۈنلا قالغان بولسىمۇ، لېكىن بۇ ھالقىلىق
سايلامنىڭ ئۆز قەرەلىدە ئۆتكۈزۈلۈش-ئۆتكۈزۈلمەسلىكى يەنىلا ئېنىق ئەمەس. يېقىندا
بېيجىڭدا زىيارەتتە بولۇۋاتقان فرانسىيەنىڭ دىپلوماتىيە ۋە خەلقئارا تەرەققىيات
مىنىستىرى فابىئۇس بۇ مەسىلە توغرىسىدا جۇڭگو ياشلار گېزىتى مۇخبىرىنىڭ سوئالىغا
جاۋاب بەرگەندە: «ياۋروپا ئىتتىپاقى ئۇكرائىنا چوڭ سايلىمىنىڭ ئۆز قەرەلىدە
ئۆتكۈزۈلۈشىنى ئۈمىد قىلىدۇ. ئەگەر رۇسىيە سايلامغا دەخلى قىلسا ياكى توسقۇنلۇق
قىلسا، ياۋروپا ئىتتىپاقى رۇسىيەگە قارىتا يەنىمۇ ئىلگىرىلىگەن ھالدا ئىنكاس
قايتۇرىدۇ» دېدى.  داۋالغۇش ئىچىدە تۇرغان ئۇكرائىنا مۇشۇ ئاينىڭ 25-كۈنى زۇڭتۇڭ
سايلىمى ئۆتكۈزۈشنى بېكىتكەن. بۇ ئۇكرائىنانىڭ «پەۋقۇلئاددە مەزگىلى»گە توغرا
كەلگەن بىر قېتىملىق «پەۋقۇلئاددە سايلام» بولغاچقا، سايلامنىڭ ئۆز قەرەلىدە
ئوڭۇشلۇق ئۆتكۈزۈلۈش-ئۆتكۈزۈلمەسلىكى ئۇكرائىنانىڭ تەقدىرىگە مۇناسىۋەتلىك بولۇپ
قالغان. ھالبۇكى، ئۇكرائىنانىڭ ھازىرقى ۋەزىيىتى كىشىنى يەنىلا ئەندىشىگە سالىدۇ.
ئۇكرائىنا ۋاقىتلىق ھۆكۈمىتى شەرقىي قىسىمغا بولغان كونتىروللۇقىنى بارغانسېرى
يوقىتىپ قويماقتا. ئۇكرائىنانىڭ ئىچكى ئىشلار مىنىستىرى ئاۋاكوف يېقىندا،
مۇستەقىللىقنى قوللىغۇچىلارنىڭ پائالىيىتى تۈپەيلىدىن، «ئۆزىنى ئۆزى ئىدارە
قىلىدىغانلىقى»نى جاكارلىغان دونېتسك ۋە لۇگانسك ئوبلاستلىرىنىڭ نۇرغۇن
رايونلىرىدا «نورمال ئۇسۇلدا سايلام ئۆتكۈزگىلى بولمايدىغانلىقى»نى ئېتىراپ
قىلدى. ئۇكرائىنا «شەرقىي جەنۇبىي قىسىم ماسلاشتۇرۇش كومىتېتى»نىڭ ئەزاسى
ئاپىخكىن: «بىزنىڭ ۋەزىپىمىز ھەر قانداق ئەھۋال ئاستىدا ئۇكرائىنانىڭ زۇڭتۇڭ
سايلىمىغا قاتناشماسلىق» دېدى. شەرقىي قىسىم مەسىلىسىدىن سىرت، ئۇكرائىنا
ۋاقىتلىق ھۆكۈمىتى يەنە مۇشۇ ئاينىڭ 25-كۈنىدىكى زۇڭتۇڭ سايلىمىدىن بۇرۇن، ئۆز
چېگراسى ئىچىنىڭ تىنچلىقىنى ساقلاشتىمۇ ھالسىراپ قالماقتا. ئۇكرائىنا مەركىزىي
سايلام كومىتېتىنىڭ دېيىشىچە، ئەگەر زوراۋانلىق ۋەقەلىرى يۈز بەرسە، 2 مىليونغا
يېقىن كىشى بېلەت تاشلاشقا قاتنىشالماسلىقى مۇمكىن ئىكەن. تېخىمۇ مۇھىمى، رۇسىيە
زۇڭتۇڭى پۇتىن رۇسىيە-ئۇكرائىنا چېگراسى ئەتراپىدا مانۋېر ئۆتكۈزۈۋاتقان رۇسىيە
ئارمىيەسىگە بازىغا قايتىش بۇيرۇقى چۈشۈرگەن بولسىمۇ، لېكىن شىمالىي ئاتلانتىك
ئەھدى تەشكىلاتىنىڭ باش كاتىپى راسمۇسسېننىڭ بىلدۈرۈشىچە، ئۇ 50 مىڭ كىشىلىك
رۇسىيە ئارمىيەسىنىڭ ئۇكرائىنا چېگراسى ئەتراپىدىن چېكىنگەنلىكىگە دائىر
ھېچقانداق پاكىتقا ئېرىشمىگەن.  يۇقىرىقى ئامىللاردىن شۇنى كۆرۈۋېلىشقا بولىدۇكى،
ئالاقىدار تەرەپلەر داۋاملىق تىركىشىۋاتقان ئەھۋال ئاستىدا، ئۇكرائىنا چوڭ
سايلىمىنىڭ ئوڭۇشلۇق ئۆتكۈزۈلۈش-ئۆتكۈزۈلمەسلىكى يەنىلا ئېنىق ئەمەس.  ھالبۇكى،
ئۇكرائىنانىڭ ئىچكى ئىشلار مىنىستىرى ئاۋاكوف چوڭ سايلام نۇرغۇن رايوندا
«بۇزغۇنچىلىق»قا ئۇچرىشى مۇمكىن بولسىمۇ، سايلامنىڭ يەنىلا ئۆتكۈزۈلىدىغانلىقىنى
بىلدۈرگەن. لېكىن قىسمەن تەھلىلچىلەرنىڭ كۆرسىتىشىچە، چوڭ سايلام باشلانغان ھەم
داۋاملاشقان تەقدىردىمۇ، ئوڭۇشلۇق تاماملىنىشى ناتايىن بولۇپ، ھالقىلىق مەسىلە
يەنىلا رۇسىيەنىڭ پوزىتسىيەسىگە باغلىق ئىكەن.  ئۇكرائىنا چوڭ سايلىمىنىڭ
ئىستىقبالىغا قارىتا، پولشا زۇڭلىسى تۇسك يېقىندا، كەلگۈسىدىكى بىر قانچە كۈن
ياكى بىر قانچە ھەپتىنىڭ ئۇكرائىنانىڭ تەقدىرىنى بەلگىلەيدىغانلىقى، ياۋروپا
ئىتتىپاقىنىڭ كۈچنى مەركەزلەشتۈرۈپ، ئۇكرائىنانىڭ چوڭ سايلامنى ئوڭۇشلۇق
ئۆتكۈزۈشىگە ياردەم بېرىشى كېرەكلىكىنى بىلدۈردى. رۇمىنىيە زۇڭتۇڭى باسېسكۇمۇ:
«ئەگەر ئۇكرائىنانىڭ چوڭ سايلىمى ئوڭۇشلۇق ئۆتكۈزۈلمىسە، ئۇكرائىنا پارچىلىنىپ
كېتىشى مۇمكىن» دېدى. شىمالىي ئاتلانتىك ئەھدى تەشكىلاتىنىڭ باش كاتىپى راسمۇسسېن
بولسا: «مۇشۇ ئاينىڭ 25-كۈنى ئۆتكۈزۈلىدىغان زۇڭتۇڭ سايلىمى ئۇكرائىنا تارىخىدىكى
مۇھىم پۇرسەت بولۇپ قالىدۇ. تەرەپلەر مۇشۇ پۇرسەتتىن پايدىلىنىپ، ئۇكرائىنا
كىرزىسىنى تىنچ ھەل قىلىدىغان يول ۋە ئۇسۇلنى تېپىپ چىقالىشى مۇمكىن» دېدى.
رۇسىيە زۇڭتۇڭى پۇتىن مۇشۇ ئاينىڭ 20-كۈنىدىن 21-كۈنىگىچە ئېلىمىزدە دۆلەت
ئىشلىرى زىيارىتىدە بولدى ھەم شاڭخەيدە ئۆتكۈزۈلگەن «ئاسىيا ئۆزئارا ھەمكارلىشىش
ۋە ئىشىنىش تەدبىرلىرى يىغىنى»نىڭ 4-قېتىملىق باشلىقلار يىغىنىغا قاتناشتى.
خەلقئارادىكى قىسمەن تاراتقۇلار پۇتىننىڭ بۇ قېتىملىق سەپىرىنى ياۋروپا ۋە
ئامېرىكىنىڭ جازاسىغا قارشى «بۆسۈپ چىقىش سەپىرى»، «ياردەم تەلەپ قىلىش سەپىرى»
دەپ سۈپەتلىدى. فرانسىيەنىڭ دىپلوماتىيە ۋە خەلقئارا تەرەققىيات مىنىستىرى
فابىئۇس ئېلىمىز مۇخبىرىنىڭ زىيارىتىنى قوبۇل قىلغاندا: «سىياسىينى بىلەن
ئىقتىساد، سودىنى ئارىلاشتۇرۇۋېتىشكە بولمايدۇ. نۆۋەتتە جۇڭگو-رۇسىيە
مۇناسىۋىتىنىڭ تەرەققىياتى ياخشى بولماقتا، بۇ دۇنيانىڭ تىنچلىقى ۋە مۇقىملىقىغا
پايدىلىق» دېدى.  ئۇكرائىنا كىرزىسى پارتلىغاندىن بۇيان، ياۋروپا ئىتتىپاقى
رۇسىيەگە قارىتا 2 قېتىم جازالاش تەدبىرى قوللاندى. فابىئۇسنىڭ بىلدۈرۈشىچە،
ياۋروپا ئىتتىپاقىنىڭ جازاسى رۇسىيەنىڭ ئۇكرائىنانىڭ ئىشلىرىغا ئارىلاشقانلىقىغا
قايتۇرۇلغان ئىنكاس ئىكەن. ئۇ: «رۇسىيەنىڭ قىرىمنى يۇتۇۋېلىشى زېمىن پۈتۈنلۈكىدىن
ئىبارەت خەلقئارا قانۇننىڭ بۇ ئاساسىي مىزانىغا خىلاپ. شۇڭا ياۋروپا ئىتتىپاقى
رۇسىيەگە قارىتىلغان جازالاش تەدبىرلىرىنى ئۆز ئىچىگە ئالغان ئىنكاسلارنى
قايتۇردى» دېدى. فابىئۇس يەنە مۇخبىرغا: «ياۋروپا ئىتتىپاقى رۇسىيەدىن
ئۇكرائىنانىڭ مۇشۇ ئاينىڭ 25-كۈنىدىكى چوڭ سايلىمىغا بىۋاستە ياكى ۋاستىلىق دەخلى
قىلماسلىقنى ئېنىق تەلەپ قىلدى. ئەگەر رۇسىيە چوڭ سايلامنىڭ ئوڭۇشلۇق ئېلىپ
بېرىلىشىغا توسقۇنلۇق قىلسا، بىز چوقۇم يەنىمۇ ئىلگىرىلىگەن ھالدا ئىنكاس
قايتۇرۇپ، يېڭى جازالاش تەدبىرلىرىنى قوللىنىمىز» دېدى. گەرچە فابىئۇس ياۋروپا
ئىتتىپاقىنىڭ قانداق يېڭى جازا تەدبىرلىرىنى قوللىنىدىغانلىقىنى ئېنىق ئېيتمىغان
بولسىمۇ، لېكىن ئۇ مۇشۇ ئاينىڭ بېشىدا، ئەگەر ئۇكرائىنانىڭ چوڭ سايلىمى ئوڭۇشلۇق
ئۆتكۈزۈلمىسە، ياۋروپا ئىتتىپاقىنىڭ رۇسىيەگە قارىتا 3-قېتىملىق جازا، يەنى
ئىقتىسادىي جازا يۈرگۈزۈشى مۇمكىنلىكىنى بىلدۈرگەنىدى. شۇنىڭ بىلەن بىرگە ئۇ
يەنە: «بىز داۋاملىق رۇسىيە ۋە باشقا ھەر قايسى تەرەپلەر بىلەن ئالاقىنى ساقلاپ

ۋىلايىتىدە  ئۆتكۈزىلىدۇ
