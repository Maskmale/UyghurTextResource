# ئارزۇ–ئارمانلارنىڭ باشلانغان جايى يېزا مەكتەپلىرىدىكى ئۆسمۈرلەر سارىيى

2015-04-09 10:31:27     مەنبەسى: شىنجاڭ خەلق رادىيو ئىستانسىسى  

جاۋ شىيا

   27 – مارت ئۈرۈمچى ناھىيەسى كىچىك ساي ئوتتۇرا مەكتىپىنىڭ يېزا مەكتىپى
ئۆسمۈرلەر سارىيىغا بارغىنىمدا، كىلاسسىك ئەسەرلەرنى دېكلاماتسىيە قىلىش
سىنىپىدىكى بالىلار ئوقۇتقۇچىنىڭ يېتەكچىلىكىدە، كىلاسسىك ئەسەرلەرنى ئوقۇۋاتقان
ئىكەن. بالىلارنىڭ كىتاب ئوقۇغاندىكى جاراڭلىق ئاۋازى كىشىگە يېقىملىق ناخشىدەك
ئاڭلىناتتى. يەنە بىر سىنىپتىكى بالىلار رەسىم قەلىمى ئارقىلىق ياپيېشىل ئوتلاق،
ئوتلاۋاتقان پادىلارنى تولىمۇ جانلىق سىزىۋاتاتتى. مەيداندىكى بالىلار بولسا
خۇشال ھالدا غوڭۇلداق ئويناۋاتقان ئىكەن.  
   مەكتەپ مۇدىرى چىن جياجۈن مۇنداق دېدى: مەكتىپىمىزدە 2011 – يىلى ئۆكتەبىردە
يېزا مەكتىپى ئۆسمۈرلەر سارىيى قۇرۇلغاندىن بۇيان غوڭۇلداق، قەغەز قىيما،
غالتەكلىك تېيىلىش، ئېلېكتىرونلۇق رويال قاتارلىق 21 تۈر بويىچە سىنىپ ئېچىلدى،
بۇ مول مەزمۇنلۇق پائالىيەتلەر بالىلارنىڭ قىزىقىشىنى قوزغاپ، ئارزۇ –
ئارمانلىرىنى يالقۇنجاتتى، ئۇلارنى تېخىمۇ ئوچۇق – يورۇق ۋە جاسارەتلىك قىلدى.  
   يوقلۇقتىن بارلىققا كېلىش، كىچىكلىكتىن چوڭىيىشقا قاراپ تەرەققىي قىلىش
جەريانىدا، مەركەز ۋە ئاپتونوم رايون سېلىنمىسىنىڭ ئۈزلۈكسىز ئېشىشىغا ئەگىشىپ،
يېزا مەكتەپلىرىدىكى ئۆسمۈرلەر سارىيى تۈر قۇرۇلۇشىنىڭ قاپلىنىش كۆلىمى
بارغانسېرى كېڭەيدى. يېزا مەكتەپلىرىدىكى ئۆسمۈرلەر سارىيى بالىلار بېرىشقا
ئامراق جاي، ساپا مائارىپىنىڭ ياخشى ۋاسىتىسى، ئوقۇتقۇچى، ئوقۇغۇچىلارنىڭ پىكىر
ئالماشتۇرۇشىدىكى ياخشى سورۇنى بولۇپ، قۇرامىغا يەتمىگەنلەرنىڭ ئىدىيە – ئەخلاق
قۇرۇلۇشىدىكى ئەمەلىي قۇرۇلۇشقا ئايلاندى.  
  ھەر قايسى مىللەتتىن ۋەكىل خاراكتېرلىك ئىختىساسلىقلارنى يېتىشتۈرۈش ۋە بايقاش  
    ج ك پ مەركىزىي كومىتېتى سىياسىي بىيۇروسىنىڭ ئەزاسى، ئاپتونوم رايونلۇق پارتكومنىڭ شۇجىسى جاڭ چۈنشيەن 2012 – يىل 4 – ئايدا قەشقەر شەھىرى شامالباغ بازارلىق مەركىزىي باشلانغۇچ مەكتەپتە تەكشۈرۈپ تەتقىق قىلغاندا مۇنداق دېدى: مەكتەپلەر دەرستىن سىرتقى تۈرلۈك مول مەزمۇنلۇق پائالىيەتلەرنى قانات يايدۇرۇش ئارقىلىق بالىلارغا خۇشاللىق بېغىشلىشى كېرەك، ئىختىساسلىقلار قانداق يېتىشىپ چىقىدۇ؟ دەل مۇشۇنداق مەكتەپلەردىن يېتىشىپ چىقىدۇ. بۇ يەردىن كەلگۈسىنىڭ تۈركۈم – تۈركۈملىگەن تەنتەربىيەچى، رەسسام، ئارتىس قاتارلىق ئىختىساسلىقلىرى يېتىشتۈرۈلىدۇ، شىنجاڭ ھەر قايسى مىللەتلەردىن بولغان ۋەكىل خاراكتېرلىق ئىختىساسلىقلارنى يېتىشتۈرۈشكە ۋە بايقاشقا ناھايىتى موھتاج.  
   ‹‹ھەر قايسى مىللەتلەردىن بولغان ۋەكىل خاراكتېرلىك ئىختىساسلىقلارنى
يېتىشتۈرۈش ۋە بايقاش›› يېزا مەكتەپلىرىدە ئۆسمۈرلەر سارىيى قۇرۇشنىڭ مۇھىم
ۋەزىپىسى قىلىندى. ئۆسمۈرلەر سارىيى قۇرۇلۇشى يېزىلاردىكى قۇرامىغا يەتمىگەنلەرگە
بولغان ئىدىيە – ئەخلاق قۇرۇلۇشىنى يەنىمۇ كۈچەيتىدىغان ۋە ئىلگىرى سۈرىدىغان
خەلق تۇرمۇشى قۇرۇلۇشى بولۇش سۈپىتى بىلەن ئاپتونوم رايونىمىزدىكى يېزا
مەكتەپلىرىدە 2011 – يىلىدىن باشلاپ رەسمىي يولغا قويۇلدى. ئەينى ۋاقىتتا ئاز بىر
قىسىم يېزا مەكتەپلىرىدىلا ئۆسمۈرلەر سارىيى بار بولۇپ، كۆپلىگەن يېزا
مەكتەپلىرىدە تېخى قۇرۇلمىغان ئىدى. ئاپتونوم رايونىمىزدىكى جايلار ۋە تارماقلار
يېزا مەكتەپلىرىدە ئۆسمۈرلەر سارىيى قۇرۇشنى ئىلگىرى سۈرۈش ئۈچۈن، زىچ
ھەمكارلىشىپ، ‹‹ئاۋۋال يولغا قويۇپ، ئاندىن مۇكەممەللەشتۈرۈش››نى پىلانلاپ، تۈر
قۇرۇلۇشىنى باشلىۋەتتى. قەشقەر ۋىلايىتى، ئۈرۈمچى ناھىيەسى قاتارلىق جايلار
ئىجتىمائىي بايلىقلاردىن ئاكتىپ پايدىلىنىپ، يېزا مەكتەپلىرىدە ئۆسمۈرلەر سارىيى
قۇرۇش خىزمىتىنى تېزدىن يولغا قويۇپ، ئۆسمۈرلەر سارىيىنى تېزرەك ئىشلىتىشكە
كىرىشتۈرۈلۈشنى قولغا كەلتۈردى.  
   ئاپتونوم رايون خەلقكە نەپ يەتكۈزىدىغان بۇ قۇرۇلۇشنى يەنىمۇ ئىلگىرى سۈرۈش
ئۈچۈن، 2014 – يىلى يېزا مەكتەپلىرىنىڭ ئۆسمۈرلەر سارىيى قۇرۇلۇشىنى ئاپتونوم
رايوننىڭ خەلق تۇرمۇشى قۇرۇلۇشىغا كىرگۈزۈپ، 8 مىليون يۈەن مەبلەغ ئاجرىتىپ،
جەنۇبىي شىنجاڭدىكى ئۈچ ۋىلايەت، بىر ئوبلاستتىكى 40 يېزا مەكتىپىدە ئۆسمۈرلەر
سارىيى قۇرۇشنى قوللىدى. تۆت يىلدىن بۇيان، مەركەزنىڭ مەنىۋى مەدەنىيلىك
قۇرۇلۇشىغا يېتەكچىلىك قىلىش ئىشخانىسى مەركەزنىڭ مەخسۇس لاتارىيە جامائەت
پاراۋانلىق پۇلىدىن 60 مىليون 150 مىڭ يۈەننى ئاجرىتىپ، 234 يېزا مەكتىپىگە
ئۆسمۈرلەر سارىيى سالدى، بۇنىڭدىن دېھقانچىلىق، چارۋىچىلىق رايونىدىكى 200 مىڭدىن
ئارتۇق ئوقۇغۇچى مەنپەئەتلەندى. شۇنىڭ بىلەن بىللە، يېزا مەكتەپلىرىدىكى
ئۆسمۈرلەر سارىيىنىڭ نورمال ئايلىنىشىغا كاپالەتلىك قىلىش ئۈچۈن مەركەز يىلدا ھەر
بىر مەكتەپكە 50 مىڭ يۈەن يانداشما يۈرۈشتۈرۈش خىراجىتى ئاجرىتىپ بەردى.  
   ھەممە يېزا مەكتەپلىرىدە ئۆسمۈرلەر سارىيى قەد كۆتۈردى  
   قەشقەر شەھىرى بەشكېرەم يېزىلىق مەركىزىي باشلانغۇچ مەكتىپىنىڭ يېزا مەكتەپ
ئۆسمۈرلەر سارىيى ئۇيۇشتۇرغان جۇڭخۇا نەزمىلىرىنى دېكلاماتسىيە قىلىش ‹‹قوش
تىل››لىق ئالاھىدە سىنىپىدا خەنزۇتىلىدا سۆزلىشىش، ئەدەبىي ساۋاتلارنى ئۆگىنىش،
دېكلاماتسىيە قىلىش قاتارلىق دەرسلىكلەر ئۆگىتىلگەن؛ قىزىلسۇ ئوبلاستى ماناس
داستانىنى ئېيتىش، كەشتىچىلىك، راۋاب ئۆگىنىش قاتارلىقلارنى يېزا مەكتەپ
ئۆسمۈرلەر سارىيىنىڭ خاس پائالىيەت تۈرى قىلغان؛ ساۋەن ناھىيەسى سىداۋخوزا
بازىرىنىڭ يېزا مەكتىپى ئۆسمۈرلەر سارىيى قاپاق نەي، قەغەز قىيىش، مارجاندىن گۈل
ياساش قاتارلىق ئالاھىدە پائالىيەتلەرنى ئۇيۇشتۇرۇپ، نۇرغۇن ئوقۇغۇچىنى جەلپ
قىلغان؛ گۇما ناھىيەسىنىڭ ‹‹ئاتا – ئانىلارغا يېزىلغان خەت››نى ھەر بىر ئاتا –
ئانىنىڭ قولىغا يەتكۈزۈشىدە ھەر بىر ئوقۇغۇچىنى ئۆسمۈرلەر سارىيىنىڭ
پائالىيەتلىرىگە قاتناشتۇرۇش مەقسەت قىلىنغان.  
   كىشىنى ھاياجانلاندۇرىدىغىنى شۇكى، بۇ يىل مەركەز ۋە ئاپتونوم رايون خەلق
تۇرمۇشىغا نەپ يەتكۈزىدىغان بۇ تۈر قۇرۇلۇشىنىڭ سالمىقىنى داۋاملىق زورايتتى.
مەركەز بۇ يىل 20 مىليون يۈەن مەبلەغ سېلىپ، ئاپتونوم رايونىمىزدىكى 100 يېزا
مەكتىپىگە ئۆسمۈرلەر سارىيى سالماقچى. شۇنىڭ بىلەن بىللە، ئاپتونوم رايون يېزا
مەكتەپلىرىنىڭ ئۆسمۈرلەر سارىيى قۇرۇلۇشىنى داۋاملىق خەلق تۇرمۇشى قۇرۇلۇشى
قاتارىغا كىرگۈزۈپ، 10 مىليون يۈەن مەبلەغ ئاجرىتىپ، 40 يېزا مەكتىپىدە ئۆسمۈرلەر
سارىيى قۇرۇپ (بۇنىڭ ئىچىدە 2 مىليون يۈەن بۇلتۇر قۇرۇلغان 40 يېزا مەكتىپىدىكى
ئۆسمۈرلەر سارىيىنى ئايلاندۇرۇش مەبلىغى قىلىنىدۇ)، يېزىلاردىكى قۇرامىغا
يەتمىگەنلەرنىڭ مەنىۋى مەدەنىيەت تۇرمۇشىنى بېيىتىدۇ. بۇنىڭدىن باشقا، ئاپتونوم
رايون بۇ يىل بايلىقلارنى بىرلەشتۈرۈش، ئاساسنى پۇختىلاش، تەربىيەچىلەر قوشۇنى
قۇرۇلۇشىنى كۈچەيتىش قاتارلىق خىزمەتلەرنى چىڭ تۇتۇش ئاساسىدا، يېزا
مەكتەپلىرىدىكى ئۆسمۈرلەر سارىيى قۇرۇلۇشىنىڭ تەرەققىياتىنى تېزلىتىدۇ؛ ئۈرۈمچى

ئاجرىتىپ، ئۈچ يىل ئىچىدە ئۈرۈمچى شەھىرىدىكى 191 شەھەر، يېزا مەكتىپىدە
ئۆسمۈرلەر سارىيى قۇرىدۇ.  
   ئاپتونوم رايونلۇق مائارىپ نازارىتى ئىدىيەۋى – سىياسىي خىزمەت باشقارمىسىنىڭ

ئەمەلىيلىشىشى بىلەن تەڭرىتاغنىڭ جەنۇب – شىمالىدىكى ھەممە يېزا مەكتەپلىرىدە
ئۆسمۈرلەر سارىيى قەد كۆتۈرۈپ، دېھقانچىلىق، چارۋىچىلىق رايونلىرىدىكى بالىلارنىڭ
قەلبىگە ئارزۇ ۋە ئۈمىدنىڭ ئۇرۇقلىرىنى چاچىدۇ.
