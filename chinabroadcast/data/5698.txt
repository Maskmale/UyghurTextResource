# دىن ئىشلىرىنى قانۇن بويىچە باشقۇرۇشنى كۈچەيتىپ، دىن ئىناقلىقىنى ئىلگىرى
سۈرەيلى

2015-09-30 12:05:42     مەنبەسى: شىنجاڭ خەلق رادىيو ئىستانسىسى  

ئىستانسىمىزنىڭ «دىن ئىشلىرىنى قانۇن بويىچە باشقۇرۇشنى كۈچەيتىپ، دىن
ئىناقلىقىنى ئىلگىرى سۈرەيلى» دىگەن مەخسۇس ماقالىسى  
يېقىنقى بىر قانچە كۈندىن بۇيان، ئاپتونوم رايونىمىزنىڭ ھەر قايسى جايلىرىدىكى
دىنغا ئېتىقات قىلىدىغان ئاممىنىڭ قانۇنلۇق دىنىي پائالىيەتلىرى نورمال ئېلىپ
بېرىلماقتا. ئۇلار خاتىرجەم ۋە ئىناق كەيپىيتتا، قۇربان ھېيتنىڭ يىتىپ كېلىشىنى
كۈتمەكتە. ئۈرۈمچى شەھىرى قىبلىتاغ يولى ئىتتىپاق مەسچىتىنىڭ ئىمامى ئۆمەر ئېلى
مەمەتئىمىن ھېيت نامىزىنىڭ بەزى تەييارلىق ئىشلىرىنى قىلىۋاتقانىدى. ئۇ ھەر كۈنى
5 ۋاق ناماز ئۆتەپ، نەچچە يۈز مۇسۇلمان ئاممىسىغا قۇرئان تىلاۋەت قىلىپ، تەفسىر
قىلىدۇ. ئۇ مۇنداق دېدى: ھۆكۈمەت بىزنىڭ ھېيت- بايرام مەزگىلىدىكى بىخەتەرلىك،
ئوتتىن مۇداپىئە كۆرۈش، سەھىيە ۋە يۇقۇملۇق كېسەللىكلەرنىڭ ئالدىنى ئېلىش
قاتارلىق خىزمەتلەرنى ياخشى ئىشلىشىمىزگە ياردەم بىرىپ كىلىۋاتىدۇ ھەمدە
ئەتراپتىكى يوللارنى راۋانلاشتۇرىۋاتىدۇ. مۇشۇ ئەتراپتىكى مۇسۇلمان ئاممىسىمۇ بۇ
ئىشقا خالىسانە قاتنىشىۋاتىدۇ. مۇسۇلمان بولغان بەندە مىننەتدارلىقنى بىلىشى،
جەمىئىيەتكە، دۆلەتكە مىننەتدارلىق بىلدۈرۈشى كېرەك. دىنىي پائالىيەتلەر قانۇن
ئارقىلىق قوغدىلىدۇ، مۇسۇلمانلار ئاممىسمۇ قانۇنغا رىئايە قىلىشى، ۋەتەننى سۆيۈپ
قانۇنغا رىئايە قىلىدىغان پۇقرالاردىن بولۇشى كېرەك. مۇشۇنداق بولغاندىلا ئاندىن
لاياقەتلىك مۇسۇلمان بولالايدۇ.  
نۆۋەتتە، جەنۇبىي شىنجاڭ ۋە شىمالىي شىنجاڭنىڭ ھەر قايسى جايلىرىدىكى ھەر مىللەت
ئاممىسى دىنغا ئېتىقات قىلىش ئەركىنلىكىدىن بەھرىمەن بولماقتا. نورمال دىنىي
پائالىيەتلەر قانۇن تەرىپىدىن قوغدالماقتا. خوتەن شەھىرى ئىلچى يېزىسى سەيچى
ئاقتاش كەنتى 2- گۇرۇپپىسىدىكى ئاھالە ئۇبۇل ھەسەن تۇرسۇننىياز مۇنداق دېدى:
بىزنىڭ دىنىي ۋە تۇرمۇش پائالىيىتىمىز قوغدىلىنىۋاتىدۇ. دىنىي ئۆرپ-ئادەتلىرىمىز
ياخشى ھۆرمەت قىلىنىۋاتىدۇ. ھازىر تۇرمۇشىمىز ناھايىتى ياخشى كىتىۋاتىدۇ. يىمەك-
ئىچمەك ۋە كىيىم- كېچەكتىن غېمىمىز يوق. يەنە كىلىپ ھازىرقى سىياسەت بەك ياخشى
بولغاچقا، بىز مۇسۇلمانلار تىنچلىقنى، خاتىرجەم بەختلىك تۇرمۇشنى تېخىمۇ سۆيىمىز.  
ئاپتونوم رايونىمىز دىنىي زاتلار قوشۇنى قۇرۇلۇشى جەھەتتە دىنىي زاتلارنىڭ
قەرەرلىك ئۆگىنىش تۈزۈمىنى ئورناتتى، ئاپتونوم رايون ۋە ھەر قايسى ۋىلايەت،
ئوبلاست، شەھەرلەردە دىنىي زاتلارنى تەربىيىلەش رەھبەرلىك ئورگىنى قۇرۇلدى.
ئاپتونوم رايونلۇق پارتكوم تەشۋىقات بۆلۈمىنىڭ كۈزەتكۈچىسى ليۇ جوڭكاڭ مۇنداق
دېدى: ئاپتونوم رايونىمىزدىكى دىنغا ئېتىقات قىلىدىغانلارنىڭ سانى پۈتكۈل
ئاپتونوم رايون نوپۇسىنىڭ يېرىمىنى ئىگىلەيدۇ، دىنىي زاتلارنى ۋەتەننى سۆيۈپ،
قانۇنغا رىئايە قىلىش،  شىنجاڭنىڭ تەرەققىياتى ۋە مۇقىملىقى ئۈچۈن  تۆھپە قوشۇشقا
قانداق يىتەكلەش جەھەتتە ئاپتونوم رايونىمىز دىنىي زاتلارنىڭ قوشۇن قۇرۇلۇشىنى
كۈچەيتىدۇ، بۇنى شىنجاڭغا نىسبەتەن ئەھمىيىتى ناھايىتى زور دىيىشكە بولىدۇ.  
60 يىلدىن بۇيان، ئاپتونوم رايونىمىز پارتىيىنىڭ تۈرلۈك دىن سىياسىنى ئومۇميۈزلۈك
ئىزچىللاشتۇرۇپ، دىنىي ئىشلارنى قانۇن بويىچە باشقۇرۇشنى كۈچەيتىپ، دىننى
سوتسىيالىستىك جەمىئىيەت بىلەن ماسلاشتۇرۇشقا پائال يىتەكچىلىك قىلدى. بولۇپمۇ
مەركەزنىڭ شىنجاڭنىڭ خىزمىتى سۆھبەت يىغىنى ئېچىلغاندىن بۇيان، ئاپتونوم
رايونىمىز مەركەزنىڭ سىياسىتى ۋە ئورۇنلاشتۇرۇشىنى قەتئىي ئىزچىللاشتۇرۇپ،
مەركەزنىڭ دىن خىزمىتى توغرىسىدىكى چوڭ فاڭجېنىنى شىنجاڭنىڭ ئەمەلىيىتى بىلەن زىچ
بىرلەشتۈرۈپ، بىر قاتار يېڭى ئوي-پىكىر، يېڭى چارە ۋە يېڭى تەدبىرلەر ئۈستىدە
پائال ئىزدەندى. شىنجاڭ ئىسلام ئىنىستىتوتىنىڭ ئوقۇتقۇچىسى ئابىت قازىباي مۇنداق
دېدى: ئاپتونوم رايون «ئاپتونوم رايوننىڭ دىنىي پائالىيەت سورۇنلرىنى باشقۇرۇش
توغرىسدىكى ۋاقىتلىق يولغا قويۇش چارىسى» ۋە «ئىسلام دىنىي خىزمىتىنى يەنىمۇ
كۈچەيتىش ۋە مۇكەممەللەشتۈرۈشكە ئالاقىدار بىر قانچە پىكىر»قاتارلىق بىر قاتار
سىياسەت ۋە تەدبىرلەرنى ئېلان قىلىپ يولغا قويۇپ، ئاپتونوم رايونىمىزنىڭ دىن
ئىشلىرىنى قانۇن بويىچە باشقۇرۇشنى كۈچەيتىپ، دىن خىزمىتىنى ئالغا سىلجىتىشنى
سىياسەت ئاساسى بىلەن تەمىنلىدى. بۇ يىل 1- ئاينىڭ 1- كۈنى يېڭىدىن تۈزۈپ
چىقىلغان «ئاپتونوم رايوننىڭ دىن ئىشلىرى نىزامى» يولغا قويۇلۇشقا باشلىدى، بۇ
«نىزام»دا مۇنۇلار ئېنىق ئوتتۇرىغا قويۇلدى: دىن ئىشلىرىنى قانۇنلاشتۇرۇش،
قانۇنسىز دىنىي پائالىيەتنى چەكلەش، ئەسەبىيلىكنىڭ ئالدىنى ئېلىش، جىيانىي
ھەرىكەتلەرگە زەربە بىرىشتە چىڭ تۇرۇش-- ئاپتونوم رايونىمىزنىڭ دىن ئىشلىرىنىڭ
تەرەققىياتىغا نىسبەتەن ناھايىتى زور يىتەكچىلىك ۋە تۈرتكىلىك رول ئوينايدۇ.

تور بېتىمىزدىكى يېڭىلىقلار

 ئىستانسىمىز ئاپتونوم رايونلۇق خەلق ھۆكۈمىتى كەسپتىكى ناتوغرا ئىستېللارنى
تۈزىتىش ئىشخانىسى، ئاپتونوم رايونلۇق قاتناش ـ ترانسپورت نازارىتى، ئاپتونوم
رايونلۇق جامائەت خەۋپسىزلىكى نازارىتى بىلەن بىرلىشىپ ئۆتكۈزىدىغان، كەسپ
ئىستېلى قىزىق لېنىيىسى پروگراممىسىنىڭ «قاتناش 10 مىڭ چاقىرىملىق سەپىرى چوڭ
تىپتىكى بىۋاستە ئاڭلىتىش پائالىيىتى» 10- ئاينىڭ 16- كۈنى يەنى ئەتە خوتەن
ۋىلايىتىدە  ئۆتكۈزىلىدۇ

ئىستانسىمىز ئاپتونوم رايونلۇق خەلق ھۆكۈمىتى كەسپتىكى ناتوغرا ئىستېللارنى
تۈزىتىش ئىشخانىسى، ئاپتونوم رايونلۇق قاتناش ـ ترانسپورت نازارىتى، ئاپتونوم
رايونلۇق جامائەت خەۋپسىزلىكى نازارىتى بىلەن بىرلىشىپ ئۆتكۈزىدىغان، « كەسپ
ئىستېلى قىزىق لېنىيىسى پروگراممىسىنىڭ قاتناش 10 مىڭ چاقىرىملىق سەپىرى چوڭ
تىپتىكى بىۋاستە ئاڭلىتىش پائالىيىتى مۇشۇ ئاينىڭ 18- كۈنى، يەنى ئەتە بورتالا
شەھىرىدە ئۆتكۈزىلىدۇ. رادىيو ئاڭلىغۇچىلىرىمىزنىڭ بورتالا مۇڭغۇل ئاپتونوم
ئوبلاستىنىڭ يول تىرانسىپورت، يۈك ـ ترانسپورت ۋە قاتناش ئىشلىرىغا ئالاقىدار
ئىنكاسلىرى بولسا پىروگراممىمىزنى  يىپ ئۇچى بىلەن تەمىنلىشىنى قارشى ئالىمىز.
ئۇچۇر مۇنبىرىمىزنىڭ نومۇرى: 1062888957
