# ئامېرىكا قاتار بىلەن 11 مىليارد دوللارلىق قورال-ياراغ سېتىپ بېرىش توختامى
ئىمزالىدى

2014-07-18 11:53:08     مەنبەسى: شىنجاڭ خەلق رادىيو ئىستانسىسى  

 ئامېرىكا يېقىندا قاتار بىلەن ۋاشىنگىتوندا سوممىسى 11 مىليارد ئامېرىكا
دوللىرىلىق بىر تۈرلۈك قورال-ياراغ سېتىش توختىمى ئىمزالاپ، قاتارغا 10 يۈرۈش
«ۋەتەنپەرۋەر» بەلگىلىك باشقۇرىلىدىغان بومبىغا قارشى تۇرۇش سېستىمىسى، 24 دانە
«ئاپاچ» بەلگىلىك ھۇجۇمچى تىك ئۇچار ئايروپىلان ۋە 500 دانە «يېنىك نەيزە»
بەلگىلىك تانكىغا قارشى باشقۇرىلىدىغان بومبا سېتىپ بېرىدىغان بولدى.
تاراتقۇلارنىڭ خەۋىرىدە بۇنىڭ ئامېرىكا يېقىنقى يىللاردىن بۇيان ئىمزالىغان ئەڭ
يوقىرى سوممىلىق قورال-ياراغ سېتىش توختىمى ئىكەنلىكى، شۇنداقلا ئامېرىكىنىڭ
تۇنجى قېتىم «ۋەتەنپەرۋەر» بەلگىلىك باشقۇرىلىدىغان بومبىغا قارشى تۇرۇش
سېستىمىسىنى قاتارغا سېتىپ بېرىشى ئىكەنلىكى ئېيتىلدى. ئامېرىكا دۆلەت مۇداپىئە
مىنىستىرى چاك خاگېل شۇ كۈنى قاتارنىڭ دۆلەت مۇداپىئە ۋەزىرى ھاماد بىن ئەلى
ئاتتىياھ بىلەن بەش بۇرجەكلىك بىنادا بۇ قورال-ياراغ سېتىش كېلىشىمىنى ئىمزالىدى.
بەش بۇرجەكلىك بىنانىڭ باياناتچىسى جون كوبى مۇنداق دېدى: «بۈگۈن ئىمزالانغان
كېلىشىم ئامېرىكا بىلەن قاتارنىڭ بىخەتەرلىك ۋە مۇداپىئە ئىشلىرى ساھەسىدىكى
قويۇق مۇناسىۋىتىنى گەۋدىلەندۈرۈپ بەردى، بۇ بىزنىڭ كۆپ تۈرلۈك ھەربىي ئىشلار
ھەركىتىدىكى قوش تەرەپلىك ھەمكارلىقىمىزغا پايدىلىق.» ئامېرىكا ئەمەلدارى
تاراتقۇلارغا قاتارنىڭ باشقۇرىلىدىغان بومبىغا قارشى تۇرۇش سېستىمىسىنى
سېتىۋېلىشى ئاساسلىقى ئىراندىن كېلىشى مۇمكىن بولغان باشقۇرىلىدىغان بومبا
تەھدىتىدىن مۇداپىئەلىنىش ئۈچۈن ئىكەنلىكىنى ئاشكارىلىدى. ئامېرىكا بۇنىڭدىن
ئىلگىرى كۇۋەيت، سەئۇدى ئەرەبىستانى ۋە ئەرەپ بىرلەشمە خەلپىلىكى قاتارلىق پارس
قولتۇقى دۆلەتلىرىگە «ۋەتەنپەرۋەر» بەلگىلىك باشقۇرىلىدىغان بومبىغا قارشى تۇرۇش
سېستىمىسىنى ئېكسپورت قىلغان ھەمدە دېڭىز قولتۇقىدىكى ئىتتىپاقداش دۆلەتلىرىدىن
بىرلىتە باشقۇرلىدىغان بومبىغا قارشى تۇرۇش سېستىمىسى قۇرۇپ ئىرانغا تاقابىل
تۇرۇشنى مۇراجىئەت قىلغان، لېكىن ئاكتىپ ئىنكاسقا ئېرىشەلمىگەنىدى. قاتارنىڭ
ئامېرىكىدا تۇرۇشلۇق باش ئەلچىسى مۇھەممەد جېھام كۇۋال قاتارنىڭ بۇ قېتىم
ئامېرىكىدىن باشقۇرىلىدىغان بومبىغا قارشى تۇرۇش سېستىمىسىنى ئۆز ئىچىگە ئالغان
قورال-ياراغ سېتىۋېلىشىنىڭ «مۇداپىئەلىنىش مەقسىتى ئۈچۈن»ئىكەنلىكىنى ھەمدە
ھەرقانداق بىر دۆلەتكە قارىتىلمىغانلىقىنى ئېيتىپ: «بىزنىڭ رايونىمىز نۇرغۇن
مالىمانچىلىقلارنى باشتىن ئۆتكۈزدى، بىزنىڭ قورال-ياراغ سېتىۋېلىشىمىز قاتارنى
قوغداش ئۈچۈن» دېدى.  تاراتقۇلار ئومۇمىيۈزلۈك ھالدا يېقىندا ئىمزالانغان قورال-
ياراغ سېتىش كېلىشىمى ئامېرىكا بىلەن قاتارنىڭ مۇناسىۋىتىنى، بولۇپمۇ ئىككى
دۆلەتنىڭ ھەربىي ئىشلار ساھەسىدىكى ھەمكارلىقىنى يەنىمۇ يېقىنلاشتۇرىدۇ، دەپ
قارىدى. ئامېرىكىنىڭ بىر يۇقىرى دەرىجىلىك ئەمەلدارى: «بۇ بىر تۈرلۈك ئۇزاق
مۇددەتلىك مەبلەغ سېلىش» دېدى. نۆۋەتتە قاتاردىكى ئۇدەيد ھاۋا ئارمىيە بازىسى
ئامېرىكىنىڭ ئوتتۇرا شەرق رايونىدىكى ئەڭ چوڭ ھاۋا ئارمىيە بازىسى. بۇ بازا
ئامېرىكىنىڭ «بىرلەشمە ھاۋا ئارمىيە ھەرىكەت مەركىزى» بولۇش سۈپىتى بىلەن،
ئافغانىستاندىكى ئامېرىكا ئارمىيەسىنىڭ كۆرەشچى ئايروپىلانلىرىنى كۈزىتىش ھەمدە
ئوتتۇرا شەرق رايونىدىكى ھاۋا قاتنىشىنى ئىز قوغلاپ نازارەت قىلىشقا مەسئۇل. كوبى
دۆلەت مۇداپىئە مىنىستىرى خاگېلنىڭ ئاتتىياھ بىلەن سۆھبەتلەشكەندە «ئامېرىكا
قاتارنىڭ ئامېرىكا ئارمىيەسىنى ياردەم بىلەن تەمىنلىگەنلىكىدىن مىننەتدار»
دېگەنلىكىنى ئېيتتى. كوبى يەنە ئامېرىكا بىلەن قاتار مۇناسىۋىتىنىڭ «ئىنتايىن
مۇھىم»لىقىنى، خاگېلنىڭ مۇناسىۋەتنى يەنىمۇ كۈچەيتىشنى ئۈمىد قىلىدىغانلىقىنى
بىلدۈردى.  بەش بۇرجەكلىك بىنا ئەمەلدارىنىڭ مۆلچەرلىشىچە بۇ قورال-ياراغ سېتىش
توختامى 54 مىڭ خىزمەت ئورنى يارىتىپ، ئامېرىكا ئىقتىسادىنىڭ تەرەققىياتىغا
تۈرتكە بولىدىكەن. «ۋەتەنپەرۋەر» بەلگىلىك باشقۇرىلىدىغان بومبىغا قارشى تۇرۇش
سېستىمىسىنىڭ قاتتىق دېتالى ۋە «يېنىك نەيزە» بەلگىلىك تانكىغا قارشى
باشقۇرىلىدىغان بومبىنى ئامېرىكىنىڭ رايتخېئون شىركىتى ۋە لوكخېد مارتىن شىركىتى
ياسىغان. «ئاپاچ» بەلگىلىك تىك ئۇچار ئايروپىلاننى بولسا ئامېرىكا بوئېن شىركىتى
ئىشلەپچىقارغان. بۇ قورال-ياراغ سېتىش توختىمىدا «ۋەتەنپەرۋەر» بەلگىلىك
باشقۇرىلىدىغان بومبىغا قارشى تۇرۇش سېستىمىسىنىڭ باھاسى تەخمىنەن 7 مىليارد
ئامېرىكا دوللىرى، «ئاپاچ» بەلگىلىك تىك ئۇچار ئايروپىلان ۋە مۇناسىۋەتلىك
زاپچاسلىرىنىڭ باھاسى 3 مىليارد ئامېرىكا دوللىرىدىن كۆپرەك، «يېنىك نەيزە»
بەلگىلىك تانكىغا قارشى باشقۇرىلىدىغان بومبىنىڭ باھاسى بولسا تەخمىنەن بىر
مىليارد ئامېرىكا دوللىرى ئىكەن.  نۆۋەتتە قاتار ھۆكۈمىتى ھاۋا ئارمىيەسىنىڭ
كۆرەشچى ئايروپىلانلىرىنى ئالماشتۇرۇشنى ئويلىشىۋاتىدۇ، تاللايدىغان ئايروپىلان
تىپلىرىدىن ئەنگلىيە ئاۋىئاتسىيە ئالەم قاتنىشى سېستىمىسى شىركىتىنىڭ «تەيفېڭ
بورىنى» تىپلىق كۆرەشچى ئايروپىلانى، فرانسىيە داسو شىركىتىنىڭ «ئۆتكۈنچى شامال»
بەلگىلىك كۆرەشچى ئايروپىلانى ھەمدە بوئېن شىركىتىنىڭ F－18 ۋە F－15 بەلگىلىك
كۆرەشچى ئايروپىلانى قاتارلىقلار بار ئىكەن. ئامېرىكا ئەمەلدارى ئامېرىكىنىڭ
قاتار بىلەن  بۇ قېتىم ئىمزالىغان قورال-ياراغ سېتىش توختامىنى «بىر ياخشى
بېشارەت»، قاتارنىڭ ئاخىرىدا ئامېرىكا شىركەتلىرىدە ئىشلەپچىقىرىلغان كۆرەشچى
ئايروپىلانلارنى سېتىۋېلىشنى تاللىشىغا پايدىلىق بولۇشى مۇمكىن دەپ قارىغان.

تور بېتىمىزدىكى يېڭىلىقلار

 ئىستانسىمىز ئاپتونوم رايونلۇق خەلق ھۆكۈمىتى كەسپتىكى ناتوغرا ئىستېللارنى
تۈزىتىش ئىشخانىسى، ئاپتونوم رايونلۇق قاتناش ـ ترانسپورت نازارىتى، ئاپتونوم
رايونلۇق جامائەت خەۋپسىزلىكى نازارىتى بىلەن بىرلىشىپ ئۆتكۈزىدىغان، كەسپ
ئىستېلى قىزىق لېنىيىسى پروگراممىسىنىڭ «قاتناش 10 مىڭ چاقىرىملىق سەپىرى چوڭ
تىپتىكى بىۋاستە ئاڭلىتىش پائالىيىتى» 10- ئاينىڭ 16- كۈنى يەنى ئەتە خوتەن
ۋىلايىتىدە  ئۆتكۈزىلىدۇ

ئىستانسىمىز ئاپتونوم رايونلۇق خەلق ھۆكۈمىتى كەسپتىكى ناتوغرا ئىستېللارنى
تۈزىتىش ئىشخانىسى، ئاپتونوم رايونلۇق قاتناش ـ ترانسپورت نازارىتى، ئاپتونوم
رايونلۇق جامائەت خەۋپسىزلىكى نازارىتى بىلەن بىرلىشىپ ئۆتكۈزىدىغان، « كەسپ
ئىستېلى قىزىق لېنىيىسى پروگراممىسىنىڭ قاتناش 10 مىڭ چاقىرىملىق سەپىرى چوڭ
تىپتىكى بىۋاستە ئاڭلىتىش پائالىيىتى مۇشۇ ئاينىڭ 18- كۈنى، يەنى ئەتە بورتالا
شەھىرىدە ئۆتكۈزىلىدۇ. رادىيو ئاڭلىغۇچىلىرىمىزنىڭ بورتالا مۇڭغۇل ئاپتونوم
ئوبلاستىنىڭ يول تىرانسىپورت، يۈك ـ ترانسپورت ۋە قاتناش ئىشلىرىغا ئالاقىدار
ئىنكاسلىرى بولسا پىروگراممىمىزنى  يىپ ئۇچى بىلەن تەمىنلىشىنى قارشى ئالىمىز.
