# شىنجاڭ ئۇيغۇر ئاپتونۇم رايونلۇق خەلق ھۆكۈمىتىنىڭ ئالاقە تورىنىڭ ئۇچۇر
بىخەتەرلىكىگە بولغان باشقۇرۇشنى كۈچەيتىش توغرىسىدىكى ئومومىي ئۇقتۇرۇشى

2015-01-06 17:01:24     مەنبەسى: شىنجاڭ خەلق رادىيو ئىستانسىسى  

1\. بۇ ئومۇمىي ئۇقتۇرۇش ئالاقە تورىدا زوراۋانلىق-تېررورلۇق ئۇچۇرلىرىنى
تەييارلاش، كۆپەيتىش، ئېلان قىلىش، تارقىتىش، ساقلاش قاتارلىق قانۇنغا خىلاپ
جىنايى قىلمىشلارغا قاتتىق زەربە بېرىپ، دۆلەتنىڭ بىخەتەرلىكى ۋە ئىجتىمائىي
مۇقىملىقىنى قوغداش مەقسىدىدە، مەملىكەتلىك خەلق قۇرۇلتىيى دائىمىي كومىتىتىنىڭ
«تور ئۇچۇرىنى قوغداشنى كۈچەيتىش توغرىسىدىكى قارار» ى، گوۋۇيۈەننىڭ «ئالاقە تورى
ئۇچۇر مۇلازىمىتىنى باشقۇرۇش چارىسى»، شۇنداقلا دۆلەت ۋە ئاپتونۇم رايوننىڭ
ئالاقىدار بەلگىلىمىسىگە ئاساسەن، ئاپتونۇم رايونىمىزنىڭ ئەمەلىيىتىگە
بىرلەشتۈرۈلگەن ئاساستا ئېلان قىلىندى.  
2\. ئالاقە تورى ئۇچۇر مۇلازىمەتچىلىرى ئابونىتچىلارنى تېز خەۋەرلىشىش، تور
قاتتىق دېسكىسى، توردا مۇڭدىشىش، ئۈن-سىن ماترىياللىرىدىن بەھىرلىنىش ۋە باشقا
تور ئىجتىمائىي ئالاقە سۇپىسى مۇلازىمىتى بىلەن تەمىنلەشتە ئىجازەت ئېلىش ياكى
رويخەتكە ئالدۇرۇش رەسمىيىتىنى قانۇن بويىچە بېجىرىشى ھەمدە بۇ ئومۇمىي
ئۇقتۇرۇشقا ئەمەل قىلىشى كېرەك.  
3\. ئاپتونۇم رايوننىڭ مەمۇرىي رايون تەۋەلىكىدە رويخەتكە ئالدۇرۇپ تىزىملاتقان،
بۇ ئومۇمىي ئۇقتۇرۇشنىڭ ئىككىنجى ماددىسىدا كۆرسىتىلگەن ئالاقەت تورى ئۇچۇر
مۇلازىمەت پائالىيىتى بىلەن شۇغۇللىنىدىغان ياكى ئاپتونۇم رايوننىڭ مەمۇرىي رايۇن
سىرتىدا رويخىەتكە ئالدۇرۇپ تىزىملاتقان، شىنجاڭدىكى ئابونتچىلارنى بۇ ئومۇمىي
ئۇقتۇرۇشنىڭ 2-ماددىسىدا كۆرسىتىلگەن ئالاقە تورى ئۇچۇر مۇلازىمەت پائاليىتى
بىلەن تەمىنلەيدىغانلار بىردەك ئاپتونۇم رايونىمىزنىڭ مەمۇرىي رايونى تەۋەلىكىدە
مۇلازىمەت تەسىس قىلىشى؛ مەخپىي شىفىر تېخىنىكىسى ئىشلىتىدىغانلار ئالاقىدار
مەسئۇل تارماقلارغا مۇناسىۋەتلىك ئۈسكۈنە سانلىق مەلۇمات ئۇچۇرلىرىنى يوللاپ،
رويخەتكە ئالدۇرىشى كېرەك.  
4\. ئالاقە تورى ئۇچۇر مۇلازىمەتچىلىرى ئابونىتچىلارنى ئۇچۇر ئېلان قىلىش
مۇلازىمىتى بىلەن تەمىنلەشتە، ئابونىتچىلار ھەققىي سالاھىيەت ئۇچۇرىنى تەمىنلىشى
كېرەك. ئابونتچى مەلۇم بىر شەخس بولسا، كىملىك ياكى باشقا كۈچكە ئىگە سالاھىيەت
ئىسپاتى ئارقىلىق رويخەتكە ئالدۇرۇپ تىزىملىتىشى؛ ئابونتچى مەلۇم ئورۇن بولسا،
تەشكىلىي ئاپپاراتنىڭ كود كېنىشكىسى تاكى تىجارەت كېنىشكىسى ئارقىلىق رويخەتكە
ئالدۇرۇپ تىزىملىتىشى كېرەك.  
5\. ئالاقە تورى ئۇچۇر مۇلازىمەتچىلىرىنىڭ تىزىمغا ئالدۇرغان ئابونتچىلارنىڭ
سالاھىيەت ئۇچۇرى مەخپىيەتلىكىنى ساقلاش مەسئۇلىيىتى بار. قانۇنىي تەرتىپتىن
ئۆتكۈزمىگەن ئەھۋالدا، باشقىلارنى ئۇچۇر بىلەن تەمىنلەشكە، ئۇچۇرنىڭ ئىشلىتىش
ئورنىنى ئۆزگەرتىشكە بولمايدۇ؛ ئۇچۇر مەزمۇنىنى ئاشكارلاشقا بۇرمىلاشقا،
بۇزغۇنچىلىققا ئۇچرىتىشقا، ئابونتچىلارنىڭ ھەقىقىي سالاھىيەت ئۇچۇرىدىن
پايدىلىنىپ پايدا ئېلىشقا بولمايدۇ.  
6\. ئالاقە تورى ئۇچۇر مۇلازىمەتچىلىرى ۋە ئابونتچىلارنىڭ تۆۋەندىكى مەزمۇنغا
دائىر ئۇچۇرلارنى تەييارلىشى، كۆپەيتىشى، ئېلان قىلىشى، تارقىتىشى ۋە ساقلىشى
مەنئىي قىلىنىدۇ:  
1) ئاساسىي قانۇندا بەلگىلەنگەن ئاساسسىي پىرىنسىپقا قارشى تۇرۇش، ئاساسىي قانۇن
ۋە قانۇن نىزاملارغا قارا چاپلاش؛  
2) دۆلەتنىڭ شان شەرىپى ۋە مەنپەئەتىگە زىيان سېلىش، دۆلەتنىڭ خەۋپسىزلىكىگە خەۋپ
يەتكۈزۈش، دۆلەتنىڭ مەخپىيىنى ئاشكارىلاش، دۆلەتنىڭ ھاكىمىيىتىنى ئاغدۇرۇش ۋە
دۆلەتنىڭ بىرلىكىگە بۇزغۇنچىلىق قىلىش؛  
3) «جىھاد قىلىش» ۋە مىللىي بۆلگۈنچىلىك قىلىشنى تەرغىپ قىلىش، مىللىي ئۆچمەنلىك،
مىللىي كەمسىتىشكە قۇترىتىش، مىللەتلەر ئىتتىپاقلىقىغا بۇزغۇنچىلىق قىلىش؛  
4) دىنىي ئەسەبىي ئىدىيەنى تارقىتىش، دۆلەتنىڭ دىن سىياسىتىگە بۇزغۇنچىلىق قىلىش،
بىدئەتلىك ۋە فىئوداللىقنى تەرغىپ قىلىش؛  
5) ئىغۋا توقۇش ۋە تارقىتىش، جەمئىيەتنىڭ تەرتىپىنى قالايمىقانلاشتۇرۇش،
ئىجتىمائىي مۇقىملىققا بۇزغۇنچىلىق قىلىش؛  
6) زوراۋانلىق، تېررورلۇققا دائىر ئىدىيە ۋە ئۈن-سىن ماتىرياللىرىنى تارقىتىش،
پارتلاتقۇچ دورا ۋە پارتلىتىش قۇرۇلمىسى، قورال-ياراغ، باشقۇرۇلىدىغان ئەسۋاب،
خەتەرلىك بۇيۇم ياساش، ئىشلىتىش قاتارلىق زوراۋانلىق-تېررورلۇق جىنايى
قىلمىشلىرىنى سادىر قىلىش ئۇسۇلى ۋە ماھارىتىنى ئۆگىتىش؛  
7) زوراۋانلىق ۋاستىسى ئارقىلىق باشقىلارنىڭ ھاياتى ۋە ئومۇمنىڭ، شەخىسنىڭ مال-
مۈلۈك بىخەتەرلىكىگە زىيان يەتكۈزۈشكە قۇترىتىش؛  
8) شەھۋانىيلىق، ئىشقىۋازلىق، قىمارغا دائىر ياكى جىنايەت سادىر قىلىشقا
كۈشكۈرتىدىغان مەزمۇنلارنى تارقىتىش؛  
9) باشقىلارغا ھاقارەت قىلىش ياكى تۆھمەت چاپلاش، باشقىلارنىڭ قانۇنلۇق ھوقۇق-
مەنپەئەتىنى زىيانغا ئۇچرىتىش؛  
10) قانۇن-نىزامدا چەكلەنگەن باشقا مەزمۇنلار.  
7\. بۇ ئومۇمىي ئۇقتۇرۇشنىڭ 6-ماددىسىدىكى بەلگىلىمىگە خىلاپ قىلمىش كۆرۈلسە،
ئالاقەت تورى ئۇچۇر مۇلازىمەتچىلىرى دەرھال ئۇچۇر ئېلان قىلىش، تارقىتىشنى
توختىتىش ياكى ھېساب نومۇرى تاقىۋېتىلگەنگە قەدەر مۇناسىۋەتلىك ئۇچۇرلارنى
ئۆچۈرۋېتىش، يېڭىلىنىشىنى ۋاقتىنچە توختىتىش قاتارلىق بىر تەرەپ قىلىش
تەدبىرلىرىنى قوللىنىشى ھەمدە ئالاقىدار خاتىرىسىنى ساقلاپ، مۇناسىۋەتلىك
باشقۇرۇش تارماقلىرىغا ۋاقتىدا دوكلات قىلىشى كېرەك.  
بۇ ئومۇمىي ئۇقتۇرۇشنىڭ 6-ماددىسدىكى بەلگىلىمىگە خىلاپ قىلمىش كۆرۈلسە،
ھەرقانداق ئورۇن ياكى ھەرقانداق شەخسنىڭ بىردەك ئالاقىدار تارماقلارغا دوكلات
قىلىش ھوقۇقى ۋە مەجبۇرىيىتى بار.  
بۇ ئومۇمىي ئۇقتۇرۇشنىڭ 6- ماددىسىدىكى بەلگىلىمىگە خىلاپ قىلمىش كۆرۈلسە،
ئالاقىدار مەسئۇل تارماقلار ئۆز مەسئۇلىيەت دائىرىسىدە قانۇن بويىچە
مەسئۇلىيىتىنى ئادا قىلىپ، تېخنىكىلىق تەدبىر ياكى باشقا زۆرۈر تەدبىرلەرنى
قوللىنىپ، ئۇنى توسۇشى، تەكشۈرۈپ بىر تەرەپ قىلىشى كېرەك.  
8\. بۇ ئومۇمىي ئۇقتۇرۇشنىڭ 3-ماددىسىدىكى بەلگىلىمىگە خىلاپ قىلمىش كۆرۈلسە،
ئالاقىدار مەسئۇل تارماقلار ئۆز مەسئۇلىيەت دائىرىسىدە قانۇن بويىچە ئالاقە تورى
ئۇچۇر مۇلازىمەتچىلىرىنىڭ ئاپتونۇم رايوننىڭ مەمۇرىي رايونى تەۋەسىدە ئالاقە تورى
ئۇچۇر مۇلازىمىتى بىلەن شۇغۇللىنىشىنى چەكلەش ياكى ئاخىرلاشتۇرۇش توغرىسىدا قارار
چىقىرىشى كېرەك.  
بۇ ئومۇمىي ئۇقتۇرۇشتىكى بەلگىلىمىگە خىلاپ باشقا قىلمىشلار ئالاقىدار قانۇن-
نىزام بويىچە بىر تەرەپ قىلىنىدۇ.  
9\. بۇ ئومۇمىي ئۇقتۇرۇش ئېلان قىلىنىشتىن بۇرۇن ئالاقە تورى ئۇچۇر مۇلازىمىتى
پائالىيىتى بىلەن شۇغۇللانغانلار بۇ ئومۇمىي ئۇقتۇرۇش ئېلان قىلىنىپ 90 كۈن
ئىچىدە، بۇ ئومۇمىي ئۇقتۇرۇشنىڭ 3-ماددىسىدىكى بەلگىلىمە بويىچە مۇلازىمېتىر
