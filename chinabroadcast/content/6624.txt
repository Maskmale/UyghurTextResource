ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر يىلى تەخمىنەن مىڭدىن مىڭغىچە ئەيدىز بىمارى ئۆلۈپ كېتىدىكەن كورىيە ئارمىيە دائىرىلىرى چاۋشيەن يادرو سىنىقى ئېلىپ بېرىشقا تەييارلىنىپ بولدى دېدى قىزىق نوقتا خەۋەرلىرى تارىختىكى بۈگۈن مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى يىلى بۈگۈنكى كۈندە ئەنگىلىيە دېڭىز ئارمىيە ئادمىرالى سىمورنىڭ قوماندانلىقىدىكى ئەنگىلىيە ئامېرىكا گېرمانىيە فرانسىيە رۇسىيە ياپونىيە ئىتالىيە ئاۋىستىرىيە قاتارلىق سەككىز دۆلەت بىرلەشمە ئارمىيەسىنىڭ مىڭ كىشىلىك قوشۇنى داگۇدىن يولغا چىقىپ تەيەنجىن ئارقىلىق بېيجىڭغا يۈرۈش قىلماقچى بولغاندا لاڭفاڭ بىلەن ياڭسۇندا يېخېتۈەنچىلەر بىلەن ۋەتەنپەرۋەر چىڭ سۇلالىسى ئارمىيەسىنىڭ توسۇپ زەربە بېرىشىگە ئۇچراپئېغىر تالاپەتكە ئۇچرىغاندىن كېيىن چىكىنىپ كەتكەن يىلى بۈگۈنكى كۈندە ئەنگلىيەدىكى ئاياللار تۇنجى قېتىم ئىچكى كابېنتقا كىرگەن ئەنگلىيەنىڭ يېڭىدىن سايلانغان باش ۋەزىرى جېمىس موسولىن ئەنگىلىيە بىلەن ئامېرىكىغا ئۇرۇش ئېلان قىلغان شۇ كۈنى يەنە كانادا ئىتالىيەگە ئۇرۇش ئېلان قىلغان يەنە شۇ كۈنى نورۋىگىيە گېرمانىيەگە تەسلىم بولغان يىلى بۈگۈنكى كۈندە ماۋ زېدۇڭ جۇڭخۇا مەملىلىكەتلىك تەنتەربىيە باش جەمئىيىتىگە تەنتەربىيە ھەرىكىتىنى راۋاجلاندۇرۇپ خەلقنىڭ ساغلاملىقىنى ئاشۇرايلى دېگەن بېغىشلىمىنى يېزىپ بېرىپ تەنتەربىيە ھەرىكىتى بىلەن ساغلاملىقنى ئاشۇرۇشنىڭ ئىچكى باغلىنىشىنى ئېنىق كۆرسىتىپ بەرگەن جۇڭگو سوتسىيالىستىك تەتنتەربىيە ئىشلىرىنىڭ خەلق ئۈچۈن خىزمەت قىلىش مەقسىدى ۋە ۋەزىپىسىنى ئېنىق بەلگىلەپ بەرگەن شۇنىڭدىن باشلاپ ئېلىمىزنىڭ ئاممىۋى تەنتەربىيە ھەرىكىتى
