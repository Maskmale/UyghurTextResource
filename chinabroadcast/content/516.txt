ئەسلىمە ئۈنۋېرسال رادىيو سەنئەت رادىيو كەسىپ ئىستىلى تاللاپ ئاڭلاش ئۇيغۇرچە ئۈگۈنۈش خەنزۇچە ئۈگۈنۈش ئىستانسىمىز ھەققىدە ئاپتونۇم رايوننىڭ لىكسىيە ئۆمىكى پارتىيە نۆۋەتلىك مەركىزىي كومىتېتى ئومۇمىي يىغىنى ۋە ئاپتونۇم رايونلۇق پارتكوم نۆۋەتلىك كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ھەققىدە دوكلات بەردى ئاۋامنىڭ كەڭ زىمىندا ئىگىلىك تىكلىشى بىپايان جەنۇبىي شىنجاڭ زېمىنىدا كىشىلىك ھاياتنىڭ گۈزەل كارتىنىسىنى يارىتايلى ئاپتونوم رايونلۇق ئىنتىزام تەكشۈرۈش كومىتېتى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە جاڭ چۈنشيەن شۇجىنىڭ مۇھىم سۆزىنىڭ روھىنى يەتكۈزدى ئاپتونوم رايونلوق خەلق قۇرۇلتىيى دائىمىي كومىتېتى پارتگۇرۇپپىسى مەركىزى ئۆگۈنۈش گۇرۇپپىسى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە ئاپتونوم رايونلوق پارتكوم خەلق قۇرۇلتىيى خىزمىتى يىغىنىنىڭ روھىنى ئۆگەندى مىسىر زوڭتوڭ سايلىمىدا مۇسۇلمان بۇرادەرلەر جەمئىيىتىگە تەۋە ھەققانىيەت پارتىيېسىنىڭ رەئىسى مورشى بىلەن سابىق زوڭلى سەفىقنىڭ بېلەتكە ئېرىشىش نىسبىتى ئالدىنقى ئىككى ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر يىلى تەخمىنەن مىڭدىن مىڭغىچە ئەيدىز بىمارى ئۆلۈپ كېتىدىكەن كورىيە ئارمىيە دائىرىلىرى چاۋشيەن يادرو سىنىقى ئېلىپ بېرىشقا تەييارلىنىپ بولدى دېدى قىزىق نوقتا خەۋەرلىرى ئىقتىسادىي تۈزۈلمە ئىسلاھاتىنى زىچ چۆرىدەپ ئىقتىسادنىڭ ھەم ياخشى ھەم تېز تەرەققىياتىنى ئىلگىرى سۈرۈش مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى ئىستانسىمىز خەۋىرى يىقىندا ئاياغلاشقان ئاپتونوم رايوننىڭ ئىككى يىغىنى ئاپتونوم رايوننىڭ بۇ يىلقى تۈرلۈك ئاساسىي ۋەزىپىسىنى بېكىتتى ھەمدەئاپتونوم رايوننىڭ بۇ يىللىق خىزمىتىنىڭ يۆنىلىشنى ئايدىڭلاشتۇرۇپ بۇ تۈرلۈك خىزمەتنىڭ ئاساسىي ئىدىيىسى رىيال ئەھمىيىتى ۋە تەدبىر كاپالىتىنى تەپسىلىي تونۇشتۇردى بۇ يىلقىھۆكۈمەت خىزمىتىدىن دوكلات تا مۇنۇلار ئوتتۇرىغا قۇيۇلدى دۆلەت ئىلكىدىكى مال مۈلۈكنى باشقۇرۇش تۈزۈلمە ئىسلاھاتى بىلەن دۆلەت ئىلكىدىكى كارخانىلار ئىسلاھاتىنى چوڭقۇرلاشتۇرۇپ غەيرىي دۆلەت كاپىتالىنى دۆلەت كارخانىلىرىغا پاي ئورنىدا قېتىش يەرلىك كارخانىلار مەركەزگە قاراشلىق كارخانىلارغا پاي قېتىشنى ئىلگىرى سۈرۈپ ئارىلاش مۈلۈكچىلىكتىكى ئىگىلىكنى زور كۈچ بىلەن تەرەققىي قىلدۇرۇش كېرەك ئاپتونوم رايونلۇق پارتىيە مەكتىپى ئىقتىسادشۇناسلىق كافېدراسىنىڭ مۇدىرى دوتسېنت لى يەن مۇنداق دېدى ئاپتونوم رايونىمىزدا ئۇزۇندىن بۇيان دۆلەت ئىلكىدىكى كارخانىلار نىسبەتەن كۆپ سالماقنى ئىگىلەپ كەلگەن ئەمما شىنجاڭدا ئىشقا ئورۇنلاشتۇرۇش مالىيە كىرىمىنى ئاشۇرۇش خەلق تۇرمۇشىنى ياخشىلاش بىلەن چەكلىنىپ قالسا بولمايدۇ چۈنكى چوڭ كارخانا چوڭ گۇرۇھلارنىڭ پايدىسى باش ئىشتابقا تاپشۇرۇپ بېرىلىدۇ شۇڭا بۇنداق ئەھۋالدا ئارىلاش مۈلۈكچىلىكتىكى ئىگىلىكنى يولغا قۇيۇپ شىنجاڭنىڭ ئۆزىنىڭ پۇقراۋى ئىگىلىكى پۇقراۋى كارخانىلىرىنى زور كۈچ بىلەن تەرەققىي قىلدۇرۇش ئىنتايىن زۆرۈر بۇنداق بولغاندا بىرىنچىدىن شىنجاڭ تېخىمۇ كۆپ مالىيە كىرىمىنى ساقلاپ قالالايدۇ ئىككىنچىدىن تېخىمۇ كۆپ كىشىلەرنى ئىشقا ئورۇنلاشتۇرغىلى بۇلىدۇ ئۈچىنچىدىن شىنجاڭنىڭ سانائەتلەشتۈرۈش سەۋىيىسىنى يەنىمۇ ئۇزارتىپ تېخنىكىسىنى ئاشۇرغىلى بۇلىدۇ ھۆكۈمەت خىزمىتىدىن دوكلاتتا مۇنۇلار ئوتتۇرىغا قويۇلدى ئاپتونوم رايونىمىز ئىقتىسادىنىڭ ھەم ياخشى ھەم تېز تەرەققىياتىنى ئىلگىرى سۈرۈشتە رايون شەھەر يېزىلارنى ماس تەرەققىي قىلدۇرۇشنى جەزمەن بىر تۇتاش پىلانلاش تەڭرىتاغنىڭ شىمالىي يانباغرىدىكى ئىقتىساد بەلبىغىنىڭ تەرەققىيات ئۇسۇلىنى ئۆزگەرتىش ۋە قۇرۇلمىسىنى ئەلالاشتۇرۇپ دەرىجىسىنى ئۆستۈرۈشنى تېزلىتىش جەنۇبىي شىنجاڭدىكى ئۈچ ۋىلايەت ئوبلاستنىڭ تەرەققىياتىنى تېزلىتىشكە مەدەت بېرىدىغان يېتەكچى پىكرىنى ئوتتۇرىغا قويۇش كېرەك بۇنىڭغا قارىتا لى يەن مۇنداق دېدى ئىقتىسادىي تەرەققىياتنى بىر تۇتاش پىلانلاشتا ئىشنى دۆلەت زېمىنىدىكى بوشلۇقنى ئېچىش ئەندىزىسىنى ئەلالاشتۇرۇشتىن باشلاش كېرەك ئىقتىدار رايونىنى بېكتكەندىن كېيىن ئالايلۇق شىمالىي شىنجاڭ رايونىدا زامانىۋىي كەسپنى زور كۈچ بىلەن تەرەققىي قىلدۇرۇپ شىنجاڭنى ئىقتىسادىي ئاساس بىلەن تەمىنلەش كېرەك جەنۇبىي شىنجاڭدىكى ئۈچ ۋىلايەت ئوبلاستقا نىسبەتەن ئېلىپ ئېيتقاندا ئۇنى سىياسىي جەھەتتە باھالاشتا نامراتلارنى يۆلەش كۆرسەتكۈچىگە مايىللاشتۇرۇش ئىكولوگىيىلىك مۇھىت كۆرسەتكۈچىگە مايىللاشتۇرۇش كېرەك شۇنىڭ ئۈچۈن ھازىرقى بۇ يول
