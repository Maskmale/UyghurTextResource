ئەسلىمە ئۈنۋېرسال رادىيو سەنئەت رادىيو كەسىپ ئىستىلى تاللاپ ئاڭلاش ئۇيغۇرچە ئۈگۈنۈش خەنزۇچە ئۈگۈنۈش ئىستانسىمىز ھەققىدە ئاپتونۇم رايوننىڭ لىكسىيە ئۆمىكى پارتىيە نۆۋەتلىك مەركىزىي كومىتېتى ئومۇمىي يىغىنى ۋە ئاپتونۇم رايونلۇق پارتكوم نۆۋەتلىك كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ھەققىدە دوكلات بەردى ئاۋامنىڭ كەڭ زىمىندا ئىگىلىك تىكلىشى بىپايان جەنۇبىي شىنجاڭ زېمىنىدا كىشىلىك ھاياتنىڭ گۈزەل رايونلوق خەلق قۇرۇلتىيى دائىمىي كومىتېتى پارتگۇرۇپپىسى مەركىزى ئۆگۈنۈش گۇرۇپپىسى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە ئاپتونوم رايونلوق پارتكوم خەلق قۇرۇلتىيى خىزمىتى يىغىنىنىڭ روھىنى ئۆگەندى مىسىر زوڭتوڭ سايلىمىدا مۇسۇلمان بۇرادەرلەر جەمئىيىتىگە تەۋە ھەققانىيەت پارتىيېسىنىڭ رەئىسى مورشى بىلەن سابىق زوڭلى سەفىقنىڭ بېلەتكە ئېرىشىش نىسبىتى ئالدىنقى ئىككى ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر يىلى تەخمىنەن مىڭدىن مىڭغىچە ئەيدىز بىمارى ئۆلۈپ كېتىدىكەن كورىيە ئارمىيە دائىرىلىرى چاۋشيەن يادرو سىنىقى ئېلىپ بېرىشقا تەييارلىنىپ بولدى دېدى قىزىق نوقتا خەۋەرلىرى شىنجاڭ ئىقتىسادىي ئىلمىي ھالقىما تەرەققىيات يولىغا قاراپ يۈكسەلمەكتە مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى ئىستانسىمىزنىڭ شىنجاڭ ئىقتىسادىي ئىلمىي ھالقىما تەرەققىيات يولىغا قاراپ يۈكسەلمەكتە دىگەن مەخسۇس ماقالىسى شىنجاڭنىڭ توكىنى سىرتقا يەتكۈزۈشتىكى نۇقتىلىق قۇرۇلۇش بولغان شەرقىي جۇڭغار شەرقىي جۇڭگو مىڭ كىلوۋولتلۇق بىۋاستە توك يەتكۈزۈش قۇرۇلۇشىنىڭ يۈرۈشلەشكەن ئېلېكتىر مەنبەسى تۈر قۇرۇلۇشى نۆۋەتتە شەرقىي جۇڭغاردىكى چەكسىز كەتكەن كۆمۈرلۈكتە جىددىي ئېلىپ بېرىلماقتا شەرقىي جۇڭغار شەرقىي جۇڭگو مىڭ كىلوۋولتلۇق بىۋاستە توك يەتكۈزۈش قۇرۇلۇش تۈرىنىڭ ئومۇمى ئۇزۇنلىقى مىڭ كىلومېتىر بولۇپ گەنسۇ نىڭشيا شەنشى خېنەن قاتارلىق ئۆلكە ئاپتونوم رايونلارنى كىسىپ ئۆتۈپ ئاخىرىدا ئەنخۇيغا تۇتىشىدۇ ئۇ نۆۋەتتە دۇنيادىكى توك يەتكۈزۈش مۇساپىسى ئەڭ يىراق قۇرۇلۇش بولۇپ ھىساپلىنىدۇ يۈرۈشلەشكەن توك تارقىتىش تۈر قۇرۇلۇشىدا بۇ يىل ئايدا ئىش باشلانغان بولۇپ پۈتكۈل قۇرۇلۇش تاماملانغاندىن كىيىن سانائەت ئومۇمى مەھسۇلات قىممىتىنى مىليارد يۈەنگە مالىيە كىرىمىنى مىليارد يۈەنگە يەتكۈزگىلى مىڭ كىشىنىڭ ئىشقا ئورۇنلىشىشىنى ئىشقا ئاشۇرغىلى بولىدۇ شەرقىي جۇڭغار ئىقتىساد تېخنىكا تەرەققىيات رايونىنى باشقۇرۇش كومىتېتىنىڭ مۇدىرى شياۋ رېنجۈن مۇنداق دېدى شەرقىي جۇڭغارشەرقىي جۇڭگو توك يەتكۈزۈش قۇرۇلۇشى تاماملانغاندىن كىيىن توك يەتكۈزۈش ئارىلىقى ئەڭ يىراق توك يەتكۈزۈش سىغىمچانلىقى ئەڭ چوڭ ئېلېكتىر بېسىمىنىڭ دەرىجىسى ئەڭ يۇقىرى بولۇشتىن ئىبارەت تە ئەڭ يۇقىرى قۇرۇلۇشنى تاماملايمىز بۇنىڭ توك يەتكۈزۈش مىقدارى تەخمىنەن مىليارد كىلوۋاتقا يىتىدۇ بۇ مىليون توننا كۆمۈرنى سىرتقا توشۇغانغا باراۋەر كىلىدۇ پۈتكۈل شىنجاڭنىڭ كەڭ زىمىنىدا پۈتۈن مەملىكەت ھەتتا دۇنيادا تۇنجى بولۇپ بارلىققا كىلىۋاتقان ئۇل مۇئەسەسسە قۇرۇلۇش تۈرلىرى ۋە سانائەت تۈرلىرى بارغانسىرى كۆپىيىپ شىنجاڭنىڭ ئىقتىسادىي ئەمەلىي كۈچىنىڭ كۈنسىرى زورىيىشىغا تۈرتكە بولماقتا ئاپتونوم رايون قۇرۇلغان يىلدىن بۇيان شىنجاڭدا نېفىت خىمىيە سانائىتى كۆمۈر ئېلېكتىر قۇۋۋىتى رەڭلىك مېتال ئاساس قىلىنغان ئالاھىدە كەسپ سىستېمسى تەدرىجى شەكىللەندى بولۇپمۇ يېقىنقى يىلدىن بۇيان سانائەت ئىقتىسادىي تېز سۈرئەتتە تەرەققى قىلىشتەك ۋەزىيەت بارلىققا كەلدى يىلى پۈتكۈل شىنجاڭنىڭ سانائەت ئاشۇرۇلما قىممىتى مىليارد مىليون يۈەنگە يىتىپ يىلدىكىدىن ئىككى ھەسسىدىن كۆپرەك ئاشتى شىنجاڭنىڭ ئىقتىسادىي تەرەققىياتىنىڭ زورىيىشى مەملىكەت بويىچە ئالدىنقى قاتاردا دۇنيا بويىچە ئىلغار بولغان بىر تۈركۈم شىنجاڭ كارخانىلىرىنىڭ بارلىققا كىلىشىگە تۈرتكە بولدى شىنجاڭ ئالاھىدە ترانسىفورماتور سانائىتى چەكلىك پاي شىركىتى مېك گۇرۇھى گۇاڭخۈي گۇرۇھى جۇڭتەي خىمىيە سانائىتى گۇرۇھى چەكلىك پاي شىركىتى ۋەكىللىكىدىكى بىر تۈركۈم يەرلىك كارخانىلار بارلىققا كىلىپ يەرلىكنىڭ سانائەتنى تەرەققى قىلدۇرۇشتىكى ئاساسىي قوشۇنغا ئايلاندى نۆۋەتتە شىنجاڭ نېفىت نېفىت خىمىيە يېڭى تىپتىكى كۆمۈر خىمىيە سانائىتى ئېلېكتىر قۇۋۋىتى رەڭلىك مېتال ماشىنىسازلىق ئۈسكۈنەئەسۋاپ ياساش قاتارلىق مەھسۇلات قىممىتى مىليارد يۈەندىن ئاشىدىغان كەسپلەر توپىنى بارلىققا كەلتۈرۈشكە تۇتۇش قىلماقتا ئاپتونوم رايونلۇق پارتىيە مەكتىپىنىڭ پروفسسورى جى سۇلىن مۇنداق دەپ ھىساپلىدى بۇ زامانىۋى كەسپلەر سىستمېىسى شىنجاڭنىڭ ئىقتىسادىنىڭ تېخىمۇ تېز تەرەققى قىلىشىغا تۈرتكە بولىدۇ ھەر مىللەت خەلقىمۇ تېخىمۇ كۆپ تەرەققىيات نەتىجىلىرىدىن بەھرىمەن بولالايدۇ ئىلگىرى شىنجاڭدا نېفىتتىن ئىبارەت بىرلا چوڭ كەسپ بار بولغان بولسا ھازىر نېفىت كۆمۈر ئېلېكتىر قۇۋۋىتى رەڭلىك مېتال ياساش سانائىتى قاتارلىق چوڭ سانائەتلەر بار بولدى ھەر قايىسى كەسپلەر تەرەققى قىلىپ شىنجاڭنىڭ كەسپ قۇرۇلمىسىنى كۆرۈنەرلىك ياخشىلاپ شىنجاڭنىڭ ئىقتىسادىي ئومۇمى مىقدارىنى زور دەرىجىدە ئاشۇردى شۇنىڭ بىلەن بىر ۋاقىتتا ئىشقا ئورۇنلاشتۇرۇشنى ئىلگىرى سۈرۈپ ھەر مىللەت ئاممىسىنىڭ تۇرمۇش سۈپىتىنى ئۆستۈردى سانلىق مەلۇماتقا نەزەر مەركەزنىڭ شىنجاڭ خىزمىتى سۆھبەت يىغىنى ئېچىلغاندىن بۇيان شىنجاڭنىڭ ئىشلەپچىقىرىش ئومۇمى قىممىتى يىلىغا ئوتتۇرا ھىساپ بىلەن مىليارد يۈەندىن ئېشىپ ئۇدا يىل مىليارد يۈەنلىك يۈەنلىك چوڭ ئۆتكەل بۆسۈپ ئۆتۈلدى شىنجاڭنىڭ رايون ئىشلەپچىقىرىش ئومۇمى قىممىتىنىڭ يىللىق ئوتتۇرىچە ئېشىش نىسبىتى گە يىتىپ پۈتۈن مەملىكەتنىڭ ئوخشاش مەزگىلدىكى ئوتتۇرىچە ئېشىش سەۋىيىسىدىن پوئىنىت يۇقىرى بولدى ئېشىش سۈرئىتى مەملىكەت بويىچە يىلدىكى ئورۇندىن يىلى بىراقلا ئورۇنغا ئۆتتى مۇخبىرلىق خاتىرىسى تەرەققىيات بارلىق مەسىلىلەرنى ھەل قىلىشنىڭ ئاساسى ئاپتونوم رايون قۇرۇلغان يىلدىن بۇيان شىنجاڭنىڭ ئىقتىسادىي ئەمەلىي كۈچى ئۆزلۈكسىز زورايدى مالىيە كۈچى تەدرىجى ئاشتى كىشىلەرنىڭ تۇرمۇش سۈپىتى يىلمۇ يىل ئۆستى كىيىنكى قەدەمدە بىز پارتىيە مەركىزىي كومىتېتىنىڭ رەھبەرلىكىدە شىنجاڭدىكى ھەر مىللەت خەلقىنى تېخىمۇ كۆپ تەرەققىيات نەتىجىلىرىدىن بەھرىمەن قىلىمىز ئۇلارنىڭ تۇرمۇشىنىڭ تېخىمۇ بەختلىك بولىدىغانلىقىغا ئىشىنىمىز كۆرۈپ قويۇڭ ئۇيقۇنى ياخشىلايدىغان يېمەكلىكلەر گۈزەللىك قەيسەرلىكتە
