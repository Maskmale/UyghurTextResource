ئەسلىمە ئۈنۋېرسال رادىيو سەنئەت رادىيو كەسىپ ئىستىلى تاللاپ ئاڭلاش ئۇيغۇرچە ئۈگۈنۈش خەنزۇچە ئۈگۈنۈش ئىستانسىمىز ھەققىدە ئاپتونۇم رايوننىڭ لىكسىيە ئۆمىكى پارتىيە نۆۋەتلىك مەركىزىي كومىتېتى ئومۇمىي يىغىنى ۋە ئاپتونۇم رايونلۇق پارتكوم نۆۋەتلىك كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ھەققىدە دوكلات بەردى ئاۋامنىڭ كەڭ زىمىندا ئىگىلىك تىكلىشى بىپايان جەنۇبىي شىنجاڭ زېمىنىدا كىشىلىك ھاياتنىڭ گۈزەل كارتىنىسىنى يارىتايلى ئاپتونوم رايونلۇق ئىنتىزام تەكشۈرۈش كومىتېتى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە جاڭ چۈنشيەن شۇجىنىڭ مۇھىم سۆزىنىڭ روھىنى يەتكۈزدى ئاپتونوم رايونلوق خەلق قۇرۇلتىيى دائىمىي كومىتېتى پارتگۇرۇپپىسى مەركىزى ئۆگۈنۈش گۇرۇپپىسى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە ئاپتونوم رايونلوق پارتكوم خەلق قۇرۇلتىيى خىزمىتى يىغىنىنىڭ روھىنى ئۆگەندى مىسىر زوڭتوڭ سايلىمىدا مۇسۇلمان بۇرادەرلەر جەمئىيىتىگە تەۋە ھەققانىيەت پارتىيېسىنىڭ رەئىسى مورشى بىلەن سابىق زوڭلى سەفىقنىڭ بېلەتكە ئېرىشىش نىسبىتى ئالدىنقى ئىككى ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر يىلى تەخمىنەن مىڭدىن مىڭغىچە ئەيدىز بىمارى ئۆلۈپ كېتىدىكەن كورىيە ئارمىيە دائىرىلىرى چاۋشيەن يادرو سىنىقى ئېلىپ بېرىشقا تەييارلىنىپ بولدى دېدى قىزىق نوقتا خەۋەرلىرى ئېغىزدىكى غەيرى پۇراقنى يوقتىش ئالاھىدىلىكى بار يىمكلىكلەر مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى ئىغىز سىسىق پۇراش بولسا ئادەتتە ئىغىز تازلىقى ياخشى بولماسلىقيەنە ئاشقازان ئۈچەي ئىقتىدارىنىڭ قالايمىقانلىشىدىن كىلىپ چىقىدىغان توقتىشتا بىر ئىستىكان سۈت ئىچسەك غەيرى پۇراق يوقايدۇ لىمون بىر ئىستىكان ئىسسىق سۇغا ئازراق يالپۇز سىلىپ ئاندىن يىڭى لىمۇننى سىلىپ ئىچكەندە ئىغىزدىكى غەيرى پۇراقنى يوقتىدۇ يۇمغاقسۇت يۇمغاقسۇتمۇ ئىغىزدىكى غەيرى پۇراقنى يوقتىش ئالاھىدىلىكىگە ئىگە قېتىق تەتقىقاتلاردا ئىسپاتلىنىشچە ھەركۈنى قېتىق ئىچىشنى داۋاملاشتۇرۇپ بەرگەندە ئىغىز بوشلۇقىدىكى ھىدروگىن سولڧىتنىڭ مىقدارىنى تۆۋەلىتىنىش رولى بار
