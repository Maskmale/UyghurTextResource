 ئاۋات ناھىيىسى مىڭ مو يەرگە شالغۇت سورتلۇق كېۋەز تېرىغان قانچە كۈندە بىر ئالماشتۇرىسىلەر ئوخشاش كەسىپتىكىلەر قان سېلىش ئۆمۈر ئۇزارتىدۇ يەنە ئىچەمدۇق تېتىپ بېقىش تىلەك تىلەش ۋارقىراش بالام ئاپاڭ قېنى ئېلان بېرىش كەسىپ ئىستىلى قىزىق لىنيىسى ئالاھىدە پروگراممىسى لېي فىڭدىن ئۆگىنىش ئۆلگىلىرى ياش ناخشىچىلار شادلىق مىنۇتلىرى پرگوراممىسىدا ئاپتونوم رايونلۇق قېتىملىىق پارتىيە قۇرۇلتىيى ئاساسىي قاتلامغا بارايلى گۈزەل شىنجاڭ يېپەك يولىغا سەپەر مەملىكەتلىك رادىئو ئىستانسىسىلىرىنىڭ چوڭ تىپتىكى بىرلەشمە زىيارەت قىلىش پائالىيىتى تۈنجى نۆۋەتلىك جۇڭگو ئاسىيا ياۋروپا يەرمەنكىسى جۇڭگو شىنجاڭ خەلقئارا مىللەتلەر ئۇسسۇل بايرىمى ئەخلاق نەمۇنىچىلىرى نامزاتلىرى ئاددىي كىشىلەرنىڭ يېڭى ھىكايىلىرى سۈمۈرۈش تىپىدىكى يېمەكلىك ماشىنىسى كەشىپ قىلىندى ئىلىمپەن ساھەسىدىكى ئەڭ يېڭى بايقاشلار كەلگۈسىدە ئېلېكتىروموبىللارغا يېشىل ئېنېرگىيە ئىشلىتىلىدۇ كەلگۈسىدە نېفىت نەدىن ئېلىنىدۇ ھەرخىل لىفتلار دۇنيانىڭ قىياپىتىنى تۈپتىن ئۆزگەرتكەن سەككىز پاتېنت مۇخبىرىمىز دولان كەنتىدىكى يۇقىرى مەھسۇلاتلىق شالغۇت سورت ئىشلەپچىقىرىشىدا ئۈلگە كۆرسىتىش كېۋەزلىكىدە ئاۋات ناھىيىلىك تېخنىكا كېڭەيتىش پونكىتىنىڭ مۇئاۋىن باشلىقى پەرىزات ئىزىزنىڭ چۈشەندۈرۈشنى ئاڭلىماقتا شىنجاڭ خەلق رادىيو ئىستانسىسى خەۋەرلەر تورى ئۆز مۇخبىرىمىز ماھىرە يۈسۈپ خەۋىرىفوتوسى ئاۋات ناھىيىسى جۇڭگودىكى پاختا شەھىرى ئۇزۇن تالالىق پاختا ماكانى دېگەندەك شەرەپلەرگە ئىرىشكەن پاختىچىلىق كەسپىي دېھقانلىرىنىڭ تۈۋرۈكلۈك كەسىپىگە ئايلانغان ناھىيىلەرنىڭ بىرى بىراق يېقىنقى يىللاردىن بۇيان ھەر يىلى ئاينىڭ ئوتتۇرىلىرىدىن ئاينىڭ ئوتتۇرىلىرىغىچە كۆرۈلىۋاتقان سىلتسىيە گرادۇستىن يۇقىرى تىنجىق ھاۋارايى ئاۋاتتا تېرىلىدىغان ئۇزۇن تالالىق كېۋەزنىڭ گۈلغۇنچىلىرىنى تۆكۈپ مەھسۇلاتىغا ئېغىر زىيان سالماقتا شۇ سەۋەبلىك يىلىدىن بۇيان ئاۋات ناھىيىسىدىكى ئالاقىدار تارماقلار ئىنچىكە تالالىق پاختىنىڭ ئۆسۈپ يېتىلىش ئالاھىدىلىكى مەھسۇلات مىقدارى جەھەتتىكى ئەۋزەللىكى بىلەن ئۇزۇن تالالىق پاختىنىڭ سۈپىتىنى چىقىش قىلغان ھالدا خەيلۇۋ شالغۇت سورتىنى يېتىشتۈرۈپ ھەر يىلى بەلگىلىك كۆلەمدە كېڭەيتىپ ئىلغار سورت ئارقىلىق پاختىنىڭ بىرلىك مەھسۇلاتى شۇ ئارقىلىق دېھقانلارنىڭ پاختىچىلىق كىرىمىنى ئاشۇرۇشنىڭ يېڭى يولى ئۈستىدە ئىزدىنىۋاتاتتى بىراق شالغۇت سورتنىڭ ئۇرۇقلۇقنىڭ ھەر كىلوگرامىنىڭ باھاسى يۈەن بولۇپ ئادەتتىكى ئۇرۇقلۇق باھاسىدىن كۆرىنەرلىك يۇقىرى بولىۋاتقاچقا گەرچە ناھىيە مالىيەسىدىن شالغۇت سورت تېرىغان دېھقانلارغا ھەر كىلوگرامغا يۈەندىن سورتلۇق ئۇرۇقلۇق ياردەم پۇلى بېرىش ئالاقىدار تېخنىكىلىق تارماقلارتۈجۈپىلەشكەن تېرىقچىلىقىنى كېڭەيتىپ ئۇرۇق تەننەرق سېلنمىسىنى ئاشۇرۇش ئۇسۇللىرىنى قوللىنۋاتقىنى بىلەن دېھقانلارنىڭ قوبۇل قىلىشى يەنىلا تەس بولماقتا ئىگىلىنشىچە بۇ يىل ئاۋات ناھىيىسى مىڭ مو يەرگە شالغۇت سورتلۇق كېۋەز تېرىغان بولۇپ نۆۋەتتىكى ئەھۋالدىن قارىغاندا شالغۇت سورتنىڭ مو بېشى ئوتتۇرىچە چىگىتلىك پاختا مەھسۇلاتى كىلوگرامغا يېتىپ ئاۋات دېھقانلىرى ئۇزاقتىن بۇيان تېرىپ كېلىۋاتقان ئۇزۇن تالالىق ۋە ئىنچىكە تالالىق بىرلىك پاختا مەھسۇلاتىدىن كىلوگرام ئۆپچۆرىسىدە يۇقىرى مەھسۇلات ئېلىنغان باھاسىمۇ بۇ يىل باھاسى بىر قەدەر ياخشى بولغان ئىنچىكە تالالىق پاختا باھاسى بىلەن ئوخشاش بولغان ئاۋات ناھىيىسى دولان كەنتىدىكى مولۇق شالغۇت سورت تېرىقچىلىقىدا ئۈلگە كۆرسىتىش كېۋەزلىكى
