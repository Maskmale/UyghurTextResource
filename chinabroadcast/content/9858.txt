ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر ھۆكۈمىتى يېزا بازارلىق پونكىتلارنى باشقۇرۇش تۈزۈلمىسى ئىسلاھاتىنى ئىلگىرى سۈرۈش توغرىسىدىكى يېتەكچى پىكرى ئېلان قىلدى مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى ئىستانسىمىز خەۋىرى بۇ يىل ئايدا ئاپتونوم ئىسلاھات تەرتىپلىك يولغا قويۇلدى يېقىندا ئاپتونوم رايونلۇق پارتكوم ئىسلاھاتنى ئومۇميۈزلۈك چوڭقۇرلاشتۇرۇش كومىتېتى ئىشخانىسى پىكىر نى چوڭقۇر شەرھىيلەپ جايلار تارماقلارنىڭ ئىسلاھاتنى ئىلگىرى سۈرۈشتە ئىدىيەنى بىرلىككە كەلتۈرۈپ ئورتاق تونۇشنى مۇجەسسەملەپ بىرىكمە كۈچ ھاسىل قىلىش لازىملىقىنى بىلدۈردى يېزا بازار ھاكىمىيىتى يۇقىرىدا دۆلەتكە تۆۋەندە يېزا كەنت جەمئىيىتىگە باغلىنىدۇ يېزا بازارلىق پونكىتلار دېھقان چارۋىچىلارنى ئاممىۋىي مۇلازىمەت بىلەن تەمىنلەيدىغان ئاساسلىق ۋاستە يېزا بازارلىق پونكىتلارنى باشقۇرۇش تۈزۈلمە ئىسلاھاتى ئاساسىي قاتلام ھاكىمىيەت قۇرۇلۇشى بولۇپ يېزا بازارلارنىڭ فونكسىيەلىك رولىنى جارى قىلدۇرۇشقا بىۋاستە تەسىر كۆرسىتىدۇ نۆۋەتتە ئاپتونوم رايونىمىزدىكى ۋىلايەت ۋىلايەت ئوبلاست شەھەردە جەمئىي يېزا بازار مىڭ يېزا بازارلىق پونكىت بار ئۇلارنىڭ ئىچىدە كەسپىي يېزا بازارلىق پارتكوم ھۆكۈمەتلەرنىڭ مەسئۇلىيىتىنى ئادا قىلىشى بىلەن پونكىتلارنى باشقۇرۇش تۈزۈلمىسىدە ماسلاشماسلىق ئەمەلىيلەشمەسلىك خاھىشلىرى كۆرۈلدى ماسلىشىش قىيىن بولۇش باشقۇرۇش ئوربېتىسىدىن چىقىپ
