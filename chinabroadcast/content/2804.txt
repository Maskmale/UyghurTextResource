 پاكلانغان روھ دېگەن درامما ئاڭلىتىلغان كۆرسىتىلگەندىن كېيىن جايلاردىكى ئاممىنىڭ بىردەك ياخشى باھاسىغا ئېرىشتى بىر پارچە خەت يېڭى ئۆزگىرىشلەرنى خەۋەرلەندۈردى ئويغىنىش ناملىق دىراما ئونسۇ ناھىيەسىنىڭ ھەر قايسى يېزا بازارلىرىدا سەييارە قويۇلدى ھەمدە شۇ يەردىكى ئاممىنىڭ بىردەك ياخشى باھاسىغا ئېرىشتى ئويغىنىش ناملىق دراما سەھنىدە كۆرسىتىلگەندىن كېيىن ياخشى باھاسىغا ئېرىشتى ئويغىنىش ناملىق دراما سەھنىدە كۆرسىتىلگەندىن كېيىن ياخشى باھاسىغا ئېرىشتى ئەسەبىيلىكنى تۈگىتىش دىرامما ھەپتىلىكى ئۈچۈن تەييارلىغان ئايخان ئانا ناملىق دىراممىسى شىنجاڭ سەنئەت يۇرتى تىياتىرخانىسىدا ئورۇندالدى ئون نەچچە مىللىي سودا شىركىتىنىڭ مەسئوللىرى ئۈرۈمچىگە جەم بولۇپ مۇقىملىق ئىتتىپاقلىق تەرەققىيات تىمىسىدا سۆھبەت ئۆتكۈزدى ئاپتونوم رايونىمىزنىڭ ئەسەبىيلىكنى تۈگىتىش تىياتىر ھەپتىلىكى تاماشىبىنلارنىڭ كۈچلۈك ئىنكاسىنى قوزغىدى ئاپتونوم رايونلۇق پارتكوم تەشۋىقات بۆلۈمىنىڭ ئورۇنلاشتۇرۇشى بىلەن ئاپتونوم رايونىمىزنىڭ ئەسەبىيلىكنى تۈگىتىش دىرامما ھەپتىلىكى باشلاندى يادرولۇق قىممەت قارىشىنى ئەخلاق ۋە ئۆرپئادىتىمىزگە يېتەكچى قىلايلى ئاپتونوم رايونىمىزدىكى ئەدەبىيات سەنئەت خىزمەتچىلىرى ئەدەبىيات سەنئەت ئىشلىرىنى تىرىشىپ راۋاجلاندۇرۇپ خەلق ئۈچۈن ياخشى مۇلازىمەت قىلىمىز دېيىشتى ئەسەبىيلىكنى تۈگىتىش دىرامما ھەپتىلىكى ئۈچۈن تەييارلىغان ئايخان ئانا ناملىق دىراممىسى شىنجاڭ سەنئەت يۇرتى تىياتىرخانىسىدا ئورۇندالدى بۇ بەتنىڭ نەشىر ھوقۇقى شىنجاڭ خەلق رادىيو ئىستانسىسىغا تەۋە ئالاقىلىشىش نومۇرى ئادرىسىمىز ئۈرۈمچى شەھىرى ئىتتىپاق يولى نومۇر شىنجاڭ خەلق رادىيو ئسىتانسىسى تور مەركىزى
