ئەسلىمە ئۈنۋېرسال رادىيو سەنئەت رادىيو كەسىپ ئىستىلى تاللاپ ئاڭلاش ئۇيغۇرچە ئۈگۈنۈش خەنزۇچە ئۈگۈنۈش ئىستانسىمىز ھەققىدە ئاپتونۇم رايوننىڭ لىكسىيە ئۆمىكى پارتىيە نۆۋەتلىك مەركىزىي كومىتېتى ئومۇمىي يىغىنى ۋە ئاپتونۇم رايونلۇق پارتكوم نۆۋەتلىك كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ھەققىدە دوكلات بەردى ئاۋامنىڭ كەڭ زىمىندا ئىگىلىك تىكلىشى بىپايان جەنۇبىي شىنجاڭ زېمىنىدا كىشىلىك ھاياتنىڭ گۈزەل كارتىنىسىنى يارىتايلى ئاپتونوم رايونلۇق ئىنتىزام تەكشۈرۈش كومىتېتى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە جاڭ چۈنشيەن شۇجىنىڭ مۇھىم سۆزىنىڭ روھىنى يەتكۈزدى ئاپتونوم رايونلوق خەلق قۇرۇلتىيى دائىمىي كومىتېتى پارتگۇرۇپپىسى مەركىزى ئۆگۈنۈش گۇرۇپپىسى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە ئاپتونوم رايونلوق پارتكوم خەلق قۇرۇلتىيى خىزمىتى يىغىنىنىڭ روھىنى ئۆگەندى مىسىر زوڭتوڭ سايلىمىدا مۇسۇلمان بۇرادەرلەر جەمئىيىتىگە تەۋە ھەققانىيەت پارتىيېسىنىڭ رەئىسى مورشى بىلەن سابىق زوڭلى سەفىقنىڭ بېلەتكە ئېرىشىش نىسبىتى ئالدىنقى ئىككى ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر يىلى تەخمىنەن مىڭدىن مىڭغىچە ئەيدىز بىمارى ئۆلۈپ كېتىدىكەن كورىيە ئارمىيە دائىرىلىرى چاۋشيەن يادرو سىنىقى ئېلىپ بېرىشقا تەييارلىنىپ بولدى دېدى قىزىق نوقتا خەۋەرلىرى ئىلى قازاق ئاپتونوم ئوبلاستىنىڭ تۇنجى نۆۋەتلىك دالىدا موتسىكىلىت ھەيدەپ توساقتىن ئاتلاش ۋە موتسىكلىت چاپتۇرۇش مۇسابىقىسى ئۆتكۈزۈلدى مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى ئىستانسىمىز خەۋىرى ئىلى قازاق ئاپتونوم ئوبلاستىنىڭ تۇنجى نۆۋەتلىك دالىدا موتسىكىلىت ھەيدەپ توساقتىن ئاتلاش ۋە موتسىكلىت چاپتۇرۇش مۇسابىقىسى يېقىندا غۇلجا شەھىرىدىكى گۈزەل غەربىي يۇرت چاڭغا تېيىلىش مەيدانىدا ئۆتكۈزۈلدى ھەر قايسى مۇسابىقە تۈرلىرىگە قاتنىشىدىغان ماھىرلار غۇلجا شەھىرىدىكى گۈزەل غەربىي يۇرت چاڭغا تېيىلىش مەيدانىدا كەسكىن مۇسابىقىلەشتى موتسىكىلىتلەر بىرىنىڭ كەينىدىن بىرى ھەر خىل توساقلاردىن ئاتلاپ ئۆتۈپ بىر دەم تۈز يەردە ماڭسا بىر دەم بوشلۇقتا پەرۋاز قىلدى ماھىرلار ئۆزلىرىنىڭ بارلىق ماھارەتلىرىنى ئىشقا سېلىپ مۇسابىقە مەيدانىنى قاينام تاشقىنلىققا چۆمدۈردى مۇسابىقىگە قاتناشقان ماھىر ئىلى قارلىق بۆرە دالا موتسىكىلىتىنى كېڭەيتىش كۇلۇبىنىڭ مەسئۇلى ليۇ جېندوڭ مۇخبىرغا مۇنداق دېدى دالا موتسىكىلىتى ھەرىكىتى دەسلەپكى تەرەققىيات باسقۇچىدا تۇرۇۋاتقانلىقتىن بىز تۇنجى قېتىملىق دالا موتسىكىلىت مەخسۇس مۇسابىقىسىنى ئۆتكۈزدۇققارلىق بۆرە دالا موتسىكىلىتىنى كېڭەيتىش كۇلۇبى ئارقىلىق ئىلى ئوبلاستىنىڭ دالا موتسىكىلىت ھەرىكىتىگە تۈرتكە بولىمىز نۆۋەتتە بىزدە ئۈندىدار توپى بار توپتىكى كىشىلەر دىن ئاشىدۇ ئۇلار ياشتىن ياشقىچە ئارىلىقتىكى كىشىلەر تىكەس ناھىيىسىدىن كەلگەن ماھىر لى فېي بىر ئاشخانىىنىڭ خوجايىنى بولۇپ بۇ قېتىم ئۇ بىرىنچىلىكنى قولغا كەلتۈردى يىل ئىلگىرى لى فېي دالىدا موتسىكىلىت چاپتۇرۇشنى ياقتۇرۇپ قالغان ھەمدە كۆپ قېتىم مۇسابىقىگە قاتنىشىش تەجرىبىسىگە ئىگە ئىدى ئۇنىڭ نەزىرىدە ئىلى ئوبلاستى دالا موتسىكىلىت ھەرىكىتى تۈرىنى قانات يايدۇرۇشتىكى جەننەت جاي ھىسابلىناتتى ئۇ مۇنداق دېدى ئىلى ئوبلاستىدا بىزگە تەۋە مۇسابىقە مەيدانلىرى بىر قەدەر كەسپلەشكەن بۇنىڭدىن كىيىن مۇسابىقە ئىشلىرىنىڭ بارغانسىرى كۆلەملىشىپ كۆلىمىنىڭ بارغانسىرى زورىيىشىنى سىرتتىكى ماھىرلارنىڭ بۇ پائالىيەتكە قاتنىشىشىنى ئۈمىد قىلىمەن موتسىكىلىت ماھىرى جې لى مۇنداق دېدى ئىلى ئوبلاستى موتسىكىلىت چاپتۇرىدىغان ئەڭ ئىسىل جاي بولۇپ كىلىماتى بىر قەدەر ياخشى ھەر خىل تاغ يوللىرى ئىچكىرىي ئۆلكىلەردە يوق بۇ بىر ناھايىتى ياخشى ئالماشتۇرۇش سۇپىسى شۇنداقلا ئىلىنىڭ ئوبرازىنى كېڭەيتىشتىكى بىر سۇپا ھىساپلىنىدۇ بۇ قېتىمقى مۇسابىقىنى ئۇيۇشتۇرغۇچى ئورۇن بولغان ئىلى گۈزەل غەربىي يۇرت ساياھەت چەكلىك شىركىتىنىڭ مۇئاۋىن دېرىكتورى لى جېڭ مۇنداق دېدى كەلگۈسىدە غۇلجا شەھىرىدىكى شىمالىي تاغ باغرىنى ئىلى ئوبلاستىنىڭ دالا تەنتەربىيە ھەرىكىتى ئېلىپ بارىدىغان بازا ۋە چوڭ لاگىر قىلىپ قۇرۇپ چىقىمىز باغچا رايونىنىڭ بىرىنچى قارارلىق كۆچەت تىكىش ھەرىكىتى ئاساسىي جەھەتتىن تاماملاندى كىيىن ئۆزىمىزنىڭ ئالاھىدىلىكىمىزگە بىرلەشتۈرۈپ كۆپ خىل ئىگىلىكنى تەرەققى قىلدۇرىمىز بىز دالا ھەرىكىتى ئاساسىي تېما قىلىنغان باغچىنى مۇھىم نۇقتا قىلىپ چاڭغا تېيىلىشتىن سىرت موتسىكىلىت ئاپتوموبىل ۋېلسىپىتنى دالىدا چاپتۇرۇشنى ئۆز ئىچىگە ئالغان كىلومېتىرلىق ۋېلسىپىت مۇسابىقە يولىنى قۇرۇپ چىقتۇق بىز بۇ يەرنى ئىلى ئوبلاستىنىڭ دالا ھەرىكىتى ئېلىپ بارىدىغان چوڭ لاگىر قىلىپ
