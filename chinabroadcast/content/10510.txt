ئەسلىمە ئۈنۋېرسال رادىيو سەنئەت رادىيو كەسىپ ئىستىلى تاللاپ ئاڭلاش ئۇيغۇرچە ئۈگۈنۈش خەنزۇچە ئۈگۈنۈش ئىستانسىمىز ھەققىدە ئاپتونۇم رايوننىڭ لىكسىيە ئۆمىكى پارتىيە نۆۋەتلىك مەركىزىي كومىتېتى ئومۇمىي يىغىنى ۋە ئاپتونۇم رايونلۇق پارتكوم نۆۋەتلىك كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ھەققىدە دوكلات بەردى ئاۋامنىڭ كەڭ زىمىندا ئىگىلىك تىكلىشى بىپايان جەنۇبىي شىنجاڭ زېمىنىدا كىشىلىك ھاياتنىڭ گۈزەل كارتىنىسىنى يارىتايلى ئاپتونوم رايونلۇق ئىنتىزام تەكشۈرۈش كومىتېتى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە جاڭ چۈنشيەن شۇجىنىڭ مۇھىم سۆزىنىڭ روھىنى يەتكۈزدى ئاپتونوم رايونلوق خەلق قۇرۇلتىيى دائىمىي كومىتېتى پارتگۇرۇپپىسى مەركىزى ئۆگۈنۈش گۇرۇپپىسى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە ئاپتونوم رايونلوق پارتكوم خەلق قۇرۇلتىيى خىزمىتى يىغىنىنىڭ روھىنى ئۆگەندى مىسىر زوڭتوڭ سايلىمىدا مۇسۇلمان بۇرادەرلەر جەمئىيىتىگە تەۋە ھەققانىيەت پارتىيېسىنىڭ رەئىسى مورشى بىلەن سابىق زوڭلى سەفىقنىڭ بېلەتكە ئېرىشىش نىسبىتى ئالدىنقى ئىككى ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر يىلى تەخمىنەن مىڭدىن مىڭغىچە ئەيدىز بىمارى ئۆلۈپ كېتىدىكەن كورىيە ئارمىيە دائىرىلىرى چاۋشيەن يادرو سىنىقى ئېلىپ بېرىشقا تەييارلىنىپ بولدى دېدى قىزىق نوقتا خەۋەرلىرى ئىلى ئوبلاستى توقۇلما كىيىم كىچەك كەسپىنى زور كۈچ بىلەن تەرەققىي قىلدۇردى مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى يېقىنقى يىللاردىن بۇيان ئىلى ئوبلاستى دۆلەتنىڭ ئىتىبار سىياسەتلىرىدىن پايدىلىنىپ توقۇلما كىيىم كىچەك كەسپىنى زور كۈچ بىلەن تەرەققىي قىلدۇردى مەركەزنىڭ شىنجاڭ خىزمىتى سۆھبەت يىغىنىدىن كىيىن ئاپتونوم رايون توقۇلما كىيىم كىچەك كەسپىنى جانلاندۇرۇپ ئىشقا ئورۇنلىشىشنى ئىلگىرى سۈرۈش توقۇلما كىيىم كىچەك كەسپىنىڭ تەرەققىياتىنى قوللاش پىلانىنى ئوتتۇرغا قويدى بۇ ئىلى ئوبلاستىنىڭ توقۇلما كىيىم كىچەك كەسپىگە تېپىلغۇسىز تەرەققىيات پۇرسىتىنى ئېلىپ كەلدى ئىلى ئوبلاستلىق كىيىم كىچەك جەمىيىتىنىڭ رەئىسى باۋ شىياۋيىڭنىڭ كىيىم كىچەك كەسپى بىلەن شۇغۇللانغىنىغا نەچچە ئون يىل بولغان بولۇپ ئىلىنىڭ كىيىم كىچەك كەسپىدىكى داڭلىق شەخىس بولۇپ قالغانىدى يىلى ئۇ ئىلى كۈنمۇ كىيىم كىچەك شىركىتىنى قۇرۇپ ئىلى كىيىم كىچەك تاۋار ماركىسىنى بەرپا قىلىشقا كۈچ چىقاردى توقۇلما كىيىم كىچەك كەسپى تەرەققىيات پىلانىنىڭ يولغا قويۇلىشى باۋ شىياۋيىڭنى ئىلھاملاندۇردى ئۇ باشقا كارخانىچىلار بىلەن بىللە ئىلى ئوبلاستىنىڭ توقۇلما كىيىم كىچەك كەسپى ئارقىلىق ئىشقا ئورۇنلىشىش يېڭى يولىنى ئاچتى يېقىنقى ئۈچ يىلدىن بۇيان شىركەت ئاپتونوم رايوننىڭ مەبلەغ جەھەتتىن قوللىشىدا زور يۈكسىلىشلەرنى قولغا كەلتۈردى نۆۋەتتە شىركەت تېخنىكا بويىچە تەربىيىلەش ئىقتىساسلىقلارنى كىرگۈزۈش بازار ئېچىش جۈملىدىن سۇ توك باج قاتارلىق جەھەتلەردە ئون تۈرلۈك ئىتىبار سىياسىتىنى پۇرسەت بىلىپ توقۇلما كىيىم كىچەك كەسپىنىڭ تەرەققىياتىنى ئىلگىرى سۈرۈپ تېخىمۇ ھاياتى كۈچكە ئىگە بولدى ئىگەللىنىشىچە نۆۋەتتە باۋ شىياۋيىڭ دىن ئارتۇق ئىشىچى قوبۇل قىلغان غالىم غۇلجا شەھەر تاشكۆۋرۈك يېزىسىدىكى دېھقان بولۇپ ئۇنىڭ تېرىلغۇ يېرى ئاز بولغاچقا ئائىلە ئىقتىسادىي ئىزچىل ياخشى ئەمەس ئىدى غالىم يىلى كۈنمۇ شىركىتىدە مۇنتىزىم تەربىيىلەنگەندىن كېيىن كەسپى ئىشچىغا ئايلاندى بۇنىڭ بىلەن بىر ئائىلە كىشىلىرىنىڭ تۇرمۇشى كۈندىن كۈنگە ياخشىلاندى غالىم مۇنداق دېدى بۇ يەردە ئىشلىگەندىن بۇيان تۇرمۇشىم ياخشىلاندى بىر ئايدا مىڭ يۈەندىن ئارتۇق ئىش ھەققى ئالىمەن ئاپتونوم رايون توقۇلما كىيىم كىچەك كەسپىنىڭ تەرەققىياتىنى تىزلىتىش پىلانىنى يولغا قويغاندىن بۇيان ئىلى ئوبلاستىنىڭ كىيىم كىچەك كەسپى ئۇچقاندەك تەرەققىي قىلدى ئىلگىركى تارقاق ئاز ئاجىز بولۇشتەك ۋەزىيەت بارغانسىرى كۆلەملىشىش كەسىپلىشىشكە قاراپ يۈزلەندى نۆۋەتتە بىر تۈركۈم توقۇلما كىيىم كىچەك كەسپى بازىلىرى ئارقا ئارقىدىن ئىلى ئوبلاستىدا باش كۆتۈرۈپ ئېشىنچە ئەمگەك كۈچلىرىنىڭ ئىشقا ئورۇنلىشىشىنى ئىلگىرى سۈرۈپ يەرلىكنىڭ ئىقتىسادى تەرەققىياتىغا تۈرتكە بولدى بۇلتۇر بىر يىلدىلا ئىلى ئوبلاستىدىكى تۈر ئاپتونوم رايوننىڭ توقۇلما كىيىم كىچەك كەسپى بويىچە ئىشقا ئورۇنلاشتۇرۇش ھەركىتى لاھىيىسىگە كىرگۈزىلدى پىلاندىكى تۈر ئومۇمى سېلىنمىسى مىليارد مىليون يۈەن بولۇپ بۇلتۇر مىليارد مىليون يۈەن مەبلەغ سېلىندى ئاپتونوم رايونغا توققۇز توقۇلما كىيىم كىچەك كەسپى باغچىسىنىڭ ئۇل مۇئەسسەسەلەر قۇرۇلىشى تۈرى پىلانى يوللاندى تۈر سېلىنما پىلانى مىليارد مىليون يۈەن بولۇپ بۇلتۇر مىليون يۈەن مەبلەغ سېلىندى مىڭ كىۋادرات مېتىرلىق زاۋۇت ئۆيى قۇرۇلۇشى تاماملاندى بۇلتۇرنىڭ ئاخىرىغا قەدەر ئىلى ئوبلاستىغا بىۋاستە قاراشلىق جايلار بويىچە توقۇلما كىيىم كىچەك كەسپى كارخانىسى كە سېتىش كىرىمى مىليارد مىليون يۈەنگە بۇ كەسىپ بىلەن شۇغۇللانغۇچىلار مىڭ كىشىگە يېتىپ يىلدىكىدىن ۋە ئاشتى بىر تۈركۈم توقۇلما كىيىم كىچەك بازىسى ئىلى ئوبلاستىدىكى ھەر قايسى ناھىيە شەھەرلەرگە ماكانلىشىپ ئىشقا ئورۇنلاشتۇرۇشتىكى مۇھىم كۈچكە ئايلاندى بۇ يىل ئىلى ئوبلاستىغا
