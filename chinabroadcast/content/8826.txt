 غەنىجاننىڭ توشقان بالىلىرى ماۋزۇلۇق كىرىشتۈرۈلمە زىيارەت خاتىرىسى قانچە كۈندە بىر ئالماشتۇرىسىلەر ئوخشاش كەسىپتىكىلەر قان سېلىش ئۆمۈر ئۇزارتىدۇ يەنە ئىچەمدۇق تېتىپ بېقىش تىلەك تىلەش ۋارقىراش بالام ئاپاڭ قېنى ئېلان بېرىش كەسىپ ئىستىلى قىزىق لىنيىسى ئالاھىدە پروگراممىسى لېي فىڭدىن ئۆگىنىش ئۆلگىلىرى ياش ناخشىچىلار شادلىق مىنۇتلىرى پرگوراممىسىدا ئاپتونوم رايونلۇق قېتىملىىق پارتىيە قۇرۇلتىيى ئاساسىي قاتلامغا بارايلى گۈزەل شىنجاڭ يېپەك يولىغا سەپەر مەملىكەتلىك رادىئو ئىستانسىسىلىرىنىڭ چوڭ تىپتىكى بىرلەشمە زىيارەت قىلىش پائالىيىتى تۈنجى نۆۋەتلىك جۇڭگو ئاسىيا ياۋروپا يەرمەنكىسى جۇڭگو شىنجاڭ خەلقئارا مىللەتلەر ئۇسسۇل بايرىمى ئەخلاق نەمۇنىچىلىرى نامزاتلىرى ئاددىي كىشىلەرنىڭ يېڭى ھىكايىلىرى سۈمۈرۈش تىپىدىكى يېمەكلىك ماشىنىسى كەشىپ قىلىندى ئىلىمپەن ساھەسىدىكى ئەڭ يېڭى بايقاشلار كەلگۈسىدە ئېلېكتىروموبىللارغا يېشىل ئېنېرگىيە ئىشلىتىلىدۇ كەلگۈسىدە نېفىت نەدىن ئېلىنىدۇ ھەرخىل لىفتلار دۇنيانىڭ قىياپىتىنى تۈپتىن ئۆزگەرتكەن سەككىز پاتېنت شىنجاڭ خەلق رادىيو ئىستانسىسى خەۋەرلەر تورى ئۆز مۇخبىرىمىز نىجات ئەخمەت ئابدۇرىشىت ئابدۇغېنىلارنىڭ غەنىجاننىڭ توشقان بالىلىرى ماۋزۇلۇق كىرىشتۈرۈلمە زىيارەت خاتىرىسى باي بولۇشنىڭ يوللىرى خىلمۇ خىل بولىدۇ ئەمما ئۆزىنىڭ قۇربىتىگە يارىشا ئىشنىڭ پېشىنى تۇتۇپ شۇ ئىشنى بىر باشقا ئېلىپ چىقىش ئۈچۈن جان تىكىپ تىرىشقاندىلا ئاندىن ھالال بېيىش ئارزۇسىغا يەتكىلى بولىدۇ بىز كۈزنىڭ مېھىرلىك ئاپتىپى تەڭرىتاغنىڭ جەنۇبىنى تېخىمۇ جەلىپكارلىققا تويۇندۇرغان ئۆكتەبىر ئېيىدا ئاتۇش شەھىرىدە زىيارەتتە بولغاندا تاسسادىپىي پۇرسەتتە ئاتۇش شەھىرى سۇنتاغ يېزىسى ۋاق ۋاق كەنتىدىكى دېھقان غەنىجان ئابدۇسەمىنىڭ ئۇزۇن تىۋىتلىق توشقان بېقىپ مۇشۇ ساھەدە ئۇيغۇر دېھقانىلىرى ئىچىدە پۇل تاپقان دېھقانغا ئايلانغانلىقىنى ئاڭلاپ قالدۇق بىزنى ھەيران قالدۇرغىنى ۋە جەلىپ قىلغىنىمۇ دەل ئەزەلدىنلا يەرگە تايىنىپ تىرىكچىلىك ھارۋىسىنى سۆرەپ كىلىۋاتقان بىر ئۇيغۇر دېھقاننىڭ باقمىچىلىق قىلغاندىمۇ قوي كالا توخۇ كەپتەر باقمىچىلىقى ئەمەس ئۇيغۇرلار ئەزەلدىن قىلىپ باقمىغان ئۇزۇن تىۋىتلىق توشقان باقمىچىلىقى بىلەن شۇغۇللانغانلىقى بولدى بىز غەنىجاننىڭ قورۇسىغا كىرىپ رەت رەت سېلىنغان ئۈچ قەۋەتلىك توشقان كاتەكلىرىنى كاتەك ئىچىدىكى كۆزلىرى قىزىل تۈكلىرى ئاپپاق پارقىراق ئۇزۇن تىۋىتلىق توشقانلارنى كۆرگىنىمىزدىلا ئۇنىڭغا قايىل بولدۇق غەنىجان ئابدۇسەمى يېڭىدىن داۋاملىق كۆپىيىۋاتقان توشقانلارغا ئۆي سېلىش ئۈچۈن سىمۇنت تاختا ياساۋاتقان ئىكەن ئۇ بىزنى كۆرۈپ چاقچاق ئارىلاش توشقان بالىلىرىمغا ئۆي سېلىشنىڭ تەييارلىقىنى قىلىۋاتقان بۇ توشقانلارنى بېقىش بەك كۆڭۈللۈك دېدى خوشال ھالدا غەنىجان ئابدۇسەمى ئادەتتە دېھقانچىلىقنىڭ ئارىسال ۋاقىتلىرىدا ئۆي ئاقارتىش سىرچىلىق ئىشلىرىنى قىلىپ قويغاندىن باشقا ھۆنىرى يوق ئىدى يىلى ئايدا ئۇ قىزىلسۇ ئوبلاستلىق ياشلار دېھقانچىلىق مەيدانىدىكى خەنزۇ تونۇشى لاۋۋىينىڭ ئۆيىگە بىر ئىش بىلەن بېرىپ قالىدۇ ئۇلار كونا تونۇشلاردىن بولۇپ لاۋۋىي توخۇ باقاتتى غەنىجاننىڭ مەھەللىسىدە ئاچقان بىر دۇكىنى بار ئىدى لاۋۋىي توخۇلىرىنىڭ تۇخۇمىنى غەنىجاننىڭ دۇكىنىغا ئېلىپ چىقىپ قوياتتى غەنىجان تۇخۇملارنى سېتىپ قوياتتى ئەنە شۇنداق مۇناسىۋەت بۇ ئىككەيلەننى يېقىن قىلىپ قويغاندى غەنىجان لاۋۋىينىڭ قورۇسىغا كىرىپ سىتكىلىق كاتەكلەردىكى تۈكلىرى پارقىراپ تۇرغان توشقانلارنى كۆرۈپ قىزىقىپ قالدى لاۋۋىي غەنىجانغا ئۇزۇن تىۋىتلىق توشقاننىڭ سورتى ئېسىل كۆپىيىش ئىقتىدارى كۈچلۈك تۈكىنىڭ ئۆسۈشى تېز بېقىش ئاسان ئۇنىڭ ئۈستىگە ئۇنى باقسا ھۆكۈمەتنىڭ ئائىلە باقمىچىلىقىنى تەرەققىي قىلدۇرۇش تۈرى بويىچە توشقان بىلەن ھەقسىز تەمىنلىنىدىغانلىقىنى ئېيتىپ بەردى غەنىجان بۇلارنى ئاڭلاپ بۇنىڭ ئۆز ئىقتىسادىنى ياخشىلاشنىڭ بىر پۇرسىتى ئىكەنلىكىنى ھېس قىلىپ بىر تەۋەككۈل قىلىپ باقماقچى بولدى غەنىجاننىڭ ئۇزۇن تىۋىتلىق توشقان باقمىچىلىق پىلانىنى ئايالى رەشىدە خۇدابەردى قىزغىن قوللىدى يېزىلىق ھۆكۈمەتنىڭ ئىلتىماسىنى ۋاقتىدا تەستىقلىشى بىلەن ئاتۇش شەھەرلىك چارۋىچىلىق مال دوختۇرلۇق ئىدارىسى توشقان بىلەن ھەقسىز تەمىنلىدى شۇنداق قىلىپ ئۇنىڭ ئائىلىسىنىڭ ئۇزۇن تىۋىتلىق توشقان باقمىچىلىقى باشلاندى ئۇزۇن تىۋىتلىق توشقاننىڭ سورتى بىر قەدەر كۆپ بولۇپ غەنىجاننىڭ ئائىلىسىدە بېقىلىۋاتقىنى ئاق رەڭلىك ئۇزۇن تىۋىتلىق توشقان ئىدى ئۇ دەسلەپتە توشقان يېمىنى تەييارلاشتا قىينالدى ئۇ تەييار يەمنىڭ كىلوسىنى يۈەندىن ئالغان بولسىمۇ لېكىن توشقانلار ياقتۇرۇپ يېمىدى بۇ يەمنىڭ باھاسى قىممەت بولغاننىڭ ئۈستىگە توشقانلارنىڭ يېمىگەنلىكىنى كۆرۈپ تالقانلاش ماشىنىسى سېتىۋېلىپ ئۆزى بىدە ۋە كىپەك بىلەن يەم ياسىدى بۇنىڭ تەننەرخى يۈەندىن توختىغاننىڭ ئۈستىگە توشقانلارمۇ ياقتۇرۇپ يىدى شۇنداق قىلىپ ئۇ توشقان بالالىرىنىڭ قورسىقىنىڭ غېمىدىن قۇتۇلدى كېيىن ئاستا ئاستا توشقانلارغا دورا بېرىش ئوكۇل سېلىشنى ئۈگەندى پەقەت بولمىغاندا بۇ ئىشقا دالالەت قىلغان لاۋۋىينىڭ ئۆيىگە ياردەم سوراپ چاپاتتى زىيارەت داۋامىدا غەنىجان كاتەكتىكى توشقانلارنى ئېلىپ خۇددى بالىسىنىڭ دۈمبىسىنى سىلىغاندەك تۈكلىرىنى مېھرى بىلەن سىلايتتى مانا ھازىر ئۇنىڭ ئەسلىدكى توشقىنى كۆپىيىپ گە يېقىنلىشىپ قالغانىدى قىرقىغان تىۋىتىمۇ كىلوگرامدىن ئاشقان ئىگىلىشىمىزچە ھەر كىلوگرام توشقان تىۋىتىنىڭ ھازىرقى بازار باھاسى يۈەندىن ئاشىدىكەن توشقاننىڭ مايىغى ساپ تەبىئىي ئوغۇت بولغاچقا ئۇ ئوغۇت چىقىمىنىمۇ خېلى تېجەپ قالغان بۇ دەل بىر چالمىدا ئىككى پاختەك سوققانلىق بولغان غەنىجان ئۇزۇن تىۋىتلىق توشقاننىڭ تاۋار قىممىتىگە كۆزى يەتكەندىن كېيىن باقمىچىلىقىنى كېڭەيتمەكچى بولغان بولسىمۇ ئىقتىسادى يار بەرمىگەن ئۇ ھازىر دېھقانلار ئالاھىدە باقمىچىلىق ھەمكارلىق كوپراتىۋى قۇرۇشنى خوشنا خولۇملىرىنى توشقان باقمىچىلىقىغا دەۋەت قىلىپ ئۇزۇن تىۋىتلىق توشقان باقمىچىلىقىغا تايىنىپ بېيىش يولىغا قەدەم قويۇشنى ئۈمىد قىلىدىغانلىقىنى بىلدۈردى سۇنتاغ يېزىسىنىڭ باشلىقى مۇختەر ھاجى ھۆكۈمەت تەرەپتىن غەنىجاننىڭ ئۇزۇن تىۋىتلىق توشقان باقمىچىلىقىنى داۋاملىق قوللايدىغانلىقىنى بىلدۈردى ئىشىنىمىزكى غەنىجان ئابدۇسەمىگە ئوخشاش سەزگۈر بىلىملىك دېھقانلار تەرەققىيات يولىدا ئۆزگىچە بېيىش يولىنى ئاچالايدۇ ھەمدە تېخىمۇ كۆپ كىشىلەرنىڭ نامراتلىقتىن قۇتۇلۇشىغا باشلامچى بولالايدۇ
