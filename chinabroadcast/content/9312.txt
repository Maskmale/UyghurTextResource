 شىنجاڭ خەلق رادىئو ئىستانسىسىنىڭ مەسئۇللىرى نەچچە مىڭ يۈەن ئىئانە پۇلىنى ئېلىپ ليۇ يۆۋېننى يوقلىدى قانچە كۈندە بىر ئالماشتۇرىسىلەر ئوخشاش كەسىپتىكىلەر قان سېلىش ئۆمۈر ئۇزارتىدۇ يەنە ئىچەمدۇق تېتىپ بېقىش تىلەك تىلەش ۋارقىراش بالام ئاپاڭ قېنى ئېلان بېرىش كەسىپ ئىستىلى قىزىق لىنيىسى ئالاھىدە پروگراممىسى لېي فىڭدىن ئۆگىنىش ئۆلگىلىرى ياش ناخشىچىلار شادلىق مىنۇتلىرى پرگوراممىسىدا ئاپتونوم رايونلۇق قېتىملىىق پارتىيە قۇرۇلتىيى ئاساسىي قاتلامغا بارايلى گۈزەل شىنجاڭ يېپەك يولىغا سەپەر مەملىكەتلىك رادىئو ئىستانسىسىلىرىنىڭ چوڭ تىپتىكى بىرلەشمە زىيارەت قىلىش پائالىيىتى تۈنجى نۆۋەتلىك جۇڭگو ئاسىيا ياۋروپا يەرمەنكىسى جۇڭگو شىنجاڭ خەلقئارا مىللەتلەر ئۇسسۇل بايرىمى ئەخلاق نەمۇنىچىلىرى نامزاتلىرى ئاددىي كىشىلەرنىڭ يېڭى ھىكايىلىرى سۈمۈرۈش تىپىدىكى يېمەكلىك ماشىنىسى كەشىپ قىلىندى ئىلىمپەن ساھەسىدىكى ئەڭ يېڭى بايقاشلار كەلگۈسىدە ئېلېكتىروموبىللارغا يېشىل ئېنېرگىيە ئىشلىتىلىدۇ كەلگۈسىدە نېفىت نەدىن ئېلىنىدۇ ھەرخىل لىفتلار دۇنيانىڭ قىياپىتىنى تۈپتىن ئۆزگەرتكەن سەككىز پاتېنت شىنجاڭ خەلق رادىيو ئىستانسىسى خەۋەرلەر تورى ئىستانسىمىز خەۋىرى تۈنۈگۈن چۈشتىن بۇرۇن شىنجاڭ خەلق رادىئو ئىستانسىسىنىڭ مەسئۇللىرى نەچچە مىڭ يۈەن ئىئانە پۇلىنى ئېلىپ دوختۇرخانىدا داۋالىنىۋاتقان سىچۈەنلىك دېھقان ئىشلەمچى ليۇ يۆۋېننى يوقلىدى تۈنۈگۈن چۈشتىن بۇرۇن ئۈرۈمچى ۋاقتى سائەت دا شىنجاڭ خەلق رادىئو ئىستانسىسى پارتكومىنىڭ شۇجىسى شى لىنجې ليۇ يۆۋېننىڭ كېسەلخانىسىغا كىرىپ ليۇ يۆۋېننىڭ كېسەللىك ئەھۋالىنى سۈرۈشتۈردى ھەمدە ئۇنىڭغا نەچچە مىڭ يۈەن ئىئانە پۇلىنى بەردى شى لىنجې ليۇ يۆۋېنگە مۇنداق دېدى كۆپ جاپا تارتتىڭىز بۈگۈن سىزگە شىنجاڭ خەلق رادىئو ئىستانسىسىدىكى پارتىيە ئەزالىرى ئىئانە قىلغان مىڭ يۈەن خەۋەرلەر رادىئوسىدىكى مۇخبىر تەھرىرلەرنىڭ رادىئو ئىستانسىسى ئالدىدا جەمئىيەتتىن توپلىغان مىڭ يۈەن پۇلىنى ئېلىپ كەلدىم پۇل ئىئانە قىلىدىغانلار ناھايىتى كۆپ بولۇپ تاكسى شوپۇرى ئورگان خادىملىرى ھەربىي ۋە قوراللىق ساقچى قىسىم خادىملىرى يەككە تىجارەتچىلەر باشلانغۇچ مەكتەپ ئوقۇغۇچىلىرى سىچۈەنلىك يۇرتداشلار ۋە خەلق ئىشچىلىرى مىڭ يۈەندىن كۆپرەك ئىئانە قىلدى بۇنىڭدىن باشقا بۈگۈن ئەتىگەن سالاھىيىتىنى ئاشكارىلاشنى خالىمىغان بىر كارخانىچى مىڭ يۈەن ئىئانە قىلدى دېمەك مىڭ يۈەن پۇل يىغىلىپتۇ سىزنىڭ بۇ پۇل بىلەن داۋالىنىپ سالامەتلىكىڭىزنىڭ بالدۇراق ئەسلىگە كېلىشىنى ئۈمىد قىلىمەن بۇنداق كۆپ ئاقكۆڭۈل كىشىلەرنىڭ ياردىمى بىلەن سىزنىڭ داۋالىنىش مەسىلىڭىزنىڭ ھەل قىلىنىدىغانلىقىغا ئىشىنىمەن شىنجاڭ خەلق رادىئو ئىستانسىسىنىڭ رەھبەرلىرى ليۇ يۆۋېننى يوقلاۋاتقاندا شىنجاڭ خەۋەرلەر رادىئوسىنىڭ مېھرى شەپقەت يەتكۈزۈش پۇل ئىئانە قىلىش پائالىيىتى داۋاملىق قانات يايدۇرۇلىۋاتقان ئىدى تۈنۈگۈن ئەتىگەن ئۈرۈمچى ۋاقتى سائەت دىن چۈشتىن كېيىن سائەت كىچە بولغان سائەتتە شىنجاڭ خەلق رادىئو ئىستانسىسىنىڭ ئىشىكى ئالدىدا تەسىس قىلىنغان پۇل ئىئانە قىلىش نوقتىسىدا يەنە مىڭ يۈەن ئىئانە پۇلى يىغىلدى بۇنىڭ ئىچىدە ئۈرۈمچى بۇلاقتاغ رايونىدىكى پېنسىيىگە چىققان يۈەن جىچياڭ مىڭ يۈەن ئىئانە قىلدى ھەر مىللەت شەھەر ئاھالىلىرى ۋە رادىئو ئاڭلىغۇچى دوستلارنىڭ مېھرى شەپقەت يەتكۈزۈش ئەھۋالىغا قارىتا شى لىنجې مۇنداق دېدى بۇنداق پۇل ئىئانە قىلىش پائالىيىتى تېخىمۇ كۆپ كىشىلەرگە تەسىر كۆرسىتىپ تېخىمۇ كۆپ ئىئانە پۇلى يىغىلىدۇ بۇنداق ئاممىۋى
