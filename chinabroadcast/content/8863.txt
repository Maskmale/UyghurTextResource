ئەسلىمە ئۈنۋېرسال رادىيو سەنئەت رادىيو كەسىپ ئىستىلى تاللاپ ئاڭلاش ئۇيغۇرچە ئۈگۈنۈش خەنزۇچە ئۈگۈنۈش ئىستانسىمىز ھەققىدە ئاپتونۇم رايوننىڭ لىكسىيە ئۆمىكى پارتىيە نۆۋەتلىك مەركىزىي كومىتېتى ئومۇمىي يىغىنى ۋە ئاپتونۇم رايونلۇق پارتكوم نۆۋەتلىك كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ھەققىدە دوكلات بەردى ئاۋامنىڭ كەڭ زىمىندا ئىگىلىك تىكلىشى بىپايان جەنۇبىي شىنجاڭ زېمىنىدا كىشىلىك ھاياتنىڭ گۈزەل كارتىنىسىنى يارىتايلى ئاپتونوم رايونلۇق ئىنتىزام تەكشۈرۈش كومىتېتى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە جاڭ چۈنشيەن شۇجىنىڭ مۇھىم سۆزىنىڭ روھىنى يەتكۈزدى ئاپتونوم رايونلوق خەلق قۇرۇلتىيى دائىمىي كومىتېتى پارتگۇرۇپپىسى مەركىزى ئۆگۈنۈش گۇرۇپپىسى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە ئاپتونوم رايونلوق پارتكوم خەلق قۇرۇلتىيى خىزمىتى يىغىنىنىڭ روھىنى ئۆگەندى مىسىر زوڭتوڭ سايلىمىدا مۇسۇلمان بۇرادەرلەر جەمئىيىتىگە تەۋە ھەققانىيەت پارتىيېسىنىڭ رەئىسى مورشى بىلەن سابىق زوڭلى سەفىقنىڭ بېلەتكە ئېرىشىش نىسبىتى ئالدىنقى ئىككى ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر يىلى تەخمىنەن مىڭدىن مىڭغىچە ئەيدىز بىمارى ئۆلۈپ كېتىدىكەن كورىيە ئارمىيە دائىرىلىرى چاۋشيەن يادرو سىنىقى ئېلىپ بېرىشقا تەييارلىنىپ بولدى دېدى قىزىق نوقتا خەۋەرلىرى كۆپ مەنبەلىك مۇلازىمەتنىڭ تەرەققىياتىغا نەزەر مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى ئەتىيازلىق تېرىلغۇنىڭ يېتىپ كېلىشىگە ئەگىشىپ دېھقانلارنىڭ يەنە يېڭى بىر يىللىق ئالدىراشلىقى باشلاندى زامانىۋى يېزا ئىگىلىكىنىڭ تەرەققىي قىلىشىغا ئەگىشىپ تۈرلۈك كۆپ مەنبەلىك مۇلازىمەت ئېتىز ئېرىق باشلىرىغىچە يېتىپ باردىمۇ يوق دېھقانلارغا تېگىشلىك ياردەم بېرەلىدىمۇيوق مۇخبىر يۇقىرىقى مەسىلىلەرنى كۆزدە تۇتۇپ يېقىندا باي ناھىيەسىنىڭ يېزىلىرىغا بېرىپ بۇ ھەقتە مەخسۇس زىيارەتتە بولدى ئاينىڭ كۈنى باي ناھىيە ياتۇر يېزا ئوتبېشى كەنتىدىكى دېھقان تۇرسۇن مەمەتنىڭ كۈزگى بۇغدايلىقىغا ناھىيەلىك دېھقانچىلىق ئىدارىسى بىلەن يېزىلىق يېزا ئىگىلىك تېخنىكىسى پونكىتىنىڭ بىر نەچچە تېخنىكى كەلگەن ئىدى ئۇلارنىڭ مەقسىدى تۇرسۇن مەمەتنىڭ كۈزگى بۇغدىيىنى تېزرەك كۆكلىتىشتىن ئىبارەت ناھىيەلىك دېھقانچىلىق ئىدارىسىنىڭ تېخنىكى گاۋ لۇ بۇغدايلىقتا ئولتۇرۇپ بۇغداينىڭ قۇرۇپ كەتكەن غول يۇپۇرمىقىنى يۇلۇۋەتتى بۇنىڭ بىلەن يۇمران بۇغداي مايسىسى كۆرۈندى گاۋ لۇ تۇرسۇن مەمەتكە مۇنداق دېدى قىشتا لاتقا تۆكۈپ بۇغداينى توپىداش تەدبىرلىرىنى قوللىنىپ بۇغداي ئېتىزىنىڭ يېرىلىپ كەتكەن يەرلىرىنى تىندۇرۇپ سوغۇق ھاۋا ئېقىمىنىڭ بۇغداينىڭ يىلتىز قىسمىغا ۋە تۈپچە قىسمىغا بىۋاستە كۆرسىتىدىغان تەسىرىنى ئازايتىمىز ھازىر بۇغداي مايسىسىنىڭ كۆكلىشى سەل ئاجىز قىش كىرىشتىن ئىلگىرى ئومۇمىي تۈپ سانى شالاڭ بولغان يەرگە ئوغۇت بېرىپ كۆكلىتىپ بۇغداي مايسىسىنىڭ بالدۇر ئۈنۈپ چىقىشىنى ئىلگىرى سۈرۈپ تۈپچىنىڭ باشاق نىسبىتىنى ئۆستۈرىمىز يەنە بىر نەچچە كۈندىن كېيىن بۇغداي مايسىسىنىڭ باش سۈيىنى قۇيىمىز بۇنىڭ بىلەن بۇغداي مايسىسىنىڭ كۆكلەش سۈرئىتى تېزلىشىدۇ ئابدۇرېشىتنىڭ يېرى تۇرسۇن مەمەتنىڭ يېرىگە يانداش بولۇپ ئۇ يەرلىرىنى ئاغدۇرۇپ مەھەللىۋى ئوغۇت تۆكۈپ بۇ مو يەرگە قوناق تېرىشقا تەييارلىق كۆرۈپ قويغان ئىدى گاۋ لۇ مۇنداق دېدى كۈزدە يەرلەرنى ئاغدۇرۇپ قىشتا كۆپرەك ئاق تاشلىۋەتسەك زىيانداش ھاشارەت ۋە كېسەللىكنىڭ ئالدىنى ئالغىلى بولىدۇ ھازىر ئەتىيازلىق ئوسا قىلىدىغان چاغ بولۇپ قالدى ئابدۇرېشىت مۇنداق دېدى بۇلتۇر كۈزدە قوناقنى يىغىپ بولغاندىن كېيىن يېزا دېھقانلارنى يىغىپ كەنت كومىتېتىغا كېلىپ يېزا ئىگىلىك تېخنىكىسىنى ئۆگەتتى مەنمۇ تۇپراقنى ئۆلچەپ نىسبەت بويىچە ئوغۇتلاش بىلىملىرىنى ئۆگەندىم ئىلگىرى تۇپراقنى ئۆلچەپ نىسبەت بويىچە ئوغۇتلاشنى بىلمەيتتىم پۇل خەجلىگەن بولساممۇ زىرائەتنىڭ مەھسۇلات مىقدارىنى ئاشۇرالمىغان ئىدىم خاتالىقنىڭ نەدە ئىكەنلىكىنى بىلمىگەن ئىدىم ھازىر مەن تەشەببۇسكارلىق بىلەن تۇپراق ئەۋرىشكىسىنى ئېلىپ تۇپراق ئوغۇت پونكىتىغا بېرىپ تەجرىبە قىلىپ ئارىلاپ تەكشۈرۈش نەتىجىسىگە ئاساسەن ئوغۇت بېرىش ئۆلچىمىنى بېكىتتىم يېزىلىق يېزا ئىگىلىك تېخنىكا كېڭەيتىش پونكىتىنىڭ باشلىقى ئوسمان تاش مۇنداق دېدى پەن تېخنىكىنى ئۆگەنگەندە يەرنى تېخىمۇ ياخشى تېرىغىلى بولىدۇ كىرىم تېخىمۇ ئاشىدۇ ئۈنۈمى ئەڭ زور چەكتە نامايەن بولىدۇ ئابدۇرېشىت مۇنداق دېدى يېقىنقى بىر نەچچە يىلدىن بۇيان يېزىدىكى تېخنىكلارنىڭ يېتەكچىلىكىدە كەنت بۇغدايدىن يۇقىرى ھوسۇل ئېلىپ مەھسۇلات مىقدارى يىلدىن يىلغا ئاشتى بولۇپمۇ بۇلتۇر دېھقانلار بۇغداينى يىغىپ بولغاندىن كېيىن ئېڭىزغا يەنە قوغۇن تاۋۇز تېرىدى بۇ يىل دېھقانلارنىڭ بۇغداي تېرىش ئاكتىپلىقى ناھايىتى يۇقىرى بولدى پىرسەنت دېھقاننىڭ ھەممىسى خىمىيىۋى ئوغۇت دېھقانچىلىق دورىسى تەييارلاپ قويدى يېڭى تېخنىكىنىڭ يېتەكچىلىكىدە دېھقانلار بۇ يىل تېخىمۇ كۆپ مول ھوسۇل ئېلىشنى ئۈمىد قىلىشتى تېخنىك ئوسمان تاش مۇنداق دېدى ئەتىيازلىق ئوسا باشلاندى ھازىر دېھقانلارنىڭ ھەممىسى يېزا ئىگىلىك ئىشلەپچىقىرىش ۋاستىلىرىنى تەييارلاپ قويدى بىز ھازىر گۈزگى بۇغداينىڭ قىشتىن ئۈتۈش ئەھۋالىنى تەكشۈرۈۋاتىمىز ناھىيەلىك دېھقانچىلىق ئىدارىسىنىڭ مۇئاۋىن باشلىقى تەن شياۋخوڭ مۇنداق دېدى ئەتىيازلىق تېرىلغۇ ئىشلەپچىقىرىشىغا تەييارلىق كۆرۈش ئۈچۈن ناھىيە يېزا ئىگىلىك مۇلازىمەت قىزىق لىنىيىسى رادىيوتېلېۋىزىيە يېزا ئىگىلىك ئۇچۇرى تورى قاتارلىق تەشۋىقات ۋاستىلىرى ئارقىلىق يېڭى سورت يېڭى تېخنىكا بازار ۋە سىياسەت قاتارلىق ئۇچۇرلارنى ئۆز ۋاقتىدا دېھقانلارغا يەتكۈزۈپ دېھقانلارنى تۈرلۈك دېھقانچىلىق زىرائەتلىرىنى ئىلمىي ئۇسۇلدا مۇۋاپىق تېرىشقا يېتەكلىدۇق ئەتىيازلىق تېرىلغۇ ئىشلەپچىقىرىشى رەسمىي يېتىپ كېلىش ئالدىدا يېزا ئىگىلىك ئىشلەپچىقىرىش ۋاستىلىرى سېتىش بىلەن شۇغۇللىنىدىغان سودىگەرلەرمۇ ئالدىراش بولۇپ كەتتى ئۇلار يېتەرلىك بولغان خىمىيىۋى ئوغۇت ئۇرۇقلۇقنى زاپاس تەييارلاپ دېھقانلارنىڭ كېلىپ سېتىۋېلىشىنى كۈتۈپ تۇردى ئاينىڭ كۈنى باي ناھىيەسىنىڭ كانچى يېزىسىدىكى بىر نەچچە يېزا ئىگىلىك ئىشلەپچىقىرىش ۋاستىلىرىنى بىۋاستە سېتىش دۇكانلىرىنىڭ سودىسى تازا قىزىپ كەتتى شىنجاڭ باي دېھقان يېزا ئىگىلىك ئىشلەپچىقىرىش ۋاستىلىرىنى بىۋاستە سېتىش شىركىتى دېگەن ۋىۋىسكا ئېسىلغان بىر شىركەتنىڭ ئىشىكى ئالدىدا بىر كىچىك تىپلىق تىراكتور توختىتىپ قويۇلغان ئىدى بىر نەچچە دېھقان تىراكتورغا خىمىيىۋى ئوغۇت قاچىلاۋاتقان ئىدى يېزا ئىگىلىك ئىشلەپچىقىرىش ۋاسىتىلىرىنى بىۋاستە سېتىش دۇكىنىنىڭ خۇجايىنلىرى ئۆزئارا خىمىيىۋى ئوغۇتنىڭ باھاسى ۋە بۇ يىللىق تېرىقچىلىق ئۇچۇرى ھەققىدە پاراڭلىشىۋاتقان ئىدى ناھىيەلىك يېزا ئىگىلىك ئۇنىۋېرسال ئەترىتىنىڭ باشلىقى تۇرغۇن تۇنىياز ۋە ئۇنىڭ ھەمراھلىرى بىر يېزا ئىگىلىك ئىشلەپچىقىرىش ۋاستىلىرىنى بىۋاستە سېتىش دۇكىنىغا كىرىپ كەلدى دۇكان ئىگىسى ئەكبەر تۇردى ئالدىراش دۇكاندىن چىقىپ ئۇلارنى كۈتۈۋالدى تۇرغۇن تۇنىياز ئەكبەر تۇردىدىن تىجارەت ئەھۋالىنى سورىدى دائىم كېلىپ تەكشۈرۈش ئارقىلىق ئۇلار بىربىرى بىلەن كونا تونۇشلاردىن بولۇپ قالغان ئىدى تەكشۈرگۈچى خادىملار خىمىيىۋى ئوغۇت دېھقانچىلىق دورىسىنىڭ تاۋار ماركىسىنى تەكشۈرۈپ كۆرگەچ تىجارەتچىلەردىن سودا ئەھۋالىنى سورىدى تۇرغۇن تۇنىياز مۇنداق دېدى بۇ بىرىكمە ئوغۇتلارنىڭ ھەممىسى زاۋۇتتىن چىققاندا ئەنگە ئېلىپ تىزىملىتىلغان ئەتىيازلىق تېرىلغۇ باشلىنىشتىن ئىلگىرى ناھىيە سوداسانائەت سۈپەت نازارەتچىلىكى قاتارلىق تارماقلارنى تەشكىللەپ بىر ئاي داۋام قىلغانيېزا ئىگىلىك ئىشلەپچىقىرىش ۋاسىتىلىرىنىڭ ساختا ناچارلىرىغا زەربە بېرىپ ئەتىيازلىق تېرىلغۇنى قوغداش مەخسۇس ھەرىكىتىنى قانات يايدۇرۇپ يېزا ئىگىلىك ئىشلەپچىقىرىش ۋاستىلىرى بازارلىرىنى قېلىپلاشتۇردى نۆۋەتتە ناھىيە بويىچە ئادەم قېتىمدىن كۆپرەك يېزا ئىگىلىكىدە قانۇننى ئىجرا قىلىش خادىمىنى ئاجرىتىپ تىن كۆپرەك يېزا ئىگىلىك ئىشلەپچىقىرىش ۋاستىلىرىنى سېتىش تور نۇقتىسىنى تەكشۈردۇق نۆۋەتتە ساختا ناچار يېزا ئىگىلىك ئىشلەپچىقىرىش ۋاسىتىلىرىنى ياساش سېتىش ئەھۋاللىرى بايقالمىدى بۇ يىل باي ناھىيەسى ئەتىياز پەسىللىك يېزا ئىگىلىك ئىشلەپچىقىرىش ۋاستىلىرى بازارلىرىنى تەرتىپكە سېلىشنى قانات يايدۇرۇپ ساختا ناچار زەھەرلىك زىيانلىق ئۇرۇقلۇق خىمىيىۋى ئوغۇت دېھقانچىلىق دورىسى سۇلياۋ يۇپۇق ۋە تېمىتىپ سۇغىرىش تۇرۇببىسى قاتارلىق يېزا ئىگىلىك ئىشلەپچىقىرىش ۋاستىلىرى مەھسۇلاتلىرىنى ئىشلەپچىقىرىش سېتىش ۋە ئىشلىتىشكە قاتتىق زەرىبە بەردى دېھقانلارنى قوغداپ ئەتىيازلىق تېرىلغۇ ئىشلەپچىقىرىشىغا كاپالەتلىك قىلىشنى ئاساسىي تېما قىلىپ ئىشنى مەنبەدىن تىزگىنلەشنى كۈچەيتىپ بازارنى نازارەت قىلىپ باشقۇرۇشنى چىڭ تۇتۇپ مۇلازىمەت يېتەكچىلىك قىلىشنى كۈچەيتىش خادىملارنىڭ ساپاسىنى ئۆستۈرۈپ تۈزۈم قۇرۇلۇشىنى قېلىپلاشتۇردى زەربىدارلىق بىلەن تەكشۈرۈش بىلەن تولۇق مۇساپىلىق نازارەت قىلىپ باشقۇرۇشنى بىرلەشتۈرۈپ ساختا ناچار يېزا ئىگىلىك ئىشلەپچىقىرىش ۋاستىلىرىغا قاتتىق زەرىبە بېرىش بىلەن شەكسىز يېزا ئىگىلىك ئىشلەپچىقىرىش ۋاستىلىرى بىلەن تەمىنلەشنى بىرلەشتۈرۈپ ناھىيە بويىچە يېزا ئىگىلىك ئىشلەپچىقىرىش ۋاسىتىلىرى بازارلىرىنىڭ تەرتىپىنى ياخشىلىدى كۆرۈپ قويۇڭ ئۇيقۇنى ياخشىلايدىغان يېمەكلىكلەر گۈزەللىك قەيسەرلىكتە پىتنەپاساتقا ئىشەنمەڭ ئۆزىڭىزنىڭ كەيپىياتىغا دىققەت قىلىپ باقتىڭىزمۇ يوق كەمبەغەلنىڭ ئىلتىپاتى دادا ئانامغا ياخشى قارا سول قولنى ئىشلىتىش مېڭىگە قان چۈشۈشنىڭ ئالدىنى ئالىدۇ ئاچچىقسۇ ھاردۇقنى چىقىرىدۇ كالا سۈتىگە قەنتنى كۆپ قوشۇشقا بولمايدۇ ئەپسۇسلانماي دېسىڭىز بەخت تېمى دېھقانلارنى جەلپ قىلدى خېجىڭ ناھىيىسىدە دۆلەتنىڭ تەربىيلەش پىلانى بويىچە ئوتتۇراباشلانغۇچ مەكتەپلەرنىڭ ئوقۇتۇشنى يېزىلارغا يەتكۈزۈش پائالىيتى ئېلىپ بېرىلدى كۇچا ناھىيەسىدە تۇنجى قېتىملىق كەشتىچى ئاياللارنىڭ ماھارەت كۆرسىتىش مۇسابىقىسى ئۆتكۈزۈلدى فۇكاڭ شەھىرىدە قىشلىق تەنھەرىكەت مۇسابىقىسى كەيپىياتى قۇيۇق بولدى نادىر ناخشا مۇزىكىلار دادىڭىز چەت ئەل تىلىنى بىلەمدۇ ئىش ھەققى تەبرىكلەش پائالىيىتى كۆزۈم كۆرمىسۇن تور بېتىمىزدىكى يېڭىلىقلار ئىستانسىمىز ئاپتونوم رايونلۇق خەلق ھۆكۈمىتى كەسپتىكى ناتوغرا ئىستېللارنى تۈزىتىش ئىشخانىسى ئاپتونوم رايونلۇق قاتناش ـ ترانسپورت نازارىتى ئاپتونوم رايونلۇق جامائەت خەۋپسىزلىكى نازارىتى بىلەن بىرلىشىپ ئۆتكۈزىدىغان كەسپ ئىستېلى قىزىق لېنىيىسى پروگراممىسىنىڭ قاتناش مىڭ چاقىرىملىق سەپىرى چوڭ تىپتىكى بىۋاستە ئاڭلىتىش پائالىيىتى ئاينىڭ كۈنى يەنى ئەتە خوتەن ۋىلايىتىدە ئۆتكۈزىلىدۇ ئىستانسىمىز ئاپتونوم رايونلۇق خەلق ھۆكۈمىتى كەسپتىكى ناتوغرا ئىستېللارنى تۈزىتىش ئىشخانىسى ئاپتونوم رايونلۇق قاتناش ـ ترانسپورت نازارىتى
