ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر يىلى تەخمىنەن مىڭدىن مىڭغىچە ئەيدىز بىمارى ئۆلۈپ كېتىدىكەن كورىيە ئارمىيە دائىرىلىرى چاۋشيەن يادرو سىنىقى ئېلىپ بېرىشقا تەييارلىنىپ بولدى دېدى قىزىق نوقتا خەۋەرلىرى تارىختىكى بۈگۈن مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى يىلى بۈگۈنكى كۈندە جۇڭگو كومپارتىيەسىنىڭ تۇنجى كۈندىلىك گېزىتى ئىسسىق قان گېزىتى نەشىر قىلىنغان يىلى بۈگۈنكى كۈندە مەملىكەت ئىچى ۋە سىرتىنى زىلزىلىگە سالغان خۇاڭگۇتۇن ۋەقەسى يۈز ئارمىيەسى دونكىرىكتىن پۈتۈنلەي چىكىنىپ بولغان يىلى ئايدا گىتلېر ئەنگىلىيە بىلەن فرانسىيەنىڭ ماداراچىلىق سىياسىتىنىڭ مەھسۇلى بولغان مىيۇنخېن كىلىشىمىنى ئىجرا قىلىشىدىن پايدىلىنىپ فاشىست گېرمانىيەسى ئارمىيەسىگە قوماندانلىق قىلىپ غەربىي سەپتە چاقماق تىزلىكىدە تۇيۇقسىز ھۇجۇم قوزغاپ نورۋىگىيە دانىيەگوللاندىيەبېلگىيە قاتارلىق دۆلەتلەرنى ئىشغال قىلغان شۇنىڭ بىلەن بىر ۋاقىتتا فرانسىيەنىڭ ماگىنوت مۇداپىئە سىپىدىن ئايلىنىپ ئۆتۈپ بىۋاستە فرانسىيەنىڭ شىمالىغا كىرىپ ئەنگىلىيەفرانسىيە بىرلەشمە ئارمىيەسىنى ئىككىگە بۆلۈۋەتكەن ئەنگىلىيە قوشۇنى بىلەن بىر قىسىم فرانسىيە ئارمىيەسى دونكىرىكتا مۇھاسىرەگە قالغان ئەنگىلىيە نۇرغۇن پاراخوت ۋە كىمىنى ئىشقا سېلىپ مىڭ ئەسكەرنى ئەنگىلىيەگە چىكىندۈرۈپ كەتكەن يىلى بۈگۈنكى كۈندە دۇنيا ئۇرۇشى مەزگىلىدىكى مەشھۇر ئۇرۇش مىدۋاي ئارىلى ئۇرۇشى پارتلىغان ئامېرىكا تېنچ ياپونىيە بىرلەشمە فلوتىنىڭ ئاساسىي كۈچى بولغان ئاۋىياتمىكىنى چۆكتۈرۈۋەتكەن ئايرۇپىلانىنى ئېتىپ چۈشۈرگەن ياپونىيە ئارمىيەسىنىڭ دىن ئارتۇق ئوفېتسىر ئەسكىرى نەچچە يۈز ئۇچقۇچىسى ھالاك بولغان شۇنىڭدىن باشلاپ ياپونىيە تېنچ ئوكيان ئۇرۇشىدىكى تەشەببۇسكارلىق ھوقۇقىدىن مەھرۇم قالغانئۇرۇش ۋەزىيىتى ئىتتىپاقداش ئارمىيەگە پايدىلىق تەرەپكە بۇرالغان يىلى بۈگۈنكى كۈندە ئىتتىپاقداش ئارمىيە ئىتالىيەنىڭ پايتەختى پايتەختى شۇنداقلا ناتسىستلارنىڭ قولىدىن ئازاد قىلىنغان تۇنجى پايتەخت ئىدى يىلى بۈگۈنكى كۈندە ئېلىمىز دۇنيا بىلىم مۈلۈك ھوقۇقىغا تەشكىلاتىغا كىرگەن دۇنيا بىلىم مۈلۈك ھوقۇقى تەشكىلاتى قۇرۇش ئەھدىنامىسى
