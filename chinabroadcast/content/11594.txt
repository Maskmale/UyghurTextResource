ئەسلىمە ئۈنۋېرسال رادىيو سەنئەت رادىيو كەسىپ ئىستىلى تاللاپ ئاڭلاش ئۇيغۇرچە ئۈگۈنۈش خەنزۇچە ئۈگۈنۈش ئىستانسىمىز ھەققىدە ئاپتونۇم رايوننىڭ لىكسىيە ئۆمىكى پارتىيە نۆۋەتلىك مەركىزىي كومىتېتى ئومۇمىي يىغىنى ۋە ئاپتونۇم رايونلۇق پارتكوم نۆۋەتلىك كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ھەققىدە دوكلات بەردى ئاۋامنىڭ كەڭ زىمىندا ئىگىلىك تىكلىشى بىپايان جەنۇبىي شىنجاڭ زېمىنىدا كىشىلىك ھاياتنىڭ گۈزەل كارتىنىسىنى يارىتايلى ئاپتونوم رايونلۇق ئىنتىزام تەكشۈرۈش كومىتېتى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە جاڭ چۈنشيەن شۇجىنىڭ مۇھىم سۆزىنىڭ روھىنى يەتكۈزدى ئاپتونوم رايونلوق خەلق قۇرۇلتىيى دائىمىي كومىتېتى پارتگۇرۇپپىسى مەركىزى ئۆگۈنۈش گۇرۇپپىسى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە ئاپتونوم رايونلوق پارتكوم خەلق قۇرۇلتىيى خىزمىتى يىغىنىنىڭ روھىنى ئۆگەندى مىسىر زوڭتوڭ سايلىمىدا مۇسۇلمان بۇرادەرلەر جەمئىيىتىگە تەۋە ھەققانىيەت پارتىيېسىنىڭ رەئىسى مورشى بىلەن سابىق زوڭلى سەفىقنىڭ بېلەتكە ئېرىشىش نىسبىتى ئالدىنقى ئىككى ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر يىلى تەخمىنەن مىڭدىن مىڭغىچە ئەيدىز بىمارى ئۆلۈپ كېتىدىكەن كورىيە ئارمىيە دائىرىلىرى چاۋشيەن يادرو سىنىقى ئېلىپ بېرىشقا تەييارلىنىپ بولدى دېدى قىزىق نوقتا خەۋەرلىرى ئانامغا مەكتۈپ ناملىق كىتاپنى تارقتىش مۇراسىمى ئۆتكۈزۈلدى مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى كاشغەر مەدەنىيىتى ژورنىلىنىڭ باش مۇھەررىرى تونۇلغان ئاخباراتچى قەشقەر شەھەرلىك پارتىكوم تەشۋىقات بۆلىمنىڭ مۇئاۋىن باشلىقى ئابلىمىت سوپىھاجى تونۇلغان ئاخباراتچى جامائەت ئەربابى مەرھۇم ئېزىزجان توختىھاجىنىڭ دادىسى توختىھاجى تۇرسۇن ئاكا سۆز قىلماقتا يىغىن مەيدانىدىن بىر كۆرنۈش يىغىن مەيدانىدىن بىر كۆرنۈش يىغىن مەيدانىدىن بىر كۆرنۈش تونۇلغان ئوبزۇرچى قەشقەر پىداگوگىكا ئىنىستىتوتىنىڭ پىراففىسورى كىرىمجان ئابدۇرېھىم سۆز قىلماقتا يىغىن مەيدانىدىن بىر كۆرنۈش يىغىن مەيدانىدىن بىر كۆرنۈش يىغىن مەيدانىدىن بىر كۆرنۈش يىغىن مەيدانىدىن بىر كۆرنۈش نەسىرچى بالىلار ئويۇنلىرى تەتقىقاتچىسى مۇھەممەدجان ئابلىز بۆرەيار سۆز قىلماقتا ئەدەبىي تەرجىمان يازغۇچى مەرھۇم ئەزىزجاننىڭ يېقىن دىلكىشى ئەسقەر ئەھمىدى سۆز قىلماقتا پىشقەدەم شائىر جامائەت ئەربابى ئابلىز ئۆمەر ئەنجانى سۆز قىلماقتا تۈمەن سودا چەكلىك شىركىتىنىڭ دىرىكتورى يامبۇ سودا سارىيىنىڭ غوجايىنى ئابدۇشۈكۈر ھاجى سۆز قىلماقتا ئىختىيارىي مۇخبىرىمىز تاھىرجان ئاتاۋۇللا قوشماق خەۋىرى فۇتۇسى يىلى ئاينىڭ كۈنى چۈشتىن بۇرۇن ئەزىزانە قەشقەردە كاتتا بىر ئەنجۈمەن ئۆتكۈزۈلدى يەنى تونۇلغان ئاخباراتچى نەسىرچى ئەدەبىي ئوبزۇرچى مەرھۇم ئېزىزجان توختىھاجىنىڭ يېقىندا قەشقەر ئۇيغۇر نەشىرىياتىدىن نەشىر قىلىنغان ئانامغا مەكتۈپ ناملىق كىتابىنى تارقىتىش مۇراسىمى يامبۇ سودا سارىيىنىڭ تۆتىنجى قەۋىتىدە ئۆتكۈزۈلدى بۇ پائالىيەتكە قەشقەر شەھىرى ۋە خوشنا ناھىيىلەردىن بولۇپ دىن ئارتۇق يازغۇچى شائىر ئاخباراتچى ۋە ھەر ساھە زاتلىرى قاتناشتى كاشغەر مەدەنىيىتى ژورنىلىنىڭ باش مۇھەررىرى تونۇلغان ئاخباراتچى قەشقەر شەھەرلىك پارتكوم تەشۋېقات بۆلىمنىڭ مۇئاۋېن باشلىقى ئابلىمىت سوپىھاجى بۇ پائالىيەتكە رىياسەتچىلىك قىلدى تونۇلغان ئاخباراتچى جامائەت ئەربابى مەرھۇم ئېزىزجان توختىھاجىنىڭ دادىسى توختىھاجى ئاكا كۆڭئۈل سۆزىنى بايان قىلدى ئېزىزجان توختىھاجىنىڭ چوڭ ئوغلى دادىسىغا يازغان شىئىرىنى ئوقۇپ ئۆتتى قەشقەر شەھىرىنىڭ باشلىقى ئەنۋەر تۇرسۇن ئاتاقلىق يازغۇچى ئەخەت تۇردى ئاپتۇنۇم رايۇنمىزنىڭ ھەرقايسى جايلىرىدىكى ئاتاقلىق ئەدىپلەر رەھبەرلەر ۋە جامائەت ئەربابلىرىنىڭ تەبرىك خەتلىرى ئوقۇپ ئۆتۈلدى ئابلىز ئۆمەر ئەنجانى كىرىمجان ئابدۇرېھىم مۇھەممەدجان ئابلىز بۆرەيار ئەسقەر ئەھمىدى قاتارلىقلار مەرھۇم ئېزىزجان توختىھاجىنىڭ ھايات پائالىيىتى ئەدەبىي ئىجادىيىتىنى ئەسلەپ ئۆتتى ئۇيغۇر ئەدەبىياتىدىكى ئورنى ۋە تۆھپىسىنى مۇئەييەنلەشتۈردى يامبۇ سودا سارىيىنىڭ غوجايىنى مەرىپەتپەرۋەر كارخانىچى ئابدۈشۈكۈر ھاجى موھىم سۆز قىلدى ۋە ئۆزىنىڭ يازغان شىئىرىنى ئوقۇدى يىغىن زالىنىڭ ئوتتۇرىسىغا مەرھۇم ئېزىزجان توختىھاجىنىڭ چوڭايتىلغان سۈرىتى ئېسىلغان بولۇپ بۇ پائالىيەتتە ھەر قاتناشقۇچىغا مەرھۇم ئېزىزجان توختىھاجىنىڭ يېقىندا قەشقەر ئۇيغۇر نەشىرىياتىدىن نەشىر قىلىنغان ئانامغا مەكتۈپ ناملىق كىتابىدىن بىر تالدىن ھەقسىز تاپشۇرۇپ بىرىلدى ھەممە بىردەك تونۇلغان ئاخباراتچى جامائەت ئەربابى مەرھۇم ئېزىزجان توختىھاجىنىڭ دادىسى توختىھاجى ئاكىنىڭ ئائىلسى ۋە ئېزىزجان توختىھاجىنىڭ ئائىلىسگە بەخت ئۇزۇن ئۆمۈر تىلەشتى بىر يېرىم سائەتتىن ئارتۇق داۋاملاشقان بۇ پائاليەتتىن كىيىن توختىھاجى ئاكا بارلىق ئەزىز مىھمانلار شەرىپىگە كاتتا زىياپەت بەردى مەسئۇل مۇھەررىرى ماھىنۇر غۇپۇر كۆرۈپ قويۇڭ ئۇيقۇنى ياخشىلايدىغان
