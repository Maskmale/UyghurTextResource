مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى ئەكبەرنىڭ شىنجاڭغا مۇھەببىتىنى ئىزھار قىلىشى ناملىق ماقالە چۇڭچىڭ ئۇنىۋېرسىتېت شەھەرچىسىدە چۇڭچىڭ ئاتەش ئېلېكتىرونلۇق سودا چەكلىك شىركىتى نامىدا بىر بالا شىركەت ئوتتۇرا سىنىپىغا قوبۇل قىلىنىپ بېيجىڭدا ئوقۇدى تۆت يىلدىن كېيىن ئەلا نەتىجە بىلەن چۇڭچىڭ پېداگوگىكا ئۇنىۋېرسىتېتىنىڭ كومپيۇتېر ئىلمى ۋە تېخنىكىسى كەسپىگە قوبۇل قىلىندى بۇ يىل چاغاندا كەلپىن ناھىيەسىگە قايتقان ئەكبەر يۇرتىنىڭ چىلان سېتىش يولىنىڭ تازا راۋان ئەمەسلىكىنى بىلدى ۋە تەكشۈرۈپ دەلىللەش ئارقىلىق شىركەت قۇرۇشنى قارار قىلدى ھالبۇكى ئىگىلىك تىكلەش مەبلىغىنىڭ ئاز دېگەندە مىڭ يۈەن بولىدىغانلىقى مۆلچەرلەنگەندىن كېيىن ئىقتىساد مەسىلىسى ياشلىق بۇ يىگىتنىڭ بېشىنى قاتۇردى شۇنىڭ بىلەن ئۇ ھېيت بايرام ۋە دەم ئېلىش كۈنلىرىدىن پايدىلىنىپ ئىشلەپ ئىككى ئايدا مىڭ يۈەن پۇل تاپتى ئىگىلىك تىكلەش رايونى باشلىقىغا كەلپىن ناھىيەسىنىڭ ئەھۋالىنى ئېيتىپ تەسىرلەندۈردۈم ئۇ كىشى ئىجارە ھەققىنى ئايدا بىر تۆلەشكە ماقۇل بولدى بۇنىڭ بىلەن چىقىمىم ئازايدى دېدى ئەكبەر ئۇ پۇلنى تېجەش ئۈچۈن بىر نەچچە ئېلېكتىرون شەھەرچىسىنى ئايلىنىپ يۈەن پۇل بىلەن چۇڭچىڭ ئاتەش ئېلېكتىرونلۇق سودا چەكلىك شىركىتىنى قۇردى چۇڭچىڭ ئاتەش ئېلېكتىرونلۇق سودا چەكلىك شىركىتى ئاساسلىقى قۇرۇق يەل يېمىشلەرنى سېتىش تىجارىتى قىلىدىغان بولۇپ ئاقسۇنىڭ چىلىنى نېپىز پوستىلىق ياڭىقى قۇرۇق ئۈزۈم بادام قاتارلىقلارنى سېتىشنى ئاساس قىلىدۇ بۇ مەھسۇلاتلار بازارغا سېلىنىشى بىلەنلا ئىستۇدېنت شەھەرچىسىدىكى ئوقۇغۇچىلارنىڭ ئالقىشىغا ئېرىشتى چۇڭچىڭ ئىستۇدېنت شەھەرچىسىدە مەكتەپ مىڭدىن ئارتۇق ئوقۇغۇچى بار بولۇپ ئىستېمالچىلار توپى چوڭ ئىدى لېكىن ئوقۇغۇچىلارنىڭ پۇلى ئاز بولغاچقا كۆپ قىسىم ئوقۇغۇچىلار بىر قېتىمدا گىراملا سېتىۋالاتتى شۇنداقتىمۇ ئەكبەر ۋاز كەچمىدى ئاقسۇ مەملىكەت بويىچە نامى بار مېۋە چىۋە ماكانى مېۋىلەرنىڭ سۈپىتى ياخشى شېكەر مىقدارى يۇقىرى بولغاچقا ئىستېمالچىلارنىڭ ياقتۇرۇشىغا ئېرىشتى بارا بارا ئەكبەرنىڭ تىجارىتى پۈتۈن چۇڭچىڭ شەھىرىگە كېڭەيدى ھەتتا شەھەر ئەتراپىدىكى نۇرغۇن ئىستېمالچىلار ئۇنىڭ شىركىتىنى بىلدى ئۇنىڭ ئايلىق سېتىش سوممىسى يۈەندىن مىڭ يۈەنگە ئاشتى ئۇ ئون ئىستۇدېنتنى خىزمەتكە تەكلىپ قىلدى ھەم ناھايىتى ئالدىراش ئىشلەپ كۈندە كىلومېتىر يول يۈرۈپ ئىشلارنى يۈرۈشتۈردى تېز يوللانما شىركەتلىرىمۇ ئۇنىڭغا مەخسۇس مال يىغىش ۋاقتى تەسىس قىلىپ بېرىپ بەلگىلەنگەن ۋاقىتتا ئادەم ئەۋەتىپ ئۇنىڭ ماللىرىنى يىغدى ئۇ ئۆگەنگەن بىلىملىرىنى ئىشلىتىپ لىنىيەسىدىكىلەر لىنىيە سىرتىدىكىلەرنى يېتەكلەش كارخانىلارنىڭ شەخسىي سېتىش ئەندىزىسىدىن پايدىلىنىپ قۇرۇق يەل يېمىش سېتىش كۆلىمىنى ئاقسۇ ئالمىسى كورلا نەشپۈتى قۇمۇل قوغۇنىغىچە كېڭەيتتى ئىنتېرنېت ئەڭ كەڭ سېتىش سۇپىسى بىز بۇ سۇپىدىن پايدىلىنىپ سېتىش يولىنى ئېچىپ ئاقسۇ مېۋىلىرىنىڭ سېتىش يولىنى ئېچىپ ىنىڭ تەسىرىنى زورايتتۇق ئاقسۇنىڭ قىزىل ئالمىسىنىڭ چۇڭچىڭدا داڭقى بار نۇرغۇن چۇڭچىڭلىقلار ئاقسۇغا بارغاندا بىرەر يەشىك ئالما ئېلىۋېلىشنى ياخشى كۆرىدۇ لېكىن ئارىلىق يىراق بولغاچقا نۇرغۇن قانۇنسىز سودىگەرلەر باشقا ئالمىنى ئاقسۇ ئالمىسى قىلىپ كۆرسىتىپ ئىستېمالچىلارنى ئالدايدۇدېدى
