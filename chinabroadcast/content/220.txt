 جىڭىسگۈل بېكبولات نەچچە يىللىق ئوقۇتۇش جەريانىدا قىيىنچىلىقى بار ئوقۇغۇچىلارغا ئىزچىل خالىس ياردەم بېرىپ كەلدى فاڭ خەيلى نەچچە يىللاردىن بۇيان ئۇ پارالېچ بولۇپ قالغان قېيىنئانىسىنىڭ ھالىدىن ئوبدان خەۋەر ئالدى جاڭ يۈشىن پارالېچ بولۇپ قالغان ئايالى ساۋ جېڭيىڭنىڭ ھالىدىن ئۇزۇن مەزگىل خەۋەر ئالدى كۆكدۈڭ كەنتىدە يېڭى يۈكسىلىش ما بىن گەرچە مېيىپ بولسىمۇ لېكىن ئۇ ئىگىلمەس ئىرادىسى يېڭىلمەس ئېتىقادىغا تايىنىپ ئۆزىنىڭ ھاياتىنى تېخىمۇ گۈزەللەشتۈردى جىمىسار ناھىيىسىدىكى پۇقراۋى كارخانىچى ئوسمان تاڭقاي بارلىقىنى ئاممىۋى پاراۋانلىق ئىشلىرىغا بېغىشلاپ ئەتراپىدىكى نۇرغۇن كىشىلەرگە ياردەم بىرىپ كەلدى قەشقەر خەلق دوختۇرخانىسىنىڭ دوختۇرى خې چىڭ نامرات ئاغرىقلارغا ياردەم بىرىپ ئاغرىقلارنىڭ ئىشەنچىسى ۋە ماختىشىغا سازاۋەر بولدى ۋاڭ فېڭشيا ئۇدا يىل قان تەقدىم قىلدى يىل ھەقسىز قان تەقدىم قىلىپ كەلگەن ئۈرۈمچى شەھەر ئاھالىسى ۋاك جۈنجۇن ۋېن جىخەي كەنت ئاھالىلىرىنىڭ ئېلېكتر سايمانلىرىنى ھەقسىز ياكى تۆۋەن باھادا رېمونت قىلىپ بەردى بۇ بەتنىڭ نەشىر ھوقۇقى شىنجاڭ خەلق رادىيو ئىستانسىسىغا تەۋە ئالاقىلىشىش نومۇرى ئادرىسىمىز ئۈرۈمچى شەھىرى ئىتتىپاق يولى نومۇر شىنجاڭ خەلق رادىيو ئسىتانسىسى تور مەركىزى
