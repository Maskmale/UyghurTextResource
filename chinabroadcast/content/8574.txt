باشلىقلىرى بىلەن سۆھبەتلەشتى جاڭ چۈنشيەن ئالدى بىلەن ئاپتونوم رايونلۇق پارتكومغا ۋاكالىتەن يېڭىدىن ۋەزىپىگە تەيىنلەنگەن يولداشلارنى قىزغىن تەبرىكلىدى ئۇ مۇنداق دېدى يولداشلارنىڭ ئۆز بۇرچىنى ئۇنتۇماي ئېغىر يۈكتىن باش تارتماي ئاپتونوم رايونلۇق پارتكومنىڭ رەھبەرلىكىدە بىر نىيەت بىر يۆنىلىش بىر پىكىردە ئاپتونوم رايونىمىزنىڭ زامانىۋىيلاشتۇرۇش قۇرۇلۇشىنىڭ يېڭى ئەندىزىسىنى تىرىشىپ بەرپا قىلىشىنى چىن كۆڭلۈمدىن ئۈمىد قىلىمەن جاڭ چۈنشيەن مۇنداق دېدى غايىدە ئىتىقادتا چىڭ تۇرۇپ جۇڭگوچە سوتسىيالىزىم يولىدا مېڭىپ پارتىيە قۇرۇلتىيىدا ئوتتۇرىغا قۇيۇلغان ئىككى يۈز يىل نىشانى بويىچە جۇڭخۇا مىللەتلىرىنىڭ ئۇلۇغ گۈللىنىش ئارزۇسىنى ئىشقا ئاشۇرۇش ئۈچۈن قەتئىي كۆرەش قىلىش كېرەك جاڭ چۈنشيەن مۇنداق دېدى پارتىيەنىڭ رەھبەرلىكىنى كۈچەيتىش خەلق قۇرۇلتىيى خەلق ھۆكۈمىتى سىياسىي كېڭەش سوت مەھكىمىسى ۋە تەپتىش مەھكىمىسى خىزمىتىنى ياخشى ئىشلەشنىڭ سىياسىي كاپالىتى ئىكەنلىكىدە چىڭ تۇرۇش ۋە ئۇنى كۈچەيتىش كېرەك ئاپتونوم رايونلۇق پارتكومنىڭ رەھبەرلىكىدە ئاپتونوم رايونلۇق خەلق قۇرۇلتىيى ھەر دەرىجىلىك خەلق قۇرۇلتىيى ۋە دائىمىي كومىتېتلىرى مەسئۇلىيىتىنى ياخشى ئىجرا قىلىپ خەلق قۇرۇلىتيى ۋەكىللىرىنىڭ رولىنى تۇلۇق جارى قىلدۇرۇپ ئاپتونوم رايونىمىزدىكى ھەر مىللەت خەلق ئاممىسنىڭ ئىرادىسى ئەقىل پاراسىتى ۋە كۈچىنى سوتسىيالىستىك زامانىۋيلاشتۇرۇش قۇرۇلۇشىغا مەركەزلەشتۈرۈش كېرەك ئاپتونوم رايونلۇق خەلق ھۆكۈمىتى پارتكومنىڭ تەدبىر ئورۇنلاشتۇرمىلىرىنى قەتئىي ئىزچىللاشتۇرۇپ ۋە ئەمەليلەشتۈرۈپ ئىلمىي تەرەققىيات ئىقتىدارى ئىزچىللاشتۇرۇش ۋە ئەمەلىيلەشتۈرۈش ئىقتىدارىنى ئۆزلۈكسىز يۇقىرى كۆتۈرۈپ ئاپتونوم رايونىمىزنىڭ ئىلمىي تەرەققىياتى ۋە كېيىن تەرەققىي قىلىپ ئېشىپ كېتىشىنى تىرىشىپ ئىلگىرى سۈرۈش كېرەك ئاپتونوم رايونلۇق سىياسىي كېڭەش ھەر قايسى ساھەدىكى زاتلارنى جۇڭگوچە سوتسىيالىزىم ئۇلۇغ بايرىقىنى ئېگىز كۆتۈرۈشكە يېتەكلەشنى سىياسىي كېڭەشنىڭ فونكىسىيىسىنى ئادا قىلىدىغان تۈپ ۋەزىپىسى قىلىشتا چىڭ تۇرۇپ سىياسىي كېڭەشنىڭ ئىختىساسلىقلار يىغىلغان ئەقىل پاراسەت توپلانغان كەڭ دائىرىدە ئالاقە باغلايدىغان ئەۋزەللىكىنى تۇلۇق جارى قىلدۇرۇپ سىياسىي جەھەتتە كېڭىشىش دېموكراتىك نازارەت قىلىش ھاكىمىيەت ئىشلىرىغا قاتنىشىش ۋە ھاكىمىيەت ئىشلىرى ئۈستىدە پىكىر يۈرگۈزۈش فونكىسىيىسىنى ياخشى ئادا قىلىش كېرەك ئاپتونوم رايونىمىزدىكى سوت مەھكىمىسى ۋە تەپتىش مەھكىمىسى پارتىيىنىڭ رەھبەرلىكىدە قەتئىي چىڭ تۇرۇپ ئەدلىيىدە ئادىل ھەققانىي بولۇشنى چۆرىدەپ ئادىل سوت ئېچىش ۋە تەپتىش ھوقۇقىنى يۈرگۈزۈش ئىقتىدارىنى يۇقىرى كۆتۈرۈپ قانۇنىي ئۈنۈمىنىڭ ئىجتىمائىي ئۈنۈم بىلەن بىردەك بولۇشىنى تىرىشىپ ئىشقا ئاشۇرۇش كېرەك جاڭ چۈنشيەن مۇنداق دېدى شىنجاڭدىكى ھەر مىللەت خەلقىنى قىزغىن سۆيۈپ گۈللەنگەن باي ئىناق مۇقىم بولغان گۈزەل شىنجاڭ قۇرۇپ چىقىشتىكى مۇھىم ۋەزپىنى ئۈستىگە ئېلىش كېرەك ھەر مىللەت خەلق ئاممىسىنىڭ گۈزەل تۇرمۇشقا ئىنتىلىشنى
