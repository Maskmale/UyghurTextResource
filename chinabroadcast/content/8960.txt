يىغىنى ۋە ئاپتونۇم رايونلۇق پارتكوم نۆۋەتلىك كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ھەققىدە دوكلات بەردى ئاۋامنىڭ كەڭ زىمىندا ئىگىلىك تىكلىشى بىپايان جەنۇبىي شىنجاڭ زېمىنىدا كىشىلىك ھاياتنىڭ گۈزەل رايونلوق خەلق قۇرۇلتىيى دائىمىي كومىتېتى پارتگۇرۇپپىسى مەركىزى ئۆگۈنۈش گۇرۇپپىسى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە ئاپتونوم رايونلوق پارتكوم خەلق قۇرۇلتىيى خىزمىتى يىغىنىنىڭ روھىنى ئۆگەندى مىسىر زوڭتوڭ سايلىمىدا مۇسۇلمان بۇرادەرلەر جەمئىيىتىگە تەۋە ھەققانىيەت پارتىيېسىنىڭ رەئىسى مورشى بىلەن سابىق زوڭلى سەفىقنىڭ بېلەتكە ئېرىشىش نىسبىتى ئالدىنقى ئىككى ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر يىلى تەخمىنەن مىڭدىن مىڭغىچە ئەيدىز بىمارى ئۆلۈپ كېتىدىكەن كورىيە ئارمىيە دائىرىلىرى چاۋشيەن يادرو سىنىقى ئېلىپ بېرىشقا تەييارلىنىپ بولدى دېدى قىزىق نوقتا خەۋەرلىرى مەركىزىي ئاخبارات ئىدارىسىنىڭ گېرمانىيەلىك جاسۇس ئىشلەتكەنلىكى ئاقساراينىڭ ئوغىسىنى قايناتتى مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى ئامېرىكا نيۇ يورك ۋاقىت گېزىتىنىڭ مۇشۇ ئاينىڭ كۈنىدىكى خەۋىرىگە ئاساسلانغاندا ئامېرىكىنىڭ گېرمانىيەلىك جاسۇس قوبۇل قىلغانلىق سەتچىلىكى يېقىندا ئاشكارىلانغاندىن كېيىن ئاقساراينىڭ ئەمەلدارلىرى قاتتىق غەزەپلىنىپ مەركىزىي ئاخبارات ئىدارىسىدىن نېمىشقا ئۆزلىرىگە ۋاقتىدا مەلۇم قىلمىغانلىقىنى سۈرۈشتۈرگەن نۆۋەتتە بۇ جاسۇسلۇق دېلوسىنىڭ تەپسىلاتى ئېنىق بولمىسىمۇ لېكىن سنوۋدېن دۆلەت خەۋپسىزلىك ئىدارىسىنىڭ گېرمانىيەگە قارىتىلغان جاسۇسلۇق ھەرىكىتىنى ئاشكارىلىغاندىن بۇيان بۇ دېلو ئىككى دۆلەت ئارىسىدىكى ئىشەنچكە يەنە بىر قېتىم سەلبىي تەسىر ئېلىپ كەلگەن شۇ جاي ۋاقتى مۇشۇ ئاينىڭ كۈنى گېرمانىيە فېدىراتسىيە تەپتىش نازارىتى گېرمانىيە فېدىراتسىيە ئاخبارات ئىدارىسىنىڭ ياشلىق بىر خادىمىنى قولغا ئالغانلىقىنى جاكارلىغان بۇ خادىم ئامېرىكا قاتارلىق دۆلەتلەرنى ئاخبارات بىلەن تەمىنلەش گۇمانىغا چېتىلىپ قالغان گېرمانىيەلىك بۇ ئاخبارات خادىمى قولغا ئېلىنغاندىن كېيىن ئۆزىنىڭ بىر تەرەپتىن گېرمانىيە تاشقى ئاخبارات ئىدارىسى ئۈچۈن خىزمەت قىلسا يەنە بىر تەرەپتىن يىلىدىن باشلاپ ئامېرىكا مەركىزىي ئاخبارات ئىدارىسى ئۈچۈن خىزمەت قىلغانلىقى ھەم مەركىزىي ئاخبارات ئىدارىسىنى دىن ئارتۇق ئاخبارات بىلەن تەمىنلىگەنلىكىنى ئىقرار قىلغان گېرمانىيەلىك بۇ ئاخبارات خادىمى قولغا ئېلىنغان كۈنى ئامېرىكا زۇڭتۇڭى ئوباما بىلەن گېرمانىيە زۇڭلىسى مېركېل ئۇكرائىنا ۋەزىيىتى توغرىسىدا تېلىفونلاشقان مېركېل شۇ چاغدا بۇ ئىشنى تىلغا ئالمىغان لېكىن خەۋەردە كۆرسىتىلىشىچە ئامېرىكاگېرمانىيە مۇناسىۋىتى ئىنتايىن نازۇك بولۇپ قالغان پەيتتە ئامېرىكا زۇڭتۇڭىنىڭ تۇيۇقسىز ئاشكارىلانغان جاسۇسلۇق پائالىيىتىدىن پۈتۈنلەي خەۋەرسىز قېلىشى ئاقساراي ئەمەلدارلىرىنىڭ مەركىزىي ئاخبارات ئىدارىسىگە بولغان غەزىپىنى قوزغىغان ئۇلار مەركىزىي ئاخبارات ئىدارىسىنىڭ قوماندانلىق زەنجىرىدە كىملەرنىڭ بۇ ئىشتىن خەۋەردار ئىكەنلىكى نېمە ئۈچۈن بۇ خەۋەرنىڭ ئوباما ئۆزى تېلىفون بېرىشتىن بۇرۇن زۇڭتۇڭ ئىشخانىسىغا يەتكۈزۈلمىگەنلىكىنى ئۇقماقچى بولغان خەۋەردە ئېيتىلىشىچە مەركىزىي ئاخبارات ئىدارىسى گرمانىيە فېدىراتسىيە ئاخبارات ئىدارىسىدىكى ئۆزلىرى ئۈچۈن خىزمەت قىلىدىغان بۇ خادىمنىڭ قولغا ئېلىنغانلىق خەۋىرىنى ئاقسارايغا مەلۇم قىلمىغان بۇ ئىش ئىككى دۆلەت رەھبىرىنىڭ تېلىفون سۆھبىتىدىن كۈن بۇرۇن يۈز بەرگەنلىكتىن مەركىزىي ئاخبارات ئىدارىسىنىڭ قىلمىشى ئاقساراي ئەمەلدارلىرىنى ھەيران قالدۇرغان گېرمانىيە ئاخبارات ۋاستىلىرىنىڭ خەۋەر قىلىشىچە مەركىزىي ئاخبارات ئىدارىسى ھەپتە ئىلگىرىلا گېرمانىيە دائىرىلىرىنىڭ بۇ ئاخبارات خادىمىنى نازارەت قىلىۋاتقانلىقىنى ھېس قىلغان بولۇشى مۇمكىن ئىكەن ئامېرىكىلىق بىر ئەمەلدار مۇنداق دېگەن يادرولۇق مەسىلە بۇ ئالاھىدە خادىم ھەققىدىكى ئۇچۇرنىڭ مەركىزىي ئاخبارات ئىدارىسى قوماندانلىق زەنجىرىدىكى قايسى ئەمەلدارنىڭ قۇلىقىغا يەتكەنلىكىدە بۇ خەۋەر زادى بېرلېن تارماق پونكىتىنىڭ باشلىقى تەرىپىدىن باستۇرۇپ قويۇلغانمۇ ياكى مەركىزىي ئاخبارات ئىدارىسىنىڭ باشلىقى برۇنناننى ئۆز ئىچىگە ئالغان يۇقىرى دەرىجىلىك ئەمەلدارلارغا يەتكۈزۈلگەنمۇ ئاقسارايغا ھەقىقىي ئەھۋالنى ۋاقتىدا مەلۇم قىلىش برۇنناننىڭ مۇھىم مەجبۇرىيەتلىرىنىڭ بىرى ئىدى ھازىرچە بۇ قېتىملىق جاسۇسلۇق دېلوسىنىڭ تەپسىلاتى يەنىلا ئېنىق ئەمەس ئاخبارات ئورگانلىرىنىڭ ئەمەلدارلىرى بۇ ئىشقا باھا بېرىشنى رەت قىلغان گېرمانىيەلىك بۇ ئالاھىدە خادىمنىڭ دائىرىلەرگە قايسى ئىشلارنى ئىقرار قىلغانلىقىنىمۇ بىلگىلى بولمايدۇ بىراق بۇ ۋەقە ئاخبارات مەدەنىيىتى بىلەن دۆلەت باشقۇرۇش مەدەنىيىتىنىڭ زىددىيىتى كەلتۈرۈپ چىقارغان جىددىي ۋەزىيەتنى ئېنىق ئەكس ئەتتۈرۈپ بەرگەن سەزگۈر ماتېرىياللارنى ئىمكانقەدەر كۆپ توپلاش ئېھتىياجى ئاخبارات مەدەنىيىتىنى ئىلگىرى سۈرىدىغان ئامىل دۆلەت باشقۇرۇش مەدەنىيىتىدە بولسا دىپلوماتىك نىشان بىرىنچى ئورۇندا تۇرىدۇ بۇ دېلو يەنە يىل ئىلگىرى ئامېرىكا دۆلەت خەۋپسىزلىك ئىدارىسىنىڭ سابىق ياللانما خادىمى سنوۋدېن مەزكۇر ئىدارىنىڭ جاسۇسلۇق پائالىيىتىنى ئاشكارىلىغاندىن كېيىن بارلىققا كەلگەن ئىتتىپاقداشلىرىنى نازارەت قىلغاندا پايدا كۆپ بولامدۇ ياكى زىيانمۇ دېگەن مەسىلىنى قايتىدىن كۈچەيتكەن ئاقسارايدىكى يۇقىرى دەرىجىلىك ئەمەلدارلار بۇ ئەڭ يېڭى ئەيىبلەشنىڭ ئامېرىكا گېرمانىيە مۇناسىۋىتىگە زىيان يەتكۈزۈش ئېھتىماللىقىغا بەكرەك كۆڭۈل بۆلگەن بۈگۈنگە قەدەر ئوباما بىلەن مېركېل يەنىلا سنوۋدېننىڭ مەخپىيەتلىكنى ئاشكارىلاش ۋەقەسى كەلتۈرۈپ چىقارغان ئۆزئارا ئىشەنمەسلىك تۇيغۇسىنى تۈگىتىش بىلەن ئاۋارە بولماقتا ئامېرىكا دۆلەت خەۋپسىزلىك ئىدارىسىنىڭ مېركېلنىڭ تېلىفونىنى ئوغرىلىقچە ئاڭلىغانلىقى دەل ئاشكارىلانغان سەتچىلىكلەرنىڭ بىرى ئىدى بىر ئەمەلدار مۇنداق دېگەن گەرچە ئوباما بۇ ئىشقا يۈكسەك دىققەت قىلسىمۇ لېكىن ئۇنىڭ بۇ ئىشنى دەپ مېركېل بىلەن ئالايىتەن ئالاقىلىشىش ئويى يوق ھۆكۈمەتتىكى بەزى كىشىلەر گېرمانىيە بۇ دېلوغا زىيادە كۈچلۈك ئىنكاس قايتۇرۇپ كەتمەسلىكى كېرەك شۇنداقلا بۇ دېلونى دۆلەت خەۋپسىزلىك ئىدارىسىنىڭ نازارەت قىلىشى كەلتۈرۈپ چىقارغان خۇسۇسىي مەخپىيەتلىك مەسىلىسى بىلەن ئارىلاشتۇرۇۋېتىشكە بولمايدۇ دەپ قارىماقتا قىسمەن مۇتەخەسسىسلەر يېقىنقى جىددىي ۋەزىيەت ھۆكۈمەتنىڭ ئامېرىكاگېرمانىيە مۇناسىۋىتىگە يېتەرلىك ئېتىبار بەرمىگەنلىكىنى ئاشكارىلاپ قويدى دەپ قارىغان ئامېرىكىنىڭ گېرمانىيەدە تۇرۇشلۇق سابىق باش ئەلچىسى برۇم مۇنداق دېگەن بۇش ۋە ئوباما بولۇپمۇ ئوباما ئامېرىكاگېرمانىيە مۇناسىۋىتىنى ھەل قىلىنىپ بولغان مەسىلە ئامېرىكىنىڭ ئەندىشە قىلىشىنىڭ ئورنى يوق دەپ قارايدۇ يېقىنقى ۋەقە ئەھۋالنىڭ دەل ئەكسىچە ئىكەنلىكىنى ئاشكارىلاپ قويدى گېرمانىيەنىڭ ئۆزى ۋە تاشقى دۇنياغا بولغان قارىشىدا ھېلىھەم كۈچلۈك زىددىيەت مەۋجۇت ئاقساراي ۋە مەركىزىي ئاخبارات ئىدارىسى بۇ دېلوغا باھا بېرىشنى رەت قىلغان دۆلەت خەۋپسىزلىك كومىتېتىنىڭ ئايال باياناتچىسى ھېيدېن بولسا بىز بۇ قېتىملىق ئەيىبلەشتىن كىملەرنىڭ خەۋەردار ئىكەنلىكى نېمىلەردىن خەۋەردار ئىكەنلىكى ھەم قاچان خەۋەردار بولغانلىقىنى ھەرگىز ئاشكارا مۇزاكىرە قىلمايمىز دېگەن شۇ جاي ۋاقتى مۇشۇ ئاينىڭ كۈنى ئامېرىكا كېڭەش پالاتاسى ئاخبارات كومىتېتىنىڭ رەئىسى ۋە پېشقەدەم ھەيئەت ئەزالىرى ئوخشىمىغان زىيارەتلەر داۋامىدا برۇنناننىڭ بۇ ئىشنى كومىتېتقا مەلۇم قىلغانلىقىنى بىلدۈرگەن بىراق ئۇلار ئىككى تەرەپنىڭ سۆھبەت تەپسىلاتىنى ئاشكارىلاشنى رەت قىلغان ئاخبارات كومىتېتىنىڭ ئايال رەئىسى كالىفورنىيە ئىشتاتىنىڭ دېموكراتلار پارتىيەسىگە تەۋە كېڭەش پالاتا ئەزاسى دىئان ۋانستان بۇ ئىشنىڭ ئامېرىكاگېرمانىيە مۇناسىۋىتىنى بۇزغۇنچىلىققا ئۇچرىتىشىدىن ئەندىشە قىلغان ئۇ بىز ئومۇمىي ۋەزىيەتكە ئىتتىپاقداشلىرىمىز بىلەن بولغان مۇناسىۋەتنى قانداق بىر تەرەپ قىلىۋاتقىنىمىزغا قايتىدىن قاراپ چىقىشىمىز كېرەك مەن گېرمانىيەنىڭ پارلامېنت ئەزاسى بىلەن كۆرۈشتۈم قارشى تەرەپ بۇ دېلودىن ئىنتايىن ئەنسىرەۋاتقانلىقىنى بىلدۈردى دېگەن جۇمھۇرىيەتچىلەر پارتىيەسىنىڭ پېشقەدەم ئەزاسى جورجىيا ئىشتاتىنىڭ كېڭەش پالاتا ئەزاسى چامبىرېس بولسا بۇ قېتىملىق ئەيىبلەشكە قارىتا تولىمۇ ئېھتىياتچان ھالدا مەن بۇ ۋەقەنىڭ قانچىلىك دەرىجىدە راستلىقىنى ئۇقمايمەن ھالبۇكى بۇ دەل مەزكۇر دېلونىڭ قانچىلىك ئېغىرلىقىنى بەلگىلەيدىغان ئامىل دېگەن كۆرۈپ قويۇڭ ئۇيقۇنى ياخشىلايدىغان يېمەكلىكلەر گۈزەللىك قەيسەرلىكتە پىتنەپاساتقا ئىشەنمەڭ ئۆزىڭىزنىڭ كەيپىياتىغا دىققەت قىلىپ باقتىڭىزمۇ يوق كەمبەغەلنىڭ ئىلتىپاتى دادا ئانامغا ياخشى قارا فۇكاڭ شەھىرىدە قىشلىق تەنھەرىكەت مۇسابىقىسى كەيپىياتى قۇيۇق بولدى نادىر ناخشا مۇزىكىلار دادىڭىز چەت ئەل تىلىنى بىلەمدۇ ئىش ھەققى تەبرىكلەش پائالىيىتى كۆزۈم كۆرمىسۇن تور بېتىمىزدىكى يېڭىلىقلار
