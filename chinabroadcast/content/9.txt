 خوتەن ناھىيەسى باغچى بازىرىدىكى بىر مەكتەپتە ئېلىپ بېرىلىۋاتقان قۇرۇلۇش نەق مەيدانىدا گۈمۈرۈلۈپ چۈشۈش ۋەقەسى يۈز بەردى يەكەن ناھىيەلىك رادىيو تېلېۋىزىيە ئىدارىسى قاتارلىق ئۈچ ئورۇن مەملىكەت بويىچە دېھقانلارغا مۇلازىمەت قىلىش ئاساسىي قاتلامنىڭ مەدەنىيەت قۇرۇلۇشىغا مۇلازىمەت قىلىشتىكى ئىلغار كوللېكتىپ شەرىپىگە ئېرىشتى خوتەن ۋىلايىتىدىكى ئۈچ دەرىجىلىك رەھبىرى كادىرلار ھال ئېيتپ كەلگەن ئاممىنى كۈتىۋىلىشنى قەرەرلىك ئېلىپ بېرىپ مەسىلىلەرنى ھەل قىلىشنى ماسلاشتۇردى خوتەن ۋىلايىتى ئالىي مەكتەپنى پۈتتۈرگەنلەرنى ئىشقا ئورۇنلاشتۇرۇش مۇلازىمەت ئېيى پائالىيتىنى قانات يايدۇردى خوتەن ۋىلايىتى بوستانلىق بەرپا قىلىپ قۇملىشىشىنڭ ئالدىنى ئېلىشتا قوش ئۈنۈم ياراتتى خوتەن ۋىلايىتىنىڭ مائارىپ سېلىنمىسى مىليارد يۈەندىن ئېشىپ مائارىپ تەرەققىياتىغا پۇختا ئاساس سېلىندى بۇيىل خوتەن ۋىلايىتىنىڭ ساياھەتچىلىك كىرىمىدە يۇقۇرى نىسبەتتە ئېشىش بارلىققا كەلدى خوتەن ناھىيەسى مۇتەخەسىسلەرنى تەكلىپ نۆۋەتلىك ياڭاقچىلىق كەسپىي مۇھاكىمە مۇنبىرى ئۆتكۈزدى خوتەن ۋىلايىتىدە ئادەم ئىشلەتكۈچى ئورۇنلار بىلەن ئىشلەمچىلەرنىڭ ئەمگەك توختامى ئىمزالاش نىسبىتى گە يەتتى خوتەن ۋىلايىتىنىڭ قاراقاش ناھىيىسى پارتىيەنىڭ مىللەت ۋە دىن سىياسىتى دۆلەتنىڭ قانۇن پەرمانلىرى توغرىسىدا مەخسۇس لېكسىيە ئورۇنلاشتۇردى خوتەن ۋىلايىتىنىڭ ئەمىن ئۆي قۇرۇلۇشىدا ئىش باشلاش نىسبىتى گە يەتتى خوتەن ۋىلايىتىدىكى سانائەت كارخانىلىرىنىڭ ئومۇمى مەھسۇلات قىممىتى ئاشتى خوتەن ناھىيەسىدە دۆلەتنىڭ رادىئو تېلىۋىزىيەنى ھەممە كەنتىلەرگە يەتكۈزۈش قۇرۇلۇشىدىن مىڭ ئائىلىلىك دېھقان چارۋىچى بىۋاستە نەپ ئالدى ئەنخۇي ئۆلكىسى گۇماناھىيەسىگە مىليون يۈەن ياردەم قىلىپ نۇقتىلىق تۈرنى يولغا قويدى دۆلەتنىڭ خوتەن ۋىلايىتىدىكى مىللىي سودا مىللىي تاۋار كارخانىلىرىغا بەرگەن قەرز پۇل ئۆسۈم ياردەم پۇلى مىلىيون يۈەندىن ئاشتى خوتەن ۋىلايىتىنىڭ سودىگەر چاقىرىپ مەبلەغ جەلىب قىلىشتا ئەمەلىي ئۆنۈم بارلىققا كەلدى بۇيىل خوتەن ۋىلايىتىدە قۇرۇلىدىغان يۇرۇش يەسلى قۇرۇلىشى ئوڭۇشلۇق يولغا قويۇلدى بۇ يىلنىڭ ئالدىنقى يېرىمىدا خوتەن ۋىلايىتىدىكى دەرىجىلىك مەنزىرە رايون ۋە نوقتىلىرىنىڭ كىرىمىدە ئېشىش بارلىققا كەلدى خوتەن ۋىلايىتىدە نوقتلىق كەسىپلەر ۋە مۇقىم مۈلۈككە سېلىنغان مەبلەغ كۆپەيدى خوتەن ناھىيەسىدە ئۈچ يىلدا مىڭ نەچچە دېھقان ئائىلىسى دۆلەتنىڭ ئەمىن ئۆي ياردىمىدىن بەھرىمان بولدى خوتەن ناھىيەسى ياغاچ قاچا قۇچا ئويمىچىلىقى كەسپىنى قايتىدىن جانلاندۇردى خوتەن شەھىرىگە قاراشلىق تەۋەككۈل يېزىسىنىڭ قارادۆڭ كەنتىدە يېزا نۇسخىسىدىكى تەلەي يۇلتۇزى تېمىسىدىكى زېھىن سىناش پروگراممىسى نۇرغۇن ياشلارنى جەلىپ قىلدى خوتەن شەھىرى شورباغ يېزىسىنىڭ خىنە كەنتىدىكى دېھقانلار دېھقانچىلىقنىڭ ئارىسالدا ۋاقىتلىرىدا كەنتتىكى دېھقانلار قىرائەتخانىسىغا بېرىپ كىتاب ئوقۇيدىغان بولدى خوتەن شەھىرى سۇ بىلەن تەمىنلەش تۈر قۇرۇلۇشلىرىنى تېزلىتىپ شەھەر يېزىلارنىڭ ئىچىدىغان سۇ بىخەتەرلىكىگە كاپالەتلىك قىلدى خوتەن ناھىيەسى مىۋىچىلىك ئاساسي گەۋدە قىلىنغان ئورمانچىلىقنى تەرەققىي قىلدۇرۇپ مىۋىچىلىكنى دېھقانلارنىڭ كىرىمىنى كۆپەيتىشتىكى ئاساسىي مەنبەگە ئايلاندۇردى خوتەن ناھىيەسى پائال ئورمان تىكىپ قۇملىشىشنى تىزگىلەپ يېشىل ئېكولوگىيە قۇرۇلۇش قەدىمىنى تېزلەتتى ئاپتونوم رايونلۇق ئاخبارات نەشرىيات ئىدارىسى پارتگۇرۇپپىسى خوتەن شەھىرى شورباغ يېزىسىنىڭ خېنىي كەنتىدە نامراتلارنى يۆلەش خىزمىتىنىڭ ئەمەلىلىشىش ئەھۋالىنى مەخسۇس تەتقىق قىلدى لوپ ناھىيەسىنىڭ باغچىلاشقان سانائەت رايونى ناھىيەنى قۇدرەت تاپقۇزۇپ خەلققە نەپ يەتكۈزۈش تىپىدىكى رايونغا ئايلاندى خوتەن ناھىيەسىنىڭ سودىگەر چاقىرىپ مەبلەغ جەلپ قىلىش ئارقىلىق ئەمەلىيلەشتۇرگەن مەبلىغى بۇلتۇرقى ئوخشاش مەزگىلدىكىدىن ئاشتى قاراقاش ناھىيەسىنىڭ چارۋىچىلىق ئىشلەپچىقىرىشىدا مۇقىم ئېشىش بارلىققا كەلدى خوتەن شەھىرى كەلكۈندىن مۇداپئە كۆرۈشنىڭ تۈرلۈك تەييارلىقلىرىنى پۇختا ئىشلىدى خوتەن ۋىلايىتىدە تۈنۆگۈن مىڭ ھەرمىللەت ئوقۇغۇچىسى ئالىي مەكتەپ ئىمتىھانىغا قاتناشتى خوتەن ۋىلايىتىدىكى يۇرۇڭقاش دەرياسىغا تۇيۇقسىز كەلكۈن كېلىپ دەريادا قاشتېشى ئىزدەۋاتقان ئادەم قاپسىلىپ قالدى خوتەن ناھىيە ئىسلامئاۋات يېزىسىدا ئەمىن ئۆي قۇرۇلۇشى ئوڭۇشلۇق ئېلىپ بېرىلدى كېرىيە ناھىيەسى بۇ يىل كىرگەندىن بۇيان قىيىنچىلىقى بار ئاممىغا مىليۇن يۈەندىن كۆپرەك داۋالىنىش ياردەم پۇلى تارقىتىپ بەردى تىيەنجىن شەھىرىنىڭ ياردىمى بىلەن خوتەن ۋىلايىتىدىكى ئۈچ ناھىيەدە مىڭ ئايال ئىشقا ئورۇنلاشتى چىرا ناھىيەسىدىكى شوپۇر تەربىيلەش ئاپتومىبىل تەكشۈرۈش مەيدانى قۇرۇلۇشىدا يېقىندا ئىش باشلاندى كۈنەس ناھىيەسىنىڭ ئارال تۆپە يېزا باشلانغۇچ مەكتەپتە ئاتا ـ ئانىلاردىن ئىنئانە نامىدا پۇل يىغىپ تەكلىپلىق ئوقۇتقۇچىلارنىڭ ئىش ھەققىنى بېرىش ئەھۋالى كۆرۈلدى بۇ يىل خوتەن ۋىلايىتىنىڭ ئالدىنقى تۆت ئايدا مۇقىم مۈلۈككە سالغان مەبلىغى ئاشتى خوتەن ناھىيە ئىسلامئاۋات يېزىسىدا ئەمىن ئۆي قۇرۇلۇشى ئوڭۇشلۇق ئېلىپ بېرىلدى خوتەن ناھىيىلىك جامائەت خەۋپسىزلىكى ئىدارىسى تەۋەككۈل ساقچىخانىسى قانۇنسىز ھالدا ئەپيۇنگۈلنى كىرىشتۈرۈپ تېرىش دېلوسىنى تەكشۈرۈپ بىر تەرەپ قىلدى تىيەنجىن شەھىرى كېرىيە ناھىيىلىك تولۇق ئوتتۇرا مەكتىپىنىڭ ئوقۇتۇش شارائىتىنى ياخشىلاپ بەردى خوتەن ۋىلايىتىنىڭ خوتەن ناھىيىسى تەۋەككۈل يېزىسى ياشلار ئۆسمۈرلەرنىڭ ياشلىقتىكى ئارزۇ ــ گۈزەل شىنجاڭ قۇرۇش ئاساسىي تېما پائالىيىتىنى قانات يايدۇردى خوتەن ناھىيەسى قېتىملىق مىللەتلەر ئىتتىپاقلىقى تەربىيەسى ئېيى پائالىيتى ۋە ئۈچ چوڭ تەربىيە پائالىيتى بويىچە لېكسىيە سۆزلەش پائالىيتى ئېلىپ باردى خوتەن ۋىلايىتىنىڭ يول ترانسىپورت ئىشلىرىدا زور تەرەققىيات بارلىققا كەلدى كېرىيە ناھىيەسىدە بۇيىل سېلىنىدىغان ئەمىن ئۆي قۇرۇلۇشىدا ئىش باشلاش نىسبىتى گە يەتتى ئەنخۇي ئۆلكىسى گۇما ناھىيەسىنىڭ داۋالاش ئىشلىرىغا تېخنىكا جەھەتتىن ياردەم بېرىپ دېھقانلارنىڭ كېسەل كۆرسىتىش داۋالىنىش قىيىن بولۇش مەسىلىسىنى ئۈنۈملۈك ھەل قىلدى كېرىيە ناھىيىسى ئىقتىسادىي زىرائەتلەرنىڭ تېرىلغۇ كۆلىمىنى كېڭەيتىپ دىنىي زاتلارنى بېيىش يولىدا مېڭىشىغا تۈرتكە بولدى بۇيىلنىڭ ئالدىنقى ئېيىدا خوتەن ۋىلايىتىنىڭ سودىگەر چاقىرىپ مەبلەغ جەلب قىلىش تۈرىدە ئەمەلىيلەشكەن مەبلەغ مىليارد مىليون يۈەنگە يەتتى خوتەن ۋىلايىتىدىكى ئۈچ ناھىيە بىر شەھەردە ئائىلە ساغلام ئائىلە بولۇپ باھالاندى بۇ بەتنىڭ نەشىر ھوقۇقى شىنجاڭ خەلق رادىيو ئىستانسىسىغا تەۋە
