ئەسلىمە ئۈنۋېرسال رادىيو سەنئەت رادىيو كەسىپ ئىستىلى تاللاپ ئاڭلاش ئۇيغۇرچە ئۈگۈنۈش خەنزۇچە ئۈگۈنۈش ئىستانسىمىز ھەققىدە ئاپتونۇم رايوننىڭ لىكسىيە ئۆمىكى پارتىيە نۆۋەتلىك مەركىزىي كومىتېتى ئومۇمىي يىغىنى ۋە ئاپتونۇم رايونلۇق پارتكوم نۆۋەتلىك كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ھەققىدە دوكلات بەردى ئاۋامنىڭ كەڭ زىمىندا ئىگىلىك تىكلىشى بىپايان جەنۇبىي شىنجاڭ زېمىنىدا كىشىلىك ھاياتنىڭ گۈزەل كارتىنىسىنى يارىتايلى ئاپتونوم رايونلۇق ئىنتىزام تەكشۈرۈش كومىتېتى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە جاڭ چۈنشيەن شۇجىنىڭ مۇھىم سۆزىنىڭ روھىنى يەتكۈزدى ئاپتونوم رايونلوق خەلق قۇرۇلتىيى دائىمىي كومىتېتى پارتگۇرۇپپىسى مەركىزى ئۆگۈنۈش گۇرۇپپىسى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە ئاپتونوم رايونلوق پارتكوم خەلق قۇرۇلتىيى خىزمىتى يىغىنىنىڭ روھىنى ئۆگەندى مىسىر زوڭتوڭ سايلىمىدا مۇسۇلمان بۇرادەرلەر جەمئىيىتىگە تەۋە ھەققانىيەت پارتىيېسىنىڭ رەئىسى مورشى بىلەن سابىق زوڭلى سەفىقنىڭ بېلەتكە ئېرىشىش نىسبىتى ئالدىنقى ئىككى ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر يىلى تەخمىنەن مىڭدىن مىڭغىچە ئەيدىز بىمارى ئۆلۈپ كېتىدىكەن كورىيە ئارمىيە دائىرىلىرى چاۋشيەن يادرو سىنىقى ئېلىپ بېرىشقا تەييارلىنىپ بولدى دېدى قىزىق نوقتا خەۋەرلىرى باشقىلارغا ياردەم بېرىشنى خوشاللىق دەپ بىلىش ۋە كەسپنى قەدىرلەپ خالىسانە تەقدىم قىلىش گۈزەللىكى مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى يىل ئىلگىرى كورلا شەھىرى تىكىچى يېزىسى ئۈستۈن چەچى كەنتىدىكى دېھقان زۆرىگۈل ھېمىت مىڭە پارالىچ كېسىلىگە گىرىپتار بولغان بىر خەنزۇ قىز بۇۋاقنى بىقىۋالدى ئۇ بۇ قىز بۇۋاقنى داۋالىتىش داۋامىدا ئەتراپىدا ئۆزىنىڭ بىقىۋالغان بۇ قىزىغا ئوخشاش مىڭە پارالىچ كېسىلىگە گىرىپتار بولغان بالىلارنىڭ ناھايىتى كۆپلىكىنى بايقىدى ئۇ باشقىلارنىڭ ئەنسىزلىك ئىچىدىكى كۆزلىرىنى كۆرۈپ كورلىدا مىيىپ بالىلارنى بېقىش مەركىزى قۇرۇشنى قارار قىلدى شۇنىڭدىن كېيىنكى يىلدا ئۇ ئىلگىرى كىيىن بولۇپ ئوخشاش بولمىغان مىللەتتىكى بالىنى بۇ يەردە باقتى غۇلجا ناھىيىسىنىڭ ئۇچئۇن خۇيزۇ يېزىسى توختىيۈزى كەنتىدىكى گۇ شۇمېي ئۆزىنىڭ ھۆنىرىگە تايىنىپ ئۆز ئالدىغا تىككۈچىلىك دۇكىنى ئاچتى ھەمدە بۇنىڭ بىلەن بىيىش يولىدا ماڭدى نەچچە يىلدىن بۇيان ئۇ ئىلگىرى كېيىن بولۇپ ئويغۇر قازاق خۇيزۇ قاتارلىق ئوخشاش بولمىغان مىللەتتىن بولغان شاگىرتتىن مىڭدىن كۆپرەكنى قۇبۇل قىلىپ نەچچە ئون مىڭ يۈەن ئاجىرىتىپ شاگىرىتلىرىنىڭ مۇستەقىل ئىگىلىك تىكلىشىگە ياردەم بەردى خوتەندىكى يېزىدىن چىققان ئويغۇر يىگىتى روزى مەمەت باقى نىفېتلىكتە خىزمەتكە قاتناشقاندىن كېيىن خەنزۇ ئۇستىسى ۋە خىزمەتداشلىرىنىڭ ئورتاق ياردەم بېرىشى بىلەن بىر ئېغىزمۇ خەنزۇچىنى بىلمەيدىغان ھالەتتىن تەدرىجىي ئىلگىرىلەپ خەنزۇتىلى سەۋىيە ئىمتىھانىدا دەرىجىدىن ئۆتۈپ كېنىشكە ئالدى ئۇ پۈتۈن زېھنىنى تېخنىكا ئىسلاھاتىغا قارىتىپ ئىلگىرى كېيىن بولۇپ تۈرلۈك تېخنىكىلاردا يېڭىلىق يارىتىشقا قاتناشتى ئۇ تۈزگەن ئاز سانلىق مىللەت ئىشچى خىزمەتچىلىرىنىڭ نېفىت ئايرىش ئەمەلىي مەشغۇلاتى پىنيىنچە ئوقۇتۇش ماتىرىيالى قاتارلىقلار شىنجاڭ نېفىتلىكى شىركىتىنىڭ ئاز سانلىق مىللەت ئىشچى خىزمەتچىلىرىنى تەربىيىلەش ئوقۇتۇش ماتىرىيالى بولۇپ قالدى ئۇ يېشىدا قاراماي نېفىتلىكىنىڭ تېخنىك مۇتەخەسسىسى يېشىدا مەملىكەت بويىچە ياشلار تېخنىك ماھىرى يېشىدا گۇرۇھ شىركىتىدىكى بىردىنبىر ئويغۇر تېخنىك
