ئەسلىمە ئۈنۋېرسال رادىيو سەنئەت رادىيو كەسىپ ئىستىلى تاللاپ ئاڭلاش ئۇيغۇرچە ئۈگۈنۈش خەنزۇچە ئۈگۈنۈش ئىستانسىمىز ھەققىدە ئاپتونۇم رايوننىڭ لىكسىيە ئۆمىكى پارتىيە نۆۋەتلىك مەركىزىي كومىتېتى ئومۇمىي يىغىنى ۋە ئاپتونۇم رايونلۇق پارتكوم نۆۋەتلىك كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ھەققىدە دوكلات بەردى ئاۋامنىڭ كەڭ زىمىندا ئىگىلىك تىكلىشى بىپايان جەنۇبىي شىنجاڭ زېمىنىدا كىشىلىك ھاياتنىڭ گۈزەل كارتىنىسىنى يارىتايلى ئاپتونوم رايونلۇق ئىنتىزام تەكشۈرۈش كومىتېتى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە جاڭ چۈنشيەن شۇجىنىڭ مۇھىم سۆزىنىڭ روھىنى يەتكۈزدى ئاپتونوم رايونلوق خەلق قۇرۇلتىيى دائىمىي كومىتېتى پارتگۇرۇپپىسى مەركىزى ئۆگۈنۈش گۇرۇپپىسى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە ئاپتونوم رايونلوق پارتكوم خەلق قۇرۇلتىيى خىزمىتى يىغىنىنىڭ روھىنى ئۆگەندى مىسىر زوڭتوڭ سايلىمىدا مۇسۇلمان بۇرادەرلەر جەمئىيىتىگە تەۋە ھەققانىيەت پارتىيېسىنىڭ رەئىسى مورشى بىلەن سابىق زوڭلى سەفىقنىڭ بېلەتكە ئېرىشىش نىسبىتى ئالدىنقى ئىككى ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر يىلى تەخمىنەن مىڭدىن مىڭغىچە ئەيدىز بىمارى ئۆلۈپ كېتىدىكەن كورىيە ئارمىيە دائىرىلىرى چاۋشيەن يادرو سىنىقى ئېلىپ بېرىشقا تەييارلىنىپ بولدى دېدى قىزىق نوقتا خەۋەرلىرى ئاپتونوم رايونلۇق پارتكومنىڭ جەنۇبىي شىنجاڭ خىزمىتى يىغىنى ئېچىلدى مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى ئىستانسىمىز خەۋىرى ئاپتونوم رايونلۇق پارتكومنىڭ جەنۇبىي شىنجاڭ خىزمىتى يىغىنى ئاپتونوم رايون قۇرۇلغان يىلدىن بۇيان ئېچىلغان جەنۇبىي شىنجاڭ خىزمىتىنى تۇنجى قېتىم ئومۇمىيۈزلۈك ئورۇنلاشتۇرغان مۇھىم يىغىن ھېساپلىنىدۇ يىغىن قاتناشچىلىرى بۇ قېتىمقى يىغىن جەنۇبىي شىنجاڭنىڭ تەرەققىياتىدا يېڭى يول ئاچتى جەنۇبىي شىنجاڭ يېڭى تارىخىي پۇرسەتنى كۈتىۋالدى دەپ قاراشتى ئاپتونوم رايونلۇق سودا سانائەت مەمۇرىي باشقۇرۇش ئىدارىسىنىڭ باشلىقى مەمتىمىن فېردەۋىس جەنۇبىي شىنجاڭدا ئۇزۇن يىل خىزمەت قىلغان بولۇپ ئۇ مۇنداق دېدى ئاپتونوم رايونلۇق پارتكوم تۇنجى قېتىم جەنۇبىي شىنجاڭ خىزمىتى يىغىنى ئېچىپ جەنۇبىي شىنجاڭنىڭ پۈتكۈل ۋەزىيىتىنى تەپسىلىي ئانالىز قىلدى ساقلانغان مەسىلىلەرمۇ بىر قەدەر ئېنىق ئوتتۇرىغا قويۇلدى ماڭا ھەممىدىن بەك تەسىر قىلغىنى جەنۇبىي شىنجاڭنىڭ ئەبەدىي ئەمىنلىكى ۋە جەمئىيەت مۇقىملىقىنى ساقلاش بولدى مەن جەنۇبىي شىنجاڭدا تۇغۇلۇپ چوڭ بولغان بۇ خەۋەرنى ئاڭلاپ ناھايىتى ئىلھاملاندىم يەنە بىر جەھەتتىن ئىقتىسادنى تەرەققىي قىلدۇرۇش بولۇپمۇ ئىشقا ئورۇنلاشتۇرۇش ئىقتىسادىي تەرەققىيات ۋە ئىجتىمائىي مۇقىملىقتىكى ئاساسىي خىزمەتلەر قاتارىدا چىڭ تۇتۇلىدىغان بولدى بۇ يۈكسەكلىك ئىلگىرى كۆرۈلۈپ باقمىغان ئوخشاشلا جەنۇبىي شىنجاڭدا تۇغۇلۇپ چوڭ بولغان خوتەن ۋىلايىتىنىڭ ۋالىيسى ئەزىز مۇسا مۇنداق دېدىبۇ قېتىمقى يىغىن جەنۇبىي شىنجاڭنىڭ تەرەققىياتىدا يېڭى بەت ئاچتى ئاپتونوم رايونلۇق پارتكومنىڭ جەنۇبىي شىنجاڭ خىزمىتى يىغىنى جەنۇبىي شىنجاڭنىڭ ئىقتىسادىي ۋە ئىجتىمائىي تەرەققىياتى مىللەتلەر ئىتتىپاقلىقى جەمئىيەت مۇقىملىقى ۋە ئەبەدىي ئەمىنلىكىنىڭ يېڭى باشلىنىش نۇقتىسى بولۇپ قالىدۇ قايتىپ بارغاندىن كېيىن يىغىن روھىنى ئەستايىدىل ئۆگىنىپ ۋە ئىگىلەپ خوتەننىڭ ئەمەلىيىتىدە ئومۇمىيۈزلۈك ئىزچىللاشتۇرىمىز جەنۇبىي شىنجاڭ كەڭ زىمىنغا مول بايلىق مەنبەسىگە كۆپ نۇپۇسقا ئىگە بولۇپ جەنۇبىي شىنجاڭ ۋە شىمالىي شىنجاڭ ئاپتونوم رايونىمىزنىڭ تەرەققىياتىدىكى موھىم رايونلار ھېساپلىنىدۇ جەنۇبىي شىنجاڭنىڭ تەرەققىياتى بولمىسا شىنجاڭنىڭ ئەمەلىي كۈچىنى ئاشۇرۇش قۇرۇق گەپ بولۇپ قالىدۇ قەشقەر ۋىلايەتلىك مەمۇرىي مەھكىمىنىڭ مۇئاۋىن ۋالىيسى قەشقەر كونىشەھەر ناھىيەسى پارتكومىنىڭ شۇجىسى جۇ شۆبىڭ مۇنداق دېدى بۇ قېتىمقى يىغىن بىز ئۈچۈن تېپىلغۇسىز پۇرسەت ھېساپلىنىدۇ ناھىيەمىز پۇرسەتنى چىڭ تۇتۇپ بىر بەلباغ بىر يول سىتراگىيەسىدىكى يادرولۇق رايون قۇرۇلۇشى ۋە جۇڭگو پاكىستان كارىدورى قۇرۇلۇشىنى ئېلىپ بېرىۋاتىدۇ بىز كەسپلەرنى تەرەققىي قىلدۇرۇش ئارقىلىق ئىشقا ئورۇنلاشتۇرۇشنى ئىلگىرى سۈرۈپ ئاممىنى نامراتلىقتىن قۇتۇلۇپ بېيىش يولىدا مېڭىشقا يېتەكلەيمىز جەنۇبىي شىنجاڭ خىزمىتىدىكى يېتەكچى ئىدىيە ۋە نىشان ۋەزىپىلەر ئېنىق بېكىتىلدى بۇنى قانداق ئەمەلىيلەشتۈرۈش كېرەك ئاپتونوم رايونلۇق تەرەققىيات ۋە ئىسلاھات كومىتېتىنىڭ مۇدىرى جاڭ چۈنلىن مۇنداق دېدى ئاپتونوم رايوننىڭ بەش يىللىق يىرىك پىلانىنى تۈزۈشتىن باشقا جەنۇبىي شىنجاڭ رايونىدىكى ۋىلايەت بىر ئوبلاستنىڭ خەلق ئىگىلىكى ۋە ئىجتىمائىي تەرەققىيات پىلانىنى تۈزىۋاتىمىز بۇ بىر قېتىملىق رايون خاراكتىرلىك پىلان بولۇپ بۇ ئارقىلىق جەنۇبىي شىنجاڭدىكى ۋىلايەت بىر ئوبلاستنىڭ ئۇل مۇئەسسەسەلەر قۇرۇلۇشى ئىكولوگىيە مەدەنىيلىكى قۇرۇلۇشى كەسپ تەرەققىياتى شەھەر بازارلاشتۇرۇش ئىشى نامراتلارنى يۆلەش ئۆتكىلىگە ھوجۇم قىلىشنى بىر تۇتاش پىلانلاپ بۇ قېتىمقى يىغىندا ئوتتۇرىغا قويۇلغان جەنۇبىي شىنجاڭنىڭ مۇقىملىقى ۋە ئەبەدىي ئەمىنلىكىنى ئىشقا ئاشۇرۇش توغرىسىدىكى پىكىرلەرنى ھەقىقىي ئەمەلىيلەشتۈرگىلى
