ئەسلىمە ئۈنۋېرسال رادىيو سەنئەت رادىيو كەسىپ ئىستىلى تاللاپ ئاڭلاش ئۇيغۇرچە ئۈگۈنۈش خەنزۇچە ئۈگۈنۈش ئىستانسىمىز ھەققىدە ئاپتونۇم رايوننىڭ لىكسىيە ئۆمىكى پارتىيە نۆۋەتلىك مەركىزىي كومىتېتى ئومۇمىي يىغىنى ۋە ئاپتونۇم رايونلۇق پارتكوم نۆۋەتلىك كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ھەققىدە دوكلات بەردى ئاۋامنىڭ كەڭ زىمىندا ئىگىلىك تىكلىشى بىپايان جەنۇبىي شىنجاڭ زېمىنىدا كىشىلىك ھاياتنىڭ گۈزەل كارتىنىسىنى يارىتايلى ئاپتونوم رايونلۇق ئىنتىزام تەكشۈرۈش كومىتېتى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە جاڭ چۈنشيەن شۇجىنىڭ مۇھىم سۆزىنىڭ روھىنى يەتكۈزدى ئاپتونوم رايونلوق خەلق قۇرۇلتىيى دائىمىي كومىتېتى پارتگۇرۇپپىسى مەركىزى ئۆگۈنۈش گۇرۇپپىسى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە ئاپتونوم رايونلوق پارتكوم خەلق قۇرۇلتىيى خىزمىتى يىغىنىنىڭ روھىنى ئۆگەندى مىسىر زوڭتوڭ سايلىمىدا مۇسۇلمان بۇرادەرلەر جەمئىيىتىگە تەۋە ھەققانىيەت پارتىيېسىنىڭ رەئىسى مورشى بىلەن سابىق زوڭلى سەفىقنىڭ بېلەتكە ئېرىشىش نىسبىتى ئالدىنقى ئىككى ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر يىلى تەخمىنەن مىڭدىن مىڭغىچە ئەيدىز بىمارى ئۆلۈپ كېتىدىكەن كورىيە ئارمىيە دائىرىلىرى چاۋشيەن يادرو سىنىقى ئېلىپ بېرىشقا تەييارلىنىپ بولدى دېدى قىزىق نوقتا خەۋەرلىرى پايتەخت پىداگوگىكا ئۇنىۋېرسىتېتىيىلدىن بۇيان ئىلگىرى كېيىن بولۇپ تۈركۈمدە خوتەندىن كەلگەن شىنجاڭلىق كۇرسانتنى تەربىيەلىدى مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى پايتەخت پىداگوگىكا ئۇنىۋېرسىتېتى مەملىكىتىمىزدىكى شىنجاڭلىق ئىشقا ئورۇنلىشالمغان ئىستودېنتلارنى تۇنجى تۈركۈمدە تەربىيەلىگەن ئالىي بىلىم يۇرتى بولۇش سۈپىتى بىلەن يىلدىن بۇيان ئىلگىرى كېيىن بولۇپ تۈركۈمدە خوتەندىن كەلگەن شىنجاڭلىق كۇرسانتنى تەربىيەلىدى يىلدىن بۇيان پايتەخت پىداگوگىكا ئۇنىۋېرسىتېتى پارتكومىنىڭ شۇجىسى جاڭ شۆ يادرولىقىدىكى ئوقۇتقۇچىلار ۋە ئىشچى خىزمەتچىلەرنىڭ چىن مېھرىمۇھەببىتى شىنجاڭلىق ھەر بىر كۇرسانتنى قاتتىق تەسىرلەندۈردى كۆپچىلىك جاڭ شۆنى شۆ ئانا دەپ ئاتىشىدىغان بولدى تۆۋەندە مۇخبىرىمىزنىڭ بېيجىڭلىق شۆ ئانىنىڭ خوتەنگە بېرىپ ئوقۇغۇچىلىرىنى يوقلىشى توغرىسىدا يازغان يۈرۈشلۈك خەۋىرىنىڭ مىڭ چاقىرىم يىراقلىقتىكى ئوقۇتقۇچى ئوقۇغۇچىلارنىڭ مېھرىمۇھەببىتىدىگەن ئالدىنقى قىسمىنى ئاڭلىتىمىز دۆلەت بايرىمى ھارپىسىدا پايتەخت پىداگوگىكا ئۇنىۋېرسىتېتى پارتكومىنىڭ شۇجىسى جاڭ شۆ ئالاھىدە بېيجىڭدىن خوتەنگە كېلىپ ئوقۇغۇچىلىرىنى يوقلىدى ئۇنىڭ بۇ ئوقۇغۇچىلىرى ئىلگىرى پايتەخت پىداگوگىكا ئۇنىۋېرسىتېتىدا تەربىيەلەنگەن بولۇپ ھازىر ئۇلار ئۆز خىزمەت ئورۇنلىرىدا ئىشلەۋاتقانىدى ئىلگىرى دائىم كۆرۈشۈپ ياخشى ئۆتكەن كۆرۈنۈشلەر كۆپچىلىكنىڭ كۆز ئالدىدا نامايان بولدى مەملىكىتىمىزدىكى شىنجاڭلىق ئىشقا ئورۇنلىشالمغان ئىستودېنتلارنى تۇنجى تۈركۈمدە تەربىيەلەش ۋەزىپىسىنى ئۈستىگە ئالغان ئالىي بىلىم يۇرتى پايتەخت پىداگوگىكا ئۇنىۋېرسىتېتىدا پارتكوم شۇجىسى جاڭ شۆ باشچىلىقىدىكى مۇلازىمەت گۇرۇپپىسى قۇرۇلدى شىنجاڭلىق كۇرسانتلارغا ياخشى تۇرمۇش مۇھىتى ھازىرلاپ بىرىش ئۈچۈن مەكتەپ سىرتقا تىجارەت قىلىشقا ئېچىۋېتىلگەن مېھمانساراينىڭ ئىككى قەۋىتىنى ئىجارىگە ئېلىپ مۇسۇلمانچە ئاشخانىنى قايتىدىن زىننەتلىدى يىلى مەكتەپ يەنە بېيجىڭ شىنجاڭ ئىنىستىتوتىنى مەخسۇس قۇرۇپ چىقىپ شىنجاڭلىق كۇرسانتلارنىڭ خاتىرجەم ئوقۇشى ۋە تۇرمۇشىغا ھەقىقى تۈردە كاپالەتلىك قىلدى شىنجاڭ سىنىپىنىڭ مەسئۇلى ۋە ھەر بىر ئوقۇتقۇچىغا نىسبەتەن مەكتەپ قاتتىق تەلەپ قويدى شۇجى جاڭ شۆنىڭ يىتەكچىلىكىدە پايتەخت پىداگوگىكا ئۇنىۋېرسىتېتى يۇقىرىدىن تۆۋەنگىچە شىنجاڭ سىنىپىنىڭ كۇرسانتلىرىغا كۆڭۈل بۆلدى نۇرغۇن ئىنىستىتوتنىڭ رەھبەرلىرى ۋە ئوقۇتقۇچىلار دائىم ياتاق ۋە دەرسخانىلارغا بېرىپ كۇرسانتلارنىڭ تۇرمۇش ۋە ئۆگىنىش ئەھۋالىنى ئىگىلىدى ھەر قېتىملىق يېڭى ئوقۇش مەۋسۈمى باشلانغاندا ۋە تەتىل قويۇپ بەرگەندە مەكتەپنىڭ ئاساسلىق رەھبەرلىرى ھەر بىر كۇرسانتنى ۋوگزالغا چىقىپ كۈتۈۋالدى ۋە ئۇزىتىپ قويدى كۇرسانتلار بۇنىڭدىن بىر خىل ئىللىقلىقنى ھىس قىلدى كۇرسانتلار جاڭ شۆنى شۆ ئانا دەپ ئاتايدىغان بولدى نۇرغۇن ساۋاقداشلارنىڭ بۇ قېتىم كېچەكۈندۈز سېغىنغان شۆ ئانا بىلەن كۆرۈشۈشى ئۇلارغا نىسبەتەن بىر چۈشتەك تۇيۇلدى خوتەن شەھەرلىك تېلېۋىزىيە ئىستانسىسىدا ئىشلەۋاتقان مەڭنىسا ئابابەكرى شۆ ئانانى كۆرۈپ كۆڭلىنى ئىزھار قىلدى ئۇ مۇنداق دېدى سىز بىلەن بىللە بولسام ئىككىنچى يۇرتۇم ئېسىمگە كېلىدىكەن پايتەخت پىداگوگىكا ئۇنىۋېرسىتېتىنى بەك سېغىندىم يىلى ئايدىن بۇيان شىنجاڭلىق كۇرسانت پايتەخت پىداگوگىكا ئۇنىۋېرسىتېتىدا رادىئوتېلېۋىزىيە خەنزۇ تىلى ئەدەبىياتى تىببى ئىلىم قاتارلىق كەسپلەر بويىچە تەربىيەلەندى تەربىيەلىنىش ئارقىلىق ئۇلارنىڭ خەنزۇ تىلى ۋە كەسپىي ئىقتىدارى كۆرۈنەرلىك ئۆستى بۇنىڭ ئىچىدە خەنزۇچە سەۋىيە سىناش دەرىجە ئىمتىھانىدىن ئۆتۈش نىسبىتى تىن ئاشتى كۇرسانتلار يۇرتىغا قايتقاندىن كىيىن كۆپىنچىسى خوتەن ۋىلايىتىدىكى ئوتتۇراباشلانغۇچ مەكتەپلەردە ناھىيە يېزىلاردىكى رادىئوتېلېۋىزىيە
