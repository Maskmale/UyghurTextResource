ئەسلىمە ئۈنۋېرسال رادىيو سەنئەت رادىيو كەسىپ ئىستىلى تاللاپ ئاڭلاش ئۇيغۇرچە ئۈگۈنۈش خەنزۇچە ئۈگۈنۈش ئىستانسىمىز ھەققىدە ئاپتونۇم رايوننىڭ لىكسىيە ئۆمىكى پارتىيە نۆۋەتلىك مەركىزىي كومىتېتى ئومۇمىي يىغىنى ۋە ئاپتونۇم رايونلۇق پارتكوم نۆۋەتلىك كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ھەققىدە دوكلات بەردى ئاۋامنىڭ كەڭ زىمىندا ئىگىلىك تىكلىشى بىپايان جەنۇبىي شىنجاڭ زېمىنىدا كىشىلىك ھاياتنىڭ گۈزەل كارتىنىسىنى يارىتايلى ئاپتونوم رايونلۇق ئىنتىزام تەكشۈرۈش كومىتېتى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە جاڭ چۈنشيەن شۇجىنىڭ مۇھىم سۆزىنىڭ روھىنى يەتكۈزدى ئاپتونوم رايونلوق خەلق قۇرۇلتىيى دائىمىي كومىتېتى پارتگۇرۇپپىسى مەركىزى ئۆگۈنۈش گۇرۇپپىسى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە ئاپتونوم رايونلوق پارتكوم خەلق قۇرۇلتىيى خىزمىتى يىغىنىنىڭ روھىنى ئۆگەندى مىسىر زوڭتوڭ سايلىمىدا مۇسۇلمان بۇرادەرلەر جەمئىيىتىگە تەۋە ھەققانىيەت پارتىيېسىنىڭ رەئىسى مورشى بىلەن سابىق زوڭلى سەفىقنىڭ بېلەتكە ئېرىشىش نىسبىتى ئالدىنقى ئىككى ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر يىلى تەخمىنەن مىڭدىن مىڭغىچە ئەيدىز بىمارى ئۆلۈپ كېتىدىكەن كورىيە ئارمىيە دائىرىلىرى چاۋشيەن يادرو سىنىقى ئېلىپ بېرىشقا تەييارلىنىپ بولدى دېدى قىزىق نوقتا خەۋەرلىرى ئاپتونوم رايونىمىزنىڭ جازانى يىنىكلىتىش شەرتلىك قويۇپ بىرىش جازانى ۋاقتىنچە سىرتتا ئىجرا قىلىش دېلو خىزمىتىنىڭ ئەھۋالىنى تۇنجى قېتىم ئېلان قىلدى مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى ئىستانسىمىز خەۋىرى ئاپتونوم رايونلۇق يۇقىرى خەلق سوت مەھكىمىسى مۇشۇ ئاينىڭ كۈنى ئاپتونوم رايونىمىزنىڭ جازانى يىنىكلىتىش شەرتلىك قويۇپ بىرىش جازانى ۋاقتىنچە سىرتتا ئىجرا قىلىش دېلو خىزمىتىنىڭ ئەھۋالىنى تۇنجى قېتىم ئېلان قىلىپ جازانى يىنىكلىتىش شەرتلىك قويۇپ بىرىش جازانى ۋاقتىنچە سىرتتا ئىجرا قىلىشقا ئائىت تىپىك دېلونى ئېلان قىلدى بۇ تىپىك دېلو ئىچىدىكى شەرتلىك قويۇپ بىرىش دېلوسىنى مىسالغا ئالساق جىنايەت گۇماندارى جەي يۈپىڭ يىلى خىيانەتچىلىك جىنايىتى ۋە ئومۇمنىڭ پۇلىنى يۆتكەپ ئىشلىتىۋېلىش جىنايىتى بىلەن ئۇنىڭغا يىللىق مۇددەتلىك قاماق جازاسى بىرىلگەنىدى بۇنى ئىجرا قىلىدىغان ئورۇن ئاپتونوم رايونلۇق ئاياللار تۈرمىسى يىلى ئاينىڭ كۈنى جەي يۈپىڭنى شەرتلىك قويۇپ بىرىش پىكىرنامىسىنى ئوتتۇرىغا قويدى ئۈرۈمچى شەھەرلىك ئوتتۇرا خەلق سوت مەھكىمىسى بۇ دېلوغا قاراپ چىقىپ بىر تەرەپ قىلىشتا جازانى يىنىكلىتىش شەرتلىك قويۇپ بىرىشكە بولمايدىغانلىقى توغرىسىدا كېسىم چىقاردى ئاپتونوم رايونلۇق يۇقىرى خەلق سوت مەھكىمىسى سوت ئىشلىرىنى نازارەت قىلىش كوللىگىيەسىنىڭ باشلىقى سۈي ۋېنجۈ مۇنداق دېدى بىز بۇ دېلونى قاراپ چىقىپ بىر تەرەپ قىلىش داۋامىدا جىنايەتچى زىياننى تەشەببۇسكارلىق بىلەن تۆلەش ياكى چېگرا سىرتىدىكى ھارام پۇلنى قايتۇرۇپ كىلىش جەھەتتە ئاكتىپ ماسلاشمىدى ئۇنىڭ ئۆزگىرىش ئىپادىسى يوق دەپ قاراپ بۇ جىنايەتچىنى شەرتلىك قويۇپ بىرىشكە بولمايدۇ دەپ ھىساپلىدۇق بۇلتۇر ئاپتونوم رايونىمىزدىكى سوت مەھكىمىلىرى ۋەزىپە ئۈستىدە جىنايەت ئۆتكۈزۈش پۇل مۇئامىلە باشقۇرۇش تەرتىبىگە بۇزغۇنچىلىق قىلىش ۋە پۇل مۇئامىلە ئىشلىرىدا قاقتىسوقتى ۋە ئالدامچىلىق قىلىش جىنايىتى قارا جەمىئىيەت تۈسىنى ئالغان تەشكىللىك جىنايەت ئۆتكۈزۈش قاتارلىق خىل تۈردىكى جىنايى قىلمىشلارنىڭ جازاسىنى يىنىكلىتىش شەرتلىك قويۇپ بىرىشىنى قاتتىق كونترول قىلىپ جازا ئىجرا قىلىشتا ھەممە كىشىنىڭ باراۋەر بولۇشىغا ھەقىقى تۈردە كاپالەتلىك قىلدى ئاپتونوم رايونلۇق يۇقىرى خەلق سوت مەھكىمىسىنىڭ ئاخبارات باياناتچىسى يۈ خۈيتاڭ مۇنداق دېدى يىلى ئاپتونوم رايونىمىزدىكى سوت مەھكىمىلىرى خىل جىنايى قىلمىشتىكى جازانى يىنىكلىتىش دېلوسىدىن نى قاراپ چىقىپ بىر تەرەپ قىلدى بۇنىڭ ئىچىدە ۋەزىپە ئۈستىدە جىنايەت ئۆتكۈزۈش جىنايىتى بويىچە جازانى يىنىكلىتىش دېلوسى قا پۇل مۇئامىلە باشقۇرۇش تەرتىبىگە بۇزغۇنچىلىق قىلىش جىنايىتى بويىچە جازانى يىنىكلىتىش دېلوسى گە يىتىپ قارا جەمىئىيەت تۈسىنى ئالغان جىنايى قىلمىش بويىچە جازانى يىنىكلىتىش دېلوسى مۇشۇ خىل دېلودىكى قاماقتىكى كىشىلەر ئومۇمى سانىنىڭ ىنى ئىگىلىدى بۇ سان شۇ مەزگىلدىكى لىك باشقا جىنايى قىلمىش سادىر قىلغانلار سانىدىن خېلىلا تۆۋەن نۇقتىلىق ۋە نازۇك دېلولارنى ئەنگە ئېلىش تەكشۈرۈپ تەستىقلاش ئارقىلىق جازانىڭ ۋاقتىنچە سىرتتا ئىجرا قىلىنىشى ئۈنۈملۈك قېلىپلاشتۇرۇلدى بۇنىڭدىن كىيىن ئاپتونوم رايونىمىزدىكى ھەر دەرىجىلىك سوت مەھكىمىلىرى بەلگىلىمىدىكى تەلەپ بويىچە دېلو تۇرغۇزۇلۇپ كۈن ئىچىدە جازانى يىنىكلىتىش شەرتلىك قويۇپ بىرىش جازا ئۇچۇرلىرى شۇنىڭدەك جازانى يىنىكلىتىش شەرتلىك قويۇپ بىرىش تەكلىپنامە قاتارلىق ئالاقىدار ماتىرياللارنى قانۇن بويىچە تور ئارقىلىق پۈتكۈل جەمىئىيەتكە ئېلان قىلىپ ھەر قايسى ساھەدىكىلەرنىڭ نازارەتچىلىكىنى قوبۇل قىلىدۇ شۇنىڭ بىلەن بىر ۋاقىتتا خىل جىنايەتكە چېتىلىدىغان جازانى يىنىكلىتىش شەرتلىك قويۇپ بىرىش دېلولىرىنى قاراپ چىقىپ بىر تەرەپ قىلىشتا ئاشكارا سوت ئېچىپ قاراپ چىقىپ بىر تەرەپ قىلىپ جەمىئىيەتتە ئىنكاس كۈچلۈك بولغان ھوقۇقى پۇلى بارلارغا جازا ھۆكۈمى ئېلان قىلىنغاندىن كىيىن ئۇلارنىڭ جازاسىنى يىنىكلىتىش تېز بولۇش شەرتلىك قويۇپ بىرىش ۋە جازانى ۋاقتىنچە سىرتتا ئىجرا قىلىش نىسبىتى
