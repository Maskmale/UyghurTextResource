ئەسلىمە ئۈنۋېرسال رادىيو سەنئەت رادىيو كەسىپ ئىستىلى تاللاپ ئاڭلاش ئۇيغۇرچە ئۈگۈنۈش خەنزۇچە ئۈگۈنۈش ئىستانسىمىز ھەققىدە ئاپتونۇم رايوننىڭ لىكسىيە ئۆمىكى پارتىيە نۆۋەتلىك مەركىزىي كومىتېتى ئومۇمىي يىغىنى ۋە ئاپتونۇم رايونلۇق پارتكوم نۆۋەتلىك كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ھەققىدە دوكلات بەردى ئاۋامنىڭ كەڭ زىمىندا ئىگىلىك تىكلىشى بىپايان جەنۇبىي شىنجاڭ زېمىنىدا كىشىلىك ھاياتنىڭ گۈزەل كارتىنىسىنى يارىتايلى ئاپتونوم رايونلۇق ئىنتىزام تەكشۈرۈش كومىتېتى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە جاڭ چۈنشيەن شۇجىنىڭ مۇھىم سۆزىنىڭ روھىنى يەتكۈزدى ئاپتونوم رايونلوق خەلق قۇرۇلتىيى دائىمىي كومىتېتى پارتگۇرۇپپىسى مەركىزى ئۆگۈنۈش گۇرۇپپىسى ئاپتونوم رايونلۇق نۆۋەتلىك پارتىيە كومىتېتى ئومۇمىي كېڭەيتىلگەن يىغىنى ۋە ئاپتونوم رايونلوق پارتكوم خەلق قۇرۇلتىيى خىزمىتى يىغىنىنىڭ روھىنى ئۆگەندى مىسىر زوڭتوڭ سايلىمىدا مۇسۇلمان بۇرادەرلەر جەمئىيىتىگە تەۋە ھەققانىيەت پارتىيېسىنىڭ رەئىسى مورشى بىلەن سابىق زوڭلى سەفىقنىڭ بېلەتكە ئېرىشىش نىسبىتى ئالدىنقى ئىككى ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر يىلى تەخمىنەن مىڭدىن مىڭغىچە ئەيدىز بىمارى ئۆلۈپ كېتىدىكەن كورىيە ئارمىيە دائىرىلىرى چاۋشيەن يادرو سىنىقى ئېلىپ بېرىشقا تەييارلىنىپ بولدى دېدى قىزىق نوقتا خەۋەرلىرى تارىختىكى بۈگۈن مەنبەسىجۇڭگو ئۇيغۇرچە رادىيو تورى يىلى بۈگۈنكى كۈندە ياشلىق ئوقۇغۇچى ئۆگىنىش بېسىمىغا چىدىماي ئاپىسىنى ئۆلتۈرۈۋەتكەن جېجياڭ شەھەرلىك ئوتتۇرا مەكتەپنىڭ تولۇق يىللىق ئوقۇغۇچىسى تولۇق يىللىقتا ئۇ سىنىپنىڭ كەينىدىن سانىغاندا ئورۇندا تۇراتتى تىرىشىش ئارقىلىق مەۋسۈمدە ئالدىنقى ئورۇنغا ئىلگىرلىدى بۇنى كۆرگەن ئاپىسى خوشاللىقىدىن ئوغلىنىڭ بۇندىن كېيىنكى ئىمتىھانلاردا چوقۇم سىنىپتا ئوننىڭ ئىچىدە مېڭىشىنى ئۈمىد قىلدى تىراگىدىيە مۇشۇ ۋاقىتتىن باشلاندى ئۇ ھەر قايسى ئىمتىھانلاردا ئاپىسىنىڭ كۈتكەن يېرىدىن چىقالمىدى ئۇ پۇتبۇل ئويناشنى ياخشى كۆرەتتى ئاپىسى بۇنىڭغا يولغا قويمىدى يىلى ئاينىڭ كۈنى ئوغلى مەكتەپتىن كېلىپ تېلېۋىزور كۆرۈۋاتاتتى ئاپىسى بۇ قېتىملىق ئىمتىھاندا ئوغلىنىڭ نىڭ ئىچىگە كىرىشنى ئېيتتى بىراق ئوغلىنىڭ ئىشەنچىسىنىڭ يوقلىقى ئېيتىپ ئاپىسى بىلەن تاكاللاشتى ئوغلى ئۈمىدسىزلىك ئىچىدە بولقا بىلەن ئاپىسىنىڭ مېڭىسىگە ئۇرۇپ ئاپىسىنى ئۆلتۈرۈۋەتكەن يىلى بۈگۈنكى كۈندە ئامېرىكىنىڭ ئىككى ئايروپىلانى سوقۇلۇپ كېتىپ ۋودۇرۇد بومبىسى ئىسپانىيەنىڭ ئاسمىنىدا يوقاپ كەتكەن بۇ ئىشقا ھەقىقەتەنمۇ كىشىنىڭ ئەقلى يەتمەيدۇ ئايروپىلان خادىمىنىڭ سەككىزى قازا قىلغان بىر ۋودۇرۇد بومبىسى شىمالىي ئاتلانتىك ئوكيانىغا چۈشۈپ كېتىپ يوقاپ كەتكەن بەش بۇرجەكلىك بىناسى ئادەم ئەۋەتىپ ئىزدىەەن بولسىمۇ ھازىرغىچە تېپىلمىغان يىلى بۈگۈنكى كۈندە دۇنيا ئۇرۇشىدا سوۋىت ئىتتىپاقى ۋارشاۋانى ئىشغال قىلىۋالغان يىلى بۈگۈنكى كۈندە سەي يۈەنپېي بېيجىڭ ئۇنۋېرسىتېتىنىڭ مۇدىرلىقىدىن ئىستىپا بەرگەن سەي يۈەنپېي يىلى ئاينىڭ كۈنى دۇنياغا كەلگەن يېقىنقى
