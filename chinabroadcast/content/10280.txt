ئورۇننى ئىگىلىگەن يىلىغا بارغاندا ئېلىمىز ئىجتىمائىي ئىستېمال بۇيۇملىرىنىڭ پارچە سېتىلىش ئومۇمىي سوممىسى تىرلىيون ئامېرىكا دوللىرىغا يېتىدىكەن ھىندىستاندا ئىسسىق ھاۋا ئادەمنىڭ بېشىغا چىقتى بىرمىدا ھەر مەھەممەت مەنبەسى شىنجاڭ خەلق رادىيو ئىستانسىسى باي ناھىيە باي بازىرى قۇتاد شەھەرچىسىدىكى ئوسمان مەھەممەت ئۆزىنىڭ ئاتا مىراسى بولغان ناۋايلىق ھۈنىرىگە تايىنىپ ئاپتونۇم رايونىمىز ئىچىدە كۆپ قېتىم چوڭ شان شەرەپكە ئېرىشىپ تەشكىللنىڭ مۇئەييەلەشتۈرۈشىگە ئەلجامائەتنىڭ ئېتىراپ قىلىشىغا ئېرىشتى ئوسمان مەھەممەت يىلى ئايدا باي ناھىيە بازىرىدا بىر ناۋاي ئائىلىسىدە دۇنياغا كەلگەن ئاران تۇلۇقسىز ئوتتۇرا مەكتەپكىچە ئوقۇغان ئوسمان مەھەممەت دادىسىدىن ۋە تاغىسى قادىر غۇپۇردىن ناۋايلىقنى ئۆگەنگەن ئوسمان مەمەتنىڭ ئۇستىسى ئەينى ۋاقىتتا باي ناھىيەسىدىكى ئۇستا ناۋايلارنىڭ بىرى بۇلۇپ يىلى مەخسۇس نان يېقىپ ماۋجۇشىغا سوۋغا قىغانىكەن ئوسمان مەھەممەت تىرىشچان جاپادىن قورقمايدىغان بولغاچقا ئۇستىسى ئۇنىڭغا ئامراق بۇلۇپ ناۋايلىق ھۈنىرىنىڭ بىرىنىمۇ قويماي ئوسمان مەمەتكە ئۈگۈتۈپ قۇيغانىكەن يىلى ئۇستىسى ئۇنى ھۈنەردىن ئايرىغان ھەم دادىسىمۇ كېسەللىك سەۋەبىدىن ھۈنىرىنى ئۇنىڭغا ئۆتكۈزۈپ بەرگەن ئاق كۆڭۈل ساددا كەمتەر ۋە خۇش پىئىل ئوسمان مەھەممەت بىر تەرەپتىن باشقىلارنىڭ نېنىنى يېقىپ بەرسە يەنە بىر تەرەپتىن بازارغا نان يېقىپ سېتىپ ئائىلىسىنىڭ تۇرمۇشىنى قامدىغان ۋە دادىسىنى داۋالاتقان ئۇ ھۇنىرىنىڭ ياخشىلىنىشىغا ئەگىشىپ نان تۈرىنىمۇ تۈرگە يەتكۈزگەن ئاددىي نان يېقىشتىن ھەر خىل نەقىشلىك ئوزۇقلۇق تەركىۋى يوقىرى نان يېقىشقا يۈزلەنگەن ئۈرۈمچىگە بېرىپ ئابلاجان نانلىرىنىڭ تەييارلىنىش ئەھۋالىنى كۆرگەن ۋە تەتقىق قىلغان ئۇنىڭ ئۇزۇن يىللىق جاپالىق مېھنىتى ئۆز كەسپىگە بولغان ئوتتەك قىزغىنلىقى ۋە ئەستايىدىللىقى باش چۆكۈرۈپ جامائەت ئۈچۈن خىزمەت قىلىشى ئۇنى يىلمۇيىل چېنىقتۇرۇپ پىشقان كەسىپ ئەھلىگە كۆپكە ماھىر ناۋايغا ئايلاندۇردى ئۇنىڭ ھۈنىرىنىڭ كامالەتكە يىتىشىگە ئەگىشىپ ئۇنىڭ داڭقى يىراقيېقىنغا تارقىلىشقا باشلىدى يىلى كورلىدىكى كەسىپ ئەھلىلىرىنىڭ تەكلىپ قىلىشى بىلەن لوپنۇردا ئۆتكۈزۈلگەن كورلا بويىچە ناۋايلارنىڭ نەق مەيدان مۇسابىقىسىدە دەرىجىلىك مۇكاپاتقا ئېرىشتى ۋە ئۆزىگە بولغان ئىشەنچىسى ئاشتى يىلى ئايدا ئۈرۈمچىدە ئۆتكۈزۈلگەن ئىككىنچى نۆۋەتلىك جۇڭگو مۇسۇلمانچە يېمەكئېچمەك كۆرگەزمىسىدە ئۇنىڭ نانلىرى دانىسىنىڭ چوڭلىقىدىئامېتىرى مېتىر شەكلىنىڭ كۆركەملىكى تەمىنىڭ ياخشىلىقى ئوزۇقلۇق قىممىتىنىڭ يوقىرىلىقى بىلەن شىنجاڭ بويىچە چىمپىيون بولۇپ ئۈرۈمچىنى زىلزىلگە سالدى بۇ يىل ئاينىڭ كۈنى قۇمۇلدا ئۆتكۈزۈلگەن يىمەكئىچمەك يەرمەنكىسى نان كۆرگەزمىسىگە قاتنىشىپ دەرىجىلىك مۇكاپاتقا ئېرىشتى ئۇنىڭ نامىنىڭ چىقىشى بىلەن قەشقەر خوتەن قۇمۇل قاتارلىق جايلاردىكى كەسپىي مەكتەپلەر ناۋاي تەربىيلەش مەكتەپلىرى ناۋايلىق دۇكانلىرى ئۇنى دەرس سۆزلەپ بىرىشكە ناۋايلىق ھۈنىرىنى ئۈگىتىپ شاگرت تەربىيىلەپ بىرىشكە تەكلىپ قىلدى ئوسمان مەمەت ھازىرغىچە قەشقەر خوتەن كورلا قۇمۇل تۇرپان قاتارلىق جايلاردا دىن ئارتۇق شاگرت تەربىيىلىدى ۋە كىرىمىنى كۆپەيتتى ھازىر ئوسمان مەھەممەت باي بازىرىنىڭ قۇتاد شەھەرچىسىدە ئابلىز ئابلىكىم بىلەن بىرلىشىپ زامانغا لايىق ئەلھۇزۇر نانلىرى ناۋايخانىسىنى ئاچقان بولۇپ بۇ ناۋايخانىغا جۇڭگو كەشپىياتچىلار جەمىيىتىنىڭ پەخىرى ئەزاسى ماشىنىسازلىق ئالىي ئىنژىنېرى شىنجاڭ ئالاھىدە يېمەكئېچمەكلىرىنى ئىشلەپچىقىرىش ئۈسكۈنىلىرى تەتقىقات كومتىتىنىڭ مۇئاۋىن رەئىسى شىنجاڭ يېزا ئىگىلىك ئونۋېرسىتېتىنىڭ ماگىستېر ئاسپىراتى مىللىتىمىزدىن چىققان جۇڭگو ياشئۆسمۈرلەر پەنتېخنىكىدا يېڭىلىق يارىتىش مۇكاپاتىنىڭ تۇنجى ساھىبى ئۈرۈمچى كائىنات شىركىتىنىڭ قۇرغۇچىسى ئېلىجان رۇسۇل باشچىلىقىدا ياسىلىپ چىققان پاتنېتلىق مەھسۇلات مۇھىت ئاسراش ئېنىرگىيە تىجەش تىپىدىكى ئىستۈتەكسىز تونۇر سەپلەنگەن بولۇپ تەبىئىي گاز
